15NotTEC
16LoanBrokerSet::preflight(PreflightContext const& ctx)
17{
18 using namespace Lending;
19
20 auto const& tx = ctx.tx;
21 if (auto const data = tx[~sfData]; data && !data->empty() &&
22 !validDataLength(tx[~sfData], maxDataPayloadLength))
23 return temINVALID;
24 if (!validNumericRange(tx[~sfManagementFeeRate], maxManagementFeeRate))
25 return temINVALID;
26 if (!validNumericRange(tx[~sfCoverRateMinimum], maxCoverRate))
27 return temINVALID;
28 if (!validNumericRange(tx[~sfCoverRateLiquidation], maxCoverRate))
29 return temINVALID;
30 if (!validNumericRange(
31 tx[~sfDebtMaximum], Number(maxMPTokenAmount), Number(0)))
32 return temINVALID;
33
34 if (tx.isFieldPresent(sfLoanBrokerID))
35 {
36 // Fixed fields can not be specified if we're modifying an existing
37 // LoanBroker Object
38 if (tx.isFieldPresent(sfManagementFeeRate) ||
39 tx.isFieldPresent(sfCoverRateMinimum) ||
40 tx.isFieldPresent(sfCoverRateLiquidation))
41 return temINVALID;
42
43 if (tx[sfLoanBrokerID] == beast::zero)
44 return temINVALID;
45 }
46
47 if (auto const vaultID = tx.at(~sfVaultID))
48 {
49 if (*vaultID == beast::zero)
50 return temINVALID;
51 }
52
53 {
54 auto const minimumZero = tx[~sfCoverRateMinimum].value_or(0) == 0;
55 auto const liquidationZero =
56 tx[~sfCoverRateLiquidation].value_or(0) == 0;
57 // Both must be zero or non-zero.
58 if (minimumZero != liquidationZero)
59 {
60 return temINVALID;
61 }
62 }
63
64 return tesSUCCESS;
65}

76TER
77LoanBrokerSet::preclaim(PreclaimContext const& ctx)
78{
79 auto const& tx = ctx.tx;
80
81 auto const account = tx[sfAccount];
82 auto const vaultID = tx[sfVaultID];
83
84 auto const sleVault = ctx.view.read(keylet::vault(vaultID));
85 if (!sleVault)
86 {
87 JLOG(ctx.j.warn()) << "Vault does not exist.";
88 return tecNO_ENTRY;
89 }
90 Asset const asset = sleVault->at(sfAsset);
91
92 if (account != sleVault->at(sfOwner))
93 {
94 JLOG(ctx.j.warn()) << "Account is not the owner of the Vault.";
95 return tecNO_PERMISSION;
96 }
97
98 if (auto const brokerID = tx[~sfLoanBrokerID])
99 {
100 // Updating an existing Broker
101
102 auto const sleBroker = ctx.view.read(keylet::loanbroker(*brokerID));
103 if (!sleBroker)
104 {
105 JLOG(ctx.j.warn()) << "LoanBroker does not exist.";
106 return tecNO_ENTRY;
107 }
108 if (vaultID != sleBroker->at(sfVaultID))
109 {
110 JLOG(ctx.j.warn())
111 << "Can not change VaultID on an existing LoanBroker.";
112 return tecNO_PERMISSION;
113 }
114 if (account != sleBroker->at(sfOwner))
115 {
116 JLOG(ctx.j.warn()) << "Account is not the owner of the LoanBroker.";
117 return tecNO_PERMISSION;
118 }
119
120 if (auto const debtMax = tx[~sfDebtMaximum])
121 {
122 // Can't reduce the debt maximum below the current total debt
123 auto const currentDebtTotal = sleBroker->at(sfDebtTotal);
124 if (*debtMax != 0 && *debtMax < currentDebtTotal)
125 {
126 JLOG(ctx.j.warn())
127 << "Cannot reduce DebtMaximum below current DebtTotal.";
128 return tecLIMIT_EXCEEDED;
129 }
130 }
131 }
132 else
133 {
134 if (auto const ter = canAddHolding(ctx.view, asset))
135 return ter;
136
137 if (auto const ter = checkFrozen(
138 ctx.view, sleVault->at(sfAccount), sleVault->at(sfAsset)))
139 {
140 JLOG(ctx.j.warn()) << "Vault pseudo-account is frozen.";
141 return ter;
142 }
143 }
144
145 // Check that relevant values can be represented as the vault asset
146 // type. This is mostly only relevant for integral (non-IOU) types
147 for (auto const& field : getValueFields())
148 {
149 if (auto const value = tx[field];
150 value && STAmount{asset, *value} != *value)
151 {
152 JLOG(ctx.j.warn()) << field.f->getName() << " (" << *value
153 << ") can not be represented as a(n) "
154 << to_string(asset) << ".";
155 return tecPRECISION_LOSS;
156 }
157 }
158
159 return tesSUCCESS;
160}

162TER
163LoanBrokerSet::doApply()
164{
165 auto const& tx = ctx_.tx;
166 auto& view = ctx_.view();
167
168 if (auto const brokerID = tx[~sfLoanBrokerID])
169 {
170 // Modify an existing LoanBroker
171 auto broker = view.peek(keylet::loanbroker(*brokerID));
172 if (!broker)
173 {
174 // This should be impossible
175 // LCOV_EXCL_START
176 JLOG(j_.fatal()) << "LoanBroker does not exist.";
177 return tefBAD_LEDGER;
178 // LCOV_EXCL_STOP
179 }
180
181 auto const vault = view.read(keylet::vault(broker->at(sfVaultID)));
182 if (!vault)
183 return tecINTERNAL; // LCOV_EXCL_LINE
184
185 auto const vaultAsset = vault->at(sfAsset);
186
187 if (auto const data = tx[~sfData])
188 broker->at(sfData) = *data;
189 if (auto const debtMax = tx[~sfDebtMaximum])
190 broker->at(sfDebtMaximum) = *debtMax;
191
192 view.update(broker);
193
194 associateAsset(*broker, vaultAsset);
195 }
196 else
197 {
198 // Create a new LoanBroker pointing back to the given Vault
199 auto const vaultID = tx[sfVaultID];
200 auto const sleVault = view.read(keylet::vault(vaultID));
201 if (!sleVault)
202 {
203 // This should be impossible
204 // LCOV_EXCL_START
205 JLOG(j_.fatal()) << "Vault does not exist.";
206 return tefBAD_LEDGER;
207 // LCOV_EXCL_STOP
208 }
209 auto const vaultPseudoID = sleVault->at(sfAccount);
210 auto const vaultAsset = sleVault->at(sfAsset);
211 auto const sequence = tx.getSeqValue();
212
213 auto owner = view.peek(keylet::account(account_));
214 if (!owner)
215 {
216 // This should be impossible
217 // LCOV_EXCL_START
218 JLOG(j_.fatal()) << "Account does not exist.";
219 return tefBAD_LEDGER;
220 // LCOV_EXCL_STOP
221 }
222 auto broker =
223 std::make_shared<SLE>(keylet::loanbroker(account_, sequence));
224
225 if (auto const ter = dirLink(view, account_, broker))
226 return ter; // LCOV_EXCL_LINE
227 if (auto const ter = dirLink(view, vaultPseudoID, broker, sfVaultNode))
228 return ter; // LCOV_EXCL_LINE
229
230 // Increases the owner count by two: one for the LoanBroker object, and
231 // one for the pseudo-account.
232 adjustOwnerCount(view, owner, 2, j_);
233 auto const ownerCount = owner->at(sfOwnerCount);
234 if (mPriorBalance < view.fees().accountReserve(ownerCount))
235 return tecINSUFFICIENT_RESERVE;
236
237 auto maybePseudo =
238 createPseudoAccount(view, broker->key(), sfLoanBrokerID);
239 if (!maybePseudo)
240 return maybePseudo.error(); // LCOV_EXCL_LINE
241 auto& pseudo = *maybePseudo;
242 auto pseudoId = pseudo->at(sfAccount);
243
244 if (auto ter = addEmptyHolding(
245 view, pseudoId, mPriorBalance, sleVault->at(sfAsset), j_))
246 return ter;
247
248 // Initialize data fields:
249 broker->at(sfSequence) = sequence;
250 broker->at(sfVaultID) = vaultID;
251 broker->at(sfOwner) = account_;
252 broker->at(sfAccount) = pseudoId;
253 // The LoanSequence indexes loans created by this broker, starting at 1
254 broker->at(sfLoanSequence) = 1;
255 if (auto const data = tx[~sfData])
256 broker->at(sfData) = *data;
257 if (auto const rate = tx[~sfManagementFeeRate])
258 broker->at(sfManagementFeeRate) = *rate;
259 if (auto const debtMax = tx[~sfDebtMaximum])
260 broker->at(sfDebtMaximum) = *debtMax;
261 if (auto const coverMin = tx[~sfCoverRateMinimum])
262 broker->at(sfCoverRateMinimum) = *coverMin;
263 if (auto const coverLiq = tx[~sfCoverRateLiquidation])
264 broker->at(sfCoverRateLiquidation) = *coverLiq;
265
266 view.insert(broker);
267
268 associateAsset(*broker, vaultAsset);
269 }
270
271 return tesSUCCESS;
272}
