22 auto const digest = [&]() -> std::optional<uint256> {
23 {
24 std::lock_guard lock(mutex_);
25 auto const iter = map_.find(k.key);
26 if (iter != map_.end())
27 {
28 cacheHit = true;
29 return iter->second;
30 }
31 }
32 return base_.digest(k.key);
33 }();

35 return nullptr;
36 auto sle = cache_.fetch(*digest, [&]() {
37 baseRead = true;
38 return base_.read(k);
39 });
40 // If the sle is null, then a failure must have occurred in base_.read()
41 XRPL_ASSERT(sle || baseRead, "xrpl::CachedView::read : null SLE result from base");
42 if (cacheHit && baseRead)
43 hitsexpired.increment();
44 else if (cacheHit)
45 hits.increment();
46 else
47 misses.increment();
48
49 if (!cacheHit)
50 {
51 // Avoid acquiring this lock unless necessary. It is only necessary if
52 // the key was not found in the map_. The lock is needed to add the key
53 // and digest.
54 std::lock_guard lock(mutex_);
55 map_.emplace(k.key, *digest);
56 }
57 if (!sle || !k.check(*sle))
58 return nullptr;
59 return sle;
60}
