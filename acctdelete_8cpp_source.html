21// Close the ledger until the ledger sequence is large enough to close
22// the account. If margin is specified, close the ledger so `margin`
23// more closes are needed
24void
25incLgrSeqForAccDel(jtx::Env& env, jtx::Account const& acc, std::uint32_t margin)
26{
27 using namespace jtx;
28 auto openLedgerSeq = [](jtx::Env& env) -> std::uint32_t { return env.current()->seq(); };
29
30 int const delta = [&]() -> int {
31 if (env.seq(acc) + 255 > openLedgerSeq(env))
32 return env.seq(acc) - openLedgerSeq(env) + 255 - margin;
33 return 0;
34 }();
35 env.test.BEAST_EXPECT(margin == 0 || delta >= 0);
36 for (int i = 0; i < delta; ++i)
37 env.close();
38 env.test.BEAST_EXPECT(openLedgerSeq(env) == env.seq(acc) + 255 - margin);
39}
