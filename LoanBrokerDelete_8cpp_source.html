24TER
25LoanBrokerDelete::preclaim(PreclaimContext const& ctx)
26{
27 auto const& tx = ctx.tx;
28
29 auto const account = tx[sfAccount];
30 auto const brokerID = tx[sfLoanBrokerID];
31
32 auto const sleBroker = ctx.view.read(keylet::loanbroker(brokerID));
33 if (!sleBroker)
34 {
35 JLOG(ctx.j.warn()) << "LoanBroker does not exist.";
36 return tecNO_ENTRY;
37 }
38
39 auto const brokerOwner = sleBroker->at(sfOwner);
40
41 if (account != brokerOwner)
42 {
43 JLOG(ctx.j.warn()) << "Account is not the owner of the LoanBroker.";
44 return tecNO_PERMISSION;
45 }
46 if (auto const ownerCount = sleBroker->at(sfOwnerCount); ownerCount != 0)
47 {
48 JLOG(ctx.j.warn()) << "LoanBrokerDelete: Owner count is " << ownerCount;
49 return tecHAS_OBLIGATIONS;
50 }
51
52 auto const vault = ctx.view.read(keylet::vault(sleBroker->at(sfVaultID)));
53 if (!vault)
54 {
55 // LCOV_EXCL_START
56 JLOG(ctx.j.fatal()) << "Vault is missing for Broker " << brokerID;
57 return tefBAD_LEDGER;
58 // LCOV_EXCL_STOP
59 }
60
61 Asset const asset = vault->at(sfAsset);
62
63 if (auto const debtTotal = sleBroker->at(sfDebtTotal);
64 debtTotal != beast::zero)
65 {
66 // Any remaining debt should have been wiped out by the last Loan
67 // Delete. This check is purely defensive.
68 auto const scale = getAssetsTotalScale(vault);
69
70 auto const rounded =
71 roundToAsset(asset, debtTotal, scale, Number::towards_zero);
72
73 if (rounded != beast::zero)
74 {
75 // LCOV_EXCL_START
76 JLOG(ctx.j.warn()) << "LoanBrokerDelete: Debt total is "
77 << debtTotal << ", which rounds to " << rounded;
78 return tecHAS_OBLIGATIONS;
79 // LCOV_EXCL_STOP
80 }
81 }
82
83 auto const coverAvailable =
84 STAmount{asset, sleBroker->at(sfCoverAvailable)};
85 // If there are assets in the cover, broker will receive them on deletion.
86 // So we need to check if the broker owner is deep frozen for that asset.
87 if (coverAvailable > beast::zero)
88 {
89 if (auto const ret = checkDeepFrozen(ctx.view, brokerOwner, asset))
90 {
91 JLOG(ctx.j.warn()) << "Broker owner account is frozen.";
92 return ret;
93 }
94 }
95
96 return tesSUCCESS;
97}

99TER
100LoanBrokerDelete::doApply()
101{
102 auto const& tx = ctx_.tx;
103
104 auto const brokerID = tx[sfLoanBrokerID];
105
106 // Delete the loan broker
107 auto broker = view().peek(keylet::loanbroker(brokerID));
108 if (!broker)
109 return tefBAD_LEDGER; // LCOV_EXCL_LINE
110 auto const vaultID = broker->at(sfVaultID);
111 auto const sleVault = view().read(keylet::vault(vaultID));
112 if (!sleVault)
113 return tefBAD_LEDGER; // LCOV_EXCL_LINE
114 auto const vaultPseudoID = sleVault->at(sfAccount);
115 auto const vaultAsset = sleVault->at(sfAsset);
116
117 auto const brokerPseudoID = broker->at(sfAccount);
118
119 if (!view().dirRemove(
120 keylet::ownerDir(account_),
121 broker->at(sfOwnerNode),
122 broker->key(),
123 false))
124 {
125 return tefBAD_LEDGER; // LCOV_EXCL_LINE
126 }
127 if (!view().dirRemove(
128 keylet::ownerDir(vaultPseudoID),
129 broker->at(sfVaultNode),
130 broker->key(),
131 false))
132 {
133 return tefBAD_LEDGER; // LCOV_EXCL_LINE
134 }
135
136 {
137 auto const coverAvailable =
138 STAmount{vaultAsset, broker->at(sfCoverAvailable)};
139 if (auto const ter = accountSend(
140 view(),
141 brokerPseudoID,
142 account_,
143 coverAvailable,
144 j_,
145 WaiveTransferFee::Yes))
146 return ter;
147 }
148
149 if (auto ter = removeEmptyHolding(view(), brokerPseudoID, vaultAsset, j_))
150 return ter;
151
152 auto brokerPseudoSLE = view().peek(keylet::account(brokerPseudoID));
153 if (!brokerPseudoSLE)
154 return tefBAD_LEDGER; // LCOV_EXCL_LINE
155
156 // Making the payment and removing the empty holding should have deleted any
157 // obligations associated with the broker or broker pseudo-account.
158 if (*brokerPseudoSLE->at(sfBalance))
159 {
160 JLOG(j_.warn()) << "LoanBrokerDelete: Pseudo-account has a balance";
161 return tecHAS_OBLIGATIONS; // LCOV_EXCL_LINE
162 }
163 if (brokerPseudoSLE->at(sfOwnerCount) != 0)
164 {
165 JLOG(j_.warn())
166 << "LoanBrokerDelete: Pseudo-account still owns objects";
167 return tecHAS_OBLIGATIONS; // LCOV_EXCL_LINE
168 }
169 if (auto const directory = keylet::ownerDir(brokerPseudoID);
170 view().read(directory))
171 {
172 JLOG(j_.warn()) << "LoanBrokerDelete: Pseudo-account has a directory";
173 return tecHAS_OBLIGATIONS; // LCOV_EXCL_LINE
174 }
175
176 view().erase(brokerPseudoSLE);
177
178 view().erase(broker);
179
180 {
181 auto owner = view().peek(keylet::account(account_));
182 if (!owner)
183 return tefBAD_LEDGER; // LCOV_EXCL_LINE
184
185 // Decreases the owner count by two: one for the LoanBroker object, and
186 // one for the pseudo-account.
187 adjustOwnerCount(view(), owner, -2, j_);
188 }
189
190 associateAsset(*broker, vaultAsset);
191
192 return tesSUCCESS;
193}
