44 }

120 if (enforce)

122 // LCOV_EXCL_STOP

125 {
126 auto const [amount, amount2] = ammPoolHolds(
127 view,
128 *ammAccount_,
129 tx[sfAmount].get<Issue>(),
130 tx[sfAmount2].get<Issue>(),
131 fhIGNORE_FREEZE,
132 j);
133 // Create invariant:
134 // sqrt(amount * amount2) == LPTokens
135 // all balances are greater than zero

140 << *lptAMMBalanceAfter_;

143 }

181bool
182ValidAMM::generalInvariant(
183 xrpl::STTx const& tx,
184 xrpl::ReadView const& view,
185 ZeroAllowed zeroAllowed,
186 beast::Journal const& j) const
187{
188 auto const [amount, amount2] = ammPoolHolds(
189 view,
190 *ammAccount_,
191 tx[sfAsset].get<Issue>(),
192 tx[sfAsset2].get<Issue>(),
193 fhIGNORE_FREEZE,
194 j);
195 // Deposit and Withdrawal invariant:
196 // sqrt(amount * amount2) >= LPTokens
197 // all balances are greater than zero
198 // unless on last withdrawal
199 auto const poolProductMean = root2(amount * amount2);
200 bool const nonNegativeBalances =
201 validBalances(amount, amount2, *lptAMMBalanceAfter_, zeroAllowed);
202 bool const strongInvariantCheck = poolProductMean >= *lptAMMBalanceAfter_;
203 // Allow for a small relative error if strongInvariantCheck fails
204 auto weakInvariantCheck = [&]() {
205 return *lptAMMBalanceAfter_ != beast::zero &&
206 withinRelativeDistance(poolProductMean, Number{*lptAMMBalanceAfter_}, Number{1, -11});
207 };
208 if (!nonNegativeBalances || (!strongInvariantCheck && !weakInvariantCheck()))
209 {
210 JLOG(j.error()) << "AMM " << tx.getTxnType()
211 << " invariant failed: " << tx.getHash(HashPrefix::transactionID) << " "
212 << ammPoolChanged_ << " " << amount << " " << amount2 << " "
213 << poolProductMean << " " << lptAMMBalanceAfter_->getText() << " "
214 << ((*lptAMMBalanceAfter_ == beast::zero)
215 ? Number{1}
216 : ((*lptAMMBalanceAfter_ - poolProductMean) / poolProductMean));
217 return false;
218 }
219
220 return true;
221}

264bool
265ValidAMM::finalize(
266 STTx const& tx,
267 TER const result,
268 XRPAmount const,
269 ReadView const& view,
270 beast::Journal const& j)
271{
272 // Delete may return tecINCOMPLETE if there are too many
273 // trustlines to delete.
274 if (result != tesSUCCESS && result != tecINCOMPLETE)
275 return true;
276
277 bool const enforce = view.rules().enabled(fixAMMv1_3);
278
279 switch (tx.getTxnType())
280 {
281 case ttAMM_CREATE:
282 return finalizeCreate(tx, view, enforce, j);
283 case ttAMM_DEPOSIT:
284 return finalizeDeposit(tx, view, enforce, j);
285 case ttAMM_CLAWBACK:
286 case ttAMM_WITHDRAW:
287 return finalizeWithdraw(tx, view, enforce, j);
288 case ttAMM_BID:
289 return finalizeBid(enforce, j);
290 case ttAMM_VOTE:
291 return finalizeVote(enforce, j);
292 case ttAMM_DELETE:
293 return finalizeDelete(enforce, result, j);
294 case ttCHECK_CASH:
295 case ttOFFER_CREATE:
296 case ttPAYMENT:
297 return finalizeDEX(enforce, j);
298 default:
299 break;
300 }
301
302 return true;
303}
