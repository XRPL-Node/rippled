98inline bool
99withinRelativeDistance(Quality const& calcQuality, Quality const& reqQuality, Number const& dist)
100{
101 if (calcQuality == reqQuality)
102 return true;
103 auto const [min, max] = std::minmax(calcQuality, reqQuality);
104 // Relative distance is (max - min)/max. Can't use basic operations
105 // on Quality. Have to use Quality::rate() instead, which
106 // is inverse of quality: (1/max.rate - 1/min.rate)/(1/max.rate)
107 return ((min.rate() - max.rate()) / min.rate()) < dist;
108}

118template <typename Amt>
119 requires(
120 std::is_same_v<Amt, STAmount> || std::is_same_v<Amt, IOUAmount> ||
121 std::is_same_v<Amt, XRPAmount> || std::is_same_v<Amt, Number>)
122bool
123withinRelativeDistance(Amt const& calc, Amt const& req, Number const& dist)
124{
125 if (calc == req)
126 return true;
127 auto const [min, max] = std::minmax(calc, req);
128 return ((max - min) / max) < dist;
129}

161template <typename TIn, typename TOut>
162std::optional<TAmounts<TIn, TOut>>
163getAMMOfferStartWithTakerGets(TAmounts<TIn, TOut> const& pool, Quality const& targetQuality, std::uint16_t const& tfee)
164{
165 if (targetQuality.rate() == beast::zero)
166 return std::nullopt;
167
168 NumberRoundModeGuard mg(Number::to_nearest);
169 auto const f = feeMult(tfee);
170 auto const a = 1;
171 auto const b = pool.in * (1 - 1 / f) / targetQuality.rate() - 2 * pool.out;
172 auto const c = pool.out * pool.out - (pool.in * pool.out) / targetQuality.rate();
173
174 auto nTakerGets = solveQuadraticEqSmallest(a, b, c);
175 if (!nTakerGets || *nTakerGets <= 0)
176 return std::nullopt; // LCOV_EXCL_LINE
177
178 auto const nTakerGetsConstraint = pool.out - pool.in / (targetQuality.rate() * f);
179 if (nTakerGetsConstraint <= 0)
180 return std::nullopt;
181
182 // Select the smallest to maximize the quality
183 if (nTakerGetsConstraint < *nTakerGets)
184 nTakerGets = nTakerGetsConstraint;
185
186 auto getAmounts = [&pool, &tfee](Number const& nTakerGetsProposed) {
187 // Round downward to minimize the offer and to maximize the quality.
188 // This has the most impact when takerGets is XRP.
189 auto const takerGets = toAmount<TOut>(getIssue(pool.out), nTakerGetsProposed, Number::downward);
190 return TAmounts<TIn, TOut>{swapAssetOut(pool, takerGets, tfee), takerGets};
191 };
192
193 // Try to reduce the offer size to improve the quality.
194 // The quality might still not match the targetQuality for a tiny offer.
195 if (auto const amounts = getAmounts(*nTakerGets); Quality{amounts} < targetQuality)
196 return getAmounts(detail::reduceOffer(amounts.out));
197 else
198 return amounts;
199}

224template <typename TIn, typename TOut>
225std::optional<TAmounts<TIn, TOut>>
226getAMMOfferStartWithTakerPays(TAmounts<TIn, TOut> const& pool, Quality const& targetQuality, std::uint16_t tfee)
227{
228 if (targetQuality.rate() == beast::zero)
229 return std::nullopt;
230
231 NumberRoundModeGuard mg(Number::to_nearest);
232 auto const f = feeMult(tfee);
233 auto const& a = f;
234 auto const b = pool.in * (1 + f);
235 auto const c = pool.in * pool.in - pool.in * pool.out * targetQuality.rate();
236
237 auto nTakerPays = solveQuadraticEqSmallest(a, b, c);
238 if (!nTakerPays || nTakerPays <= 0)
239 return std::nullopt; // LCOV_EXCL_LINE
240
241 auto const nTakerPaysConstraint = pool.out * targetQuality.rate() - pool.in / f;
242 if (nTakerPaysConstraint <= 0)
243 return std::nullopt;
244
245 // Select the smallest to maximize the quality
246 if (nTakerPaysConstraint < *nTakerPays)
247 nTakerPays = nTakerPaysConstraint;
248
249 auto getAmounts = [&pool, &tfee](Number const& nTakerPaysProposed) {
250 // Round downward to minimize the offer and to maximize the quality.
251 // This has the most impact when takerPays is XRP.
252 auto const takerPays = toAmount<TIn>(getIssue(pool.in), nTakerPaysProposed, Number::downward);
253 return TAmounts<TIn, TOut>{takerPays, swapAssetIn(pool, takerPays, tfee)};
254 };
255
256 // Try to reduce the offer size to improve the quality.
257 // The quality might still not match the targetQuality for a tiny offer.
258 if (auto const amounts = getAmounts(*nTakerPays); Quality{amounts} < targetQuality)
259 return getAmounts(detail::reduceOffer(amounts.in));
260 else
261 return amounts;

280template <typename TIn, typename TOut>
281std::optional<TAmounts<TIn, TOut>>
282changeSpotPriceQuality(
283 TAmounts<TIn, TOut> const& pool,
284 Quality const& quality,
285 std::uint16_t tfee,
286 Rules const& rules,
287 beast::Journal j)
288{
289 if (!rules.enabled(fixAMMv1_1))
290 {
291 // Finds takerPays (i) and takerGets (o) such that given pool
292 // composition poolGets(I) and poolPays(O): (O - o) / (I + i) = quality.
293 // Where takerGets is calculated as the swapAssetIn (see below).
294 // The above equation produces the quadratic equation:
295 // i^2*(1-fee) + i*I*(2-fee) + I^2 - I*O/quality,
296 // which is solved for i, and o is found with swapAssetIn().
297 auto const f = feeMult(tfee); // 1 - fee
298 auto const& a = f;
299 auto const b = pool.in * (1 + f);
300 Number const c = pool.in * pool.in - pool.in * pool.out * quality.rate();
301 if (auto const res = b * b - 4 * a * c; res < 0)
302 return std::nullopt; // LCOV_EXCL_LINE
303 else if (auto const nTakerPaysPropose = (-b + root2(res)) / (2 * a); nTakerPaysPropose > 0)
304 {
305 auto const nTakerPays = [&]() {
306 // The fee might make the AMM offer quality less than CLOB
307 // quality. Therefore, AMM offer has to satisfy this constraint:
308 // o / i >= q. Substituting o with swapAssetIn() gives: i <= O /
309 // q - I / (1 - fee).
310 auto const nTakerPaysConstraint = pool.out * quality.rate() - pool.in / f;
311 if (nTakerPaysPropose > nTakerPaysConstraint)
312 return nTakerPaysConstraint;
313 return nTakerPaysPropose;
314 }();
315 if (nTakerPays <= 0)
316 {
317 JLOG(j.trace()) << "changeSpotPriceQuality calc failed: " << to_string(pool.in) << " "
318 << to_string(pool.out) << " " << quality << " " << tfee;
319 return std::nullopt;
320 }
321 auto const takerPays = toAmount<TIn>(getIssue(pool.in), nTakerPays, Number::upward);
322 // should not fail
323 if (auto const amounts = TAmounts<TIn, TOut>{takerPays, swapAssetIn(pool, takerPays, tfee)};
324 Quality{amounts} < quality && !withinRelativeDistance(Quality{amounts}, quality, Number(1, -7)))
325 {
326 JLOG(j.error()) << "changeSpotPriceQuality failed: " << to_string(pool.in) << " " << to_string(pool.out)
327 << " "
328 << " " << quality << " " << tfee << " " << to_string(amounts.in) << " "
329 << to_string(amounts.out);
330 Throw<std::runtime_error>("changeSpotPriceQuality failed");
331 }
332 else
333 {
334 JLOG(j.trace()) << "changeSpotPriceQuality succeeded: " << to_string(pool.in) << " "
335 << to_string(pool.out) << " "
336 << " " << quality << " " << tfee << " " << to_string(amounts.in) << " "
337 << to_string(amounts.out);
338 return amounts;
339 }
340 }
341 JLOG(j.trace()) << "changeSpotPriceQuality calc failed: " << to_string(pool.in) << " " << to_string(pool.out)
342 << " " << quality << " " << tfee;
343 return std::nullopt;
344 }
345
346 // Generate the offer starting with XRP side. Return seated offer amounts
347 // if the offer can be generated, otherwise nullopt.
348 auto const amounts = [&]() {
349 if (isXRP(getIssue(pool.out)))
350 return getAMMOfferStartWithTakerGets(pool, quality, tfee);
351 return getAMMOfferStartWithTakerPays(pool, quality, tfee);
352 }();
353 if (!amounts)
354 {
355 JLOG(j.trace()) << "changeSpotPrice calc failed: " << to_string(pool.in) << " " << to_string(pool.out) << " "
356 << quality << " " << tfee << std::endl;
357 return std::nullopt;
358 }
359
360 if (Quality{*amounts} < quality)
361 {
362 JLOG(j.error()) << "changeSpotPriceQuality failed: " << to_string(pool.in) << " " << to_string(pool.out) << " "
363 << quality << " " << tfee << " " << to_string(amounts->in) << " " << to_string(amounts->out);
364 return std::nullopt;
365 }
366
367 JLOG(j.trace()) << "changeSpotPriceQuality succeeded: " << to_string(pool.in) << " " << to_string(pool.out) << " "
368 << " " << quality << " " << tfee << " " << to_string(amounts->in) << " " << to_string(amounts->out);
369
370 return amounts;
371}

394template <typename TIn, typename TOut>
395TOut
396swapAssetIn(TAmounts<TIn, TOut> const& pool, TIn const& assetIn, std::uint16_t tfee)
397{
398 if (auto const& rules = getCurrentTransactionRules(); rules && rules->enabled(fixAMMv1_1))
399 {
400 // set rounding to always favor the amm. Clip to zero.
401 // calculate:
402 // pool.out -
403 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
404 // and explicitly set the rounding modes
405 // Favoring the amm means we should:
406 // minimize:
407 // pool.out -
408 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
409 // maximize:
410 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
411 // (pool.in * pool.out)
412 // minimize:
413 // (pool.in + assetIn * feeMult(tfee)),
414 // minimize:
415 // assetIn * feeMult(tfee)
416 // feeMult is: (1-fee), fee is tfee/100000
417 // minimize:
418 // 1-fee
419 // maximize:
420 // fee
421 saveNumberRoundMode _{Number::getround()};
422
423 Number::setround(Number::upward);
424 auto const numerator = pool.in * pool.out;
425 auto const fee = getFee(tfee);
426
427 Number::setround(Number::downward);
428 auto const denom = pool.in + assetIn * (1 - fee);
429
430 if (denom.signum() <= 0)
431 return toAmount<TOut>(getIssue(pool.out), 0);
432
433 Number::setround(Number::upward);
434 auto const ratio = numerator / denom;
435
436 Number::setround(Number::downward);
437 auto const swapOut = pool.out - ratio;
438
439 if (swapOut.signum() < 0)
440 return toAmount<TOut>(getIssue(pool.out), 0);
441
442 return toAmount<TOut>(getIssue(pool.out), swapOut, Number::downward);
443 }
444 else
445 {
446 return toAmount<TOut>(
447 getIssue(pool.out),
448 pool.out - (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
449 Number::downward);
450 }
451}

462template <typename TIn, typename TOut>
463TIn
464swapAssetOut(TAmounts<TIn, TOut> const& pool, TOut const& assetOut, std::uint16_t tfee)
465{
466 if (auto const& rules = getCurrentTransactionRules(); rules && rules->enabled(fixAMMv1_1))
467 {
468 // set rounding to always favor the amm. Clip to zero.
469 // calculate:
470 // ((pool.in * pool.out) / (pool.out - assetOut) - pool.in) /
471 // (1-tfee/100000)
472 // maximize:
473 // ((pool.in * pool.out) / (pool.out - assetOut) - pool.in)
474 // maximize:
475 // (pool.in * pool.out) / (pool.out - assetOut)
476 // maximize:
477 // (pool.in * pool.out)
478 // minimize
479 // (pool.out - assetOut)
480 // minimize:
481 // (1-tfee/100000)
482 // maximize:
483 // tfee/100000
484
485 saveNumberRoundMode _{Number::getround()};
486
487 Number::setround(Number::upward);
488 auto const numerator = pool.in * pool.out;
489
490 Number::setround(Number::downward);
491 auto const denom = pool.out - assetOut;
492 if (denom.signum() <= 0)
493 {
494 return toMaxAmount<TIn>(getIssue(pool.in));
495 }
496
497 Number::setround(Number::upward);
498 auto const ratio = numerator / denom;
499 auto const numerator2 = ratio - pool.in;
500 auto const fee = getFee(tfee);
501
502 Number::setround(Number::downward);
503 auto const feeMult = 1 - fee;
504
505 Number::setround(Number::upward);
506 auto const swapIn = numerator2 / feeMult;
507 if (swapIn.signum() < 0)
508 return toAmount<TIn>(getIssue(pool.in), 0);
509
510 return toAmount<TIn>(getIssue(pool.in), swapIn, Number::upward);
511 }
512 else
513 {
514 return toAmount<TIn>(
515 getIssue(pool.in),
516 ((pool.in * pool.out) / (pool.out - assetOut) - pool.in) / feeMult(tfee),
517 Number::upward);
518 }
