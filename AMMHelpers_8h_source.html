105inline bool
106withinRelativeDistance(Quality const& calcQuality, Quality const& reqQuality, Number const& dist)
107{
108 if (calcQuality == reqQuality)
109 return true;
110 auto const [min, max] = std::minmax(calcQuality, reqQuality);
111 // Relative distance is (max - min)/max. Can't use basic operations
112 // on Quality. Have to use Quality::rate() instead, which
113 // is inverse of quality: (1/max.rate - 1/min.rate)/(1/max.rate)
114 return ((min.rate() - max.rate()) / min.rate()) < dist;
115}

125template <typename Amt>
126 requires(
127 std::is_same_v<Amt, STAmount> || std::is_same_v<Amt, IOUAmount> ||
128 std::is_same_v<Amt, XRPAmount> || std::is_same_v<Amt, Number>)
129bool
130withinRelativeDistance(Amt const& calc, Amt const& req, Number const& dist)
131{
132 if (calc == req)
133 return true;
134 auto const [min, max] = std::minmax(calc, req);
135 return ((max - min) / max) < dist;
136}

168template <typename TIn, typename TOut>
169std::optional<TAmounts<TIn, TOut>>
170getAMMOfferStartWithTakerGets(
171 TAmounts<TIn, TOut> const& pool,
172 Quality const& targetQuality,
173 std::uint16_t const& tfee)
174{
175 if (targetQuality.rate() == beast::zero)
176 return std::nullopt;
177
178 NumberRoundModeGuard mg(Number::to_nearest);
179 auto const f = feeMult(tfee);
180 auto const a = 1;
181 auto const b = pool.in * (1 - 1 / f) / targetQuality.rate() - 2 * pool.out;
182 auto const c = pool.out * pool.out - (pool.in * pool.out) / targetQuality.rate();
183
184 auto nTakerGets = solveQuadraticEqSmallest(a, b, c);
185 if (!nTakerGets || *nTakerGets <= 0)
186 return std::nullopt; // LCOV_EXCL_LINE
187
188 auto const nTakerGetsConstraint = pool.out - pool.in / (targetQuality.rate() * f);
189 if (nTakerGetsConstraint <= 0)
190 return std::nullopt;
191
192 // Select the smallest to maximize the quality
193 if (nTakerGetsConstraint < *nTakerGets)
194 nTakerGets = nTakerGetsConstraint;
195
196 auto getAmounts = [&pool, &tfee](Number const& nTakerGetsProposed) {
197 // Round downward to minimize the offer and to maximize the quality.
198 // This has the most impact when takerGets is XRP.
199 auto const takerGets =
200 toAmount<TOut>(getIssue(pool.out), nTakerGetsProposed, Number::downward);
201 return TAmounts<TIn, TOut>{swapAssetOut(pool, takerGets, tfee), takerGets};
202 };
203
204 // Try to reduce the offer size to improve the quality.
205 // The quality might still not match the targetQuality for a tiny offer.
206 if (auto const amounts = getAmounts(*nTakerGets); Quality{amounts} < targetQuality)
207 return getAmounts(detail::reduceOffer(amounts.out));
208 else
209 return amounts;
210}

235template <typename TIn, typename TOut>
236std::optional<TAmounts<TIn, TOut>>
237getAMMOfferStartWithTakerPays(
238 TAmounts<TIn, TOut> const& pool,
239 Quality const& targetQuality,
240 std::uint16_t tfee)
241{
242 if (targetQuality.rate() == beast::zero)
243 return std::nullopt;
244
245 NumberRoundModeGuard mg(Number::to_nearest);
246 auto const f = feeMult(tfee);
247 auto const& a = f;
248 auto const b = pool.in * (1 + f);
249 auto const c = pool.in * pool.in - pool.in * pool.out * targetQuality.rate();
250
251 auto nTakerPays = solveQuadraticEqSmallest(a, b, c);
252 if (!nTakerPays || nTakerPays <= 0)
253 return std::nullopt; // LCOV_EXCL_LINE
254
255 auto const nTakerPaysConstraint = pool.out * targetQuality.rate() - pool.in / f;
256 if (nTakerPaysConstraint <= 0)
257 return std::nullopt;
258
259 // Select the smallest to maximize the quality
260 if (nTakerPaysConstraint < *nTakerPays)
261 nTakerPays = nTakerPaysConstraint;
262
263 auto getAmounts = [&pool, &tfee](Number const& nTakerPaysProposed) {
264 // Round downward to minimize the offer and to maximize the quality.
265 // This has the most impact when takerPays is XRP.
266 auto const takerPays =
267 toAmount<TIn>(getIssue(pool.in), nTakerPaysProposed, Number::downward);
268 return TAmounts<TIn, TOut>{takerPays, swapAssetIn(pool, takerPays, tfee)};
269 };
270
271 // Try to reduce the offer size to improve the quality.
272 // The quality might still not match the targetQuality for a tiny offer.
273 if (auto const amounts = getAmounts(*nTakerPays); Quality{amounts} < targetQuality)
274 return getAmounts(detail::reduceOffer(amounts.in));
275 else
276 return amounts;

295template <typename TIn, typename TOut>
296std::optional<TAmounts<TIn, TOut>>
297changeSpotPriceQuality(
298 TAmounts<TIn, TOut> const& pool,
299 Quality const& quality,
300 std::uint16_t tfee,
301 Rules const& rules,
302 beast::Journal j)
303{
304 if (!rules.enabled(fixAMMv1_1))
305 {
306 // Finds takerPays (i) and takerGets (o) such that given pool
307 // composition poolGets(I) and poolPays(O): (O - o) / (I + i) = quality.
308 // Where takerGets is calculated as the swapAssetIn (see below).
309 // The above equation produces the quadratic equation:
310 // i^2*(1-fee) + i*I*(2-fee) + I^2 - I*O/quality,
311 // which is solved for i, and o is found with swapAssetIn().
312 auto const f = feeMult(tfee); // 1 - fee
313 auto const& a = f;
314 auto const b = pool.in * (1 + f);
315 Number const c = pool.in * pool.in - pool.in * pool.out * quality.rate();
316 if (auto const res = b * b - 4 * a * c; res < 0)
317 return std::nullopt; // LCOV_EXCL_LINE
318 else if (auto const nTakerPaysPropose = (-b + root2(res)) / (2 * a); nTakerPaysPropose > 0)
319 {
320 auto const nTakerPays = [&]() {
321 // The fee might make the AMM offer quality less than CLOB
322 // quality. Therefore, AMM offer has to satisfy this constraint:
323 // o / i >= q. Substituting o with swapAssetIn() gives: i <= O /
324 // q - I / (1 - fee).
325 auto const nTakerPaysConstraint = pool.out * quality.rate() - pool.in / f;
326 if (nTakerPaysPropose > nTakerPaysConstraint)
327 return nTakerPaysConstraint;
328 return nTakerPaysPropose;
329 }();
330 if (nTakerPays <= 0)
331 {
332 JLOG(j.trace()) << "changeSpotPriceQuality calc failed: " << to_string(pool.in)
333 << " " << to_string(pool.out) << " " << quality << " " << tfee;
334 return std::nullopt;
335 }
336 auto const takerPays = toAmount<TIn>(getIssue(pool.in), nTakerPays, Number::upward);
337 // should not fail
338 if (auto const amounts =
339 TAmounts<TIn, TOut>{takerPays, swapAssetIn(pool, takerPays, tfee)};
340 Quality{amounts} < quality &&
341 !withinRelativeDistance(Quality{amounts}, quality, Number(1, -7)))
342 {
343 JLOG(j.error()) << "changeSpotPriceQuality failed: " << to_string(pool.in) << " "
344 << to_string(pool.out) << " "
345 << " " << quality << " " << tfee << " " << to_string(amounts.in)
346 << " " << to_string(amounts.out);
347 Throw<std::runtime_error>("changeSpotPriceQuality failed");
348 }
349 else
350 {
351 JLOG(j.trace()) << "changeSpotPriceQuality succeeded: " << to_string(pool.in) << " "
352 << to_string(pool.out) << " "
353 << " " << quality << " " << tfee << " " << to_string(amounts.in)
354 << " " << to_string(amounts.out);
355 return amounts;
356 }
357 }
358 JLOG(j.trace()) << "changeSpotPriceQuality calc failed: " << to_string(pool.in) << " "
359 << to_string(pool.out) << " " << quality << " " << tfee;
360 return std::nullopt;
361 }
362
363 // Generate the offer starting with XRP side. Return seated offer amounts
364 // if the offer can be generated, otherwise nullopt.
365 auto const amounts = [&]() {
366 if (isXRP(getIssue(pool.out)))
367 return getAMMOfferStartWithTakerGets(pool, quality, tfee);
368 return getAMMOfferStartWithTakerPays(pool, quality, tfee);
369 }();
370 if (!amounts)
371 {
372 JLOG(j.trace()) << "changeSpotPrice calc failed: " << to_string(pool.in) << " "
373 << to_string(pool.out) << " " << quality << " " << tfee << std::endl;
374 return std::nullopt;
375 }
376
377 if (Quality{*amounts} < quality)
378 {
379 JLOG(j.error()) << "changeSpotPriceQuality failed: " << to_string(pool.in) << " "
380 << to_string(pool.out) << " " << quality << " " << tfee << " "
381 << to_string(amounts->in) << " " << to_string(amounts->out);
382 return std::nullopt;
383 }
384
385 JLOG(j.trace()) << "changeSpotPriceQuality succeeded: " << to_string(pool.in) << " "
386 << to_string(pool.out) << " "
387 << " " << quality << " " << tfee << " " << to_string(amounts->in) << " "
388 << to_string(amounts->out);
389
390 return amounts;
391}

414template <typename TIn, typename TOut>
415TOut
416swapAssetIn(TAmounts<TIn, TOut> const& pool, TIn const& assetIn, std::uint16_t tfee)
417{
418 if (auto const& rules = getCurrentTransactionRules(); rules && rules->enabled(fixAMMv1_1))
419 {
420 // set rounding to always favor the amm. Clip to zero.
421 // calculate:
422 // pool.out -
423 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
424 // and explicitly set the rounding modes
425 // Favoring the amm means we should:
426 // minimize:
427 // pool.out -
428 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
429 // maximize:
430 // (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
431 // (pool.in * pool.out)
432 // minimize:
433 // (pool.in + assetIn * feeMult(tfee)),
434 // minimize:
435 // assetIn * feeMult(tfee)
436 // feeMult is: (1-fee), fee is tfee/100000
437 // minimize:
438 // 1-fee
439 // maximize:
440 // fee
441 saveNumberRoundMode _{Number::getround()};
442
443 Number::setround(Number::upward);
444 auto const numerator = pool.in * pool.out;
445 auto const fee = getFee(tfee);
446
447 Number::setround(Number::downward);
448 auto const denom = pool.in + assetIn * (1 - fee);
449
450 if (denom.signum() <= 0)
451 return toAmount<TOut>(getIssue(pool.out), 0);
452
453 Number::setround(Number::upward);
454 auto const ratio = numerator / denom;
455
456 Number::setround(Number::downward);
457 auto const swapOut = pool.out - ratio;
458
459 if (swapOut.signum() < 0)
460 return toAmount<TOut>(getIssue(pool.out), 0);
461
462 return toAmount<TOut>(getIssue(pool.out), swapOut, Number::downward);
463 }
464 else
465 {
466 return toAmount<TOut>(
467 getIssue(pool.out),
468 pool.out - (pool.in * pool.out) / (pool.in + assetIn * feeMult(tfee)),
469 Number::downward);
470 }
471}

482template <typename TIn, typename TOut>
483TIn
484swapAssetOut(TAmounts<TIn, TOut> const& pool, TOut const& assetOut, std::uint16_t tfee)
485{
486 if (auto const& rules = getCurrentTransactionRules(); rules && rules->enabled(fixAMMv1_1))
487 {
488 // set rounding to always favor the amm. Clip to zero.
489 // calculate:
490 // ((pool.in * pool.out) / (pool.out - assetOut) - pool.in) /
491 // (1-tfee/100000)
492 // maximize:
493 // ((pool.in * pool.out) / (pool.out - assetOut) - pool.in)
494 // maximize:
495 // (pool.in * pool.out) / (pool.out - assetOut)
496 // maximize:
497 // (pool.in * pool.out)
498 // minimize
499 // (pool.out - assetOut)
500 // minimize:
501 // (1-tfee/100000)
502 // maximize:
503 // tfee/100000
504
505 saveNumberRoundMode _{Number::getround()};
506
507 Number::setround(Number::upward);
508 auto const numerator = pool.in * pool.out;
509
510 Number::setround(Number::downward);
511 auto const denom = pool.out - assetOut;
512 if (denom.signum() <= 0)
513 {
514 return toMaxAmount<TIn>(getIssue(pool.in));
515 }
516
517 Number::setround(Number::upward);
518 auto const ratio = numerator / denom;
519 auto const numerator2 = ratio - pool.in;
520 auto const fee = getFee(tfee);
521
522 Number::setround(Number::downward);
523 auto const feeMult = 1 - fee;
524
525 Number::setround(Number::upward);
526 auto const swapIn = numerator2 / feeMult;
527 if (swapIn.signum() < 0)
528 return toAmount<TIn>(getIssue(pool.in), 0);
529
530 return toAmount<TIn>(getIssue(pool.in), swapIn, Number::upward);
531 }
532 else
533 {
534 return toAmount<TIn>(
535 getIssue(pool.in),
536 ((pool.in * pool.out) / (pool.out - assetOut) - pool.in) / feeMult(tfee),
537 Number::upward);
538 }
