96std::shared_ptr<NodeObject>
97DatabaseRotatingImp::fetchNodeObject(
98 uint256 const& hash,
99 std::uint32_t,
100 FetchReport& fetchReport,
101 bool duplicate)
102{
103 auto fetch = [&](std::shared_ptr<Backend> const& backend) {
104 Status status;
105 std::shared_ptr<NodeObject> nodeObject;
106 try
107 {
108 status = backend->fetch(hash.data(), &nodeObject);
109 }
110 catch (std::exception const& e)
111 {
112 JLOG(j_.fatal()) << "Exception, " << e.what();
113 Rethrow();
114 }
115
116 switch (status)
117 {
118 case ok:
119 case notFound:
120 break;
121 case dataCorrupt:
122 JLOG(j_.fatal()) << "Corrupt NodeObject #" << hash;
123 break;
124 default:
125 JLOG(j_.warn()) << "Unknown status=" << status;
126 break;
127 }
128
129 return nodeObject;
130 };
131
132 // See if the node object exists in the cache
133 std::shared_ptr<NodeObject> nodeObject;
134
135 auto [writable, archive] = [&] {
136 std::lock_guard lock(mutex_);
137 return std::make_pair(writableBackend_, archiveBackend_);
138 }();
139
140 // Try to fetch from the writable backend
141 nodeObject = fetch(writable);
142 if (!nodeObject)
143 {
144 // Otherwise try to fetch from the archive backend
145 nodeObject = fetch(archive);
146 if (nodeObject)
147 {
148 {
149 // Refresh the writable backend pointer
150 std::lock_guard lock(mutex_);
151 writable = writableBackend_;
152 }
153
154 // Update writable backend with data from the archive backend
155 if (duplicate)
156 writable->store(nodeObject);
157 }
158 }
159
160 if (nodeObject)
161 fetchReport.wasFound = true;
162
163 return nodeObject;
164}
