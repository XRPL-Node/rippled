179template <class Handler>
180void
181Door<Handler>::Detector::do_detect(boost::asio::yield_context do_yield)
182{
183 boost::beast::multi_buffer buf(16);
184 stream_.expires_after(std::chrono::seconds(15));
185 boost::system::error_code ec;
186 bool const ssl = async_detect_ssl(stream_, buf, do_yield[ec]);
187 stream_.expires_never();
188 if (!ec)
189 {
190 if (ssl)
191 {
192 if (auto sp = ios().template emplace<SSLHTTPPeer<Handler>>(
193 port_, handler_, ioc_, j_, remote_address_, buf.data(), std::move(stream_)))
194 sp->run();
195 return;
196 }
197 if (auto sp = ios().template emplace<PlainHTTPPeer<Handler>>(
198 port_, handler_, ioc_, j_, remote_address_, buf.data(), std::move(stream_)))
199 sp->run();
200 return;
201 }
202 if (ec != boost::asio::error::operation_aborted)
203 {
204 JLOG(j_.trace()) << "Error detecting ssl: " << ec.message() << " from " << remote_address_;
205 }
206}

210template <class Handler>
211void
212Door<Handler>::reOpen()
213{
214 error_code ec;
215
216 if (acceptor_.is_open())
217 {
218 acceptor_.close(ec);
219 if (ec)
220 {
221 std::stringstream ss;
222 ss << "Can't close acceptor: " << port_.name << ", " << ec.message();
223 JLOG(j_.error()) << ss.str();
224 Throw<std::runtime_error>(ss.str());
225 }
226 }
227
228 endpoint_type const local_address = endpoint_type(port_.ip, port_.port);
229
230 acceptor_.open(local_address.protocol(), ec);
231 if (ec)
232 {
233 JLOG(j_.error()) << "Open port '" << port_.name << "' failed:" << ec.message();
234 Throw<std::exception>();
235 }
236
237 acceptor_.set_option(boost::asio::ip::tcp::acceptor::reuse_address(true), ec);
238 if (ec)
239 {
240 JLOG(j_.error()) << "Option for port '" << port_.name << "' failed:" << ec.message();
241 Throw<std::exception>();
242 }
243
244 acceptor_.bind(local_address, ec);
245 if (ec)
246 {
247 JLOG(j_.error()) << "Bind port '" << port_.name << "' failed:" << ec.message();
248 Throw<std::exception>();
249 }
250
251 acceptor_.listen(boost::asio::socket_base::max_listen_connections, ec);
252 if (ec)
253 {
254 JLOG(j_.error()) << "Listen on port '" << port_.name << "' failed:" << ec.message();
255 Throw<std::exception>();
256 }
257
258 JLOG(j_.info()) << "Opened " << port_;
259}

315template <class Handler>
316void
317Door<Handler>::do_accept(boost::asio::yield_context do_yield)
318{
319 while (acceptor_.is_open())
320 {
321 if (should_throttle_for_fds())
322 {
323 backoff_timer_.expires_after(accept_delay_);
324 boost::system::error_code tec;
325 backoff_timer_.async_wait(do_yield[tec]);
326 accept_delay_ = std::min(accept_delay_ * 2, MAX_ACCEPT_DELAY);
327 JLOG(j_.warn()) << "Throttling do_accept for " << accept_delay_.count() << "ms.";
328 continue;
329 }
330
331 error_code ec;
332 endpoint_type remote_address;
333 stream_type stream(ioc_);
334 socket_type& socket = stream.socket();
335 acceptor_.async_accept(socket, remote_address, do_yield[ec]);
336 if (ec)
337 {
338 if (ec == boost::asio::error::operation_aborted)
339 break;
340
341 if (ec == boost::asio::error::no_descriptors || ec == boost::asio::error::no_buffer_space)
342 {
343 JLOG(j_.warn()) << "accept: Too many open files. Pausing for " << accept_delay_.count() << "ms.";
344
345 backoff_timer_.expires_after(accept_delay_);
346 boost::system::error_code tec;
347 backoff_timer_.async_wait(do_yield[tec]);
348
349 accept_delay_ = std::min(accept_delay_ * 2, MAX_ACCEPT_DELAY);
350 }
351 else
352 {
353 JLOG(j_.error()) << "accept error: " << ec.message();
354 }
355 continue;
356 }
357
358 accept_delay_ = INITIAL_ACCEPT_DELAY;
359
360 if (ssl_ && plain_)
361 {
362 if (auto sp =
363 ios().template emplace<Detector>(port_, handler_, ioc_, std::move(stream), remote_address, j_))
364 sp->run();
365 }
366 else if (ssl_ || plain_)
367 {
368 create(ssl_, boost::asio::null_buffers{}, std::move(stream), remote_address);
369 }
370 }

373template <class Handler>
374std::optional<typename Door<Handler>::FDStats>
375Door<Handler>::query_fd_stats() const
376{
377#if BOOST_OS_WINDOWS
378 return std::nullopt;
379#else
380 FDStats s;
381 struct rlimit rl;
382 if (getrlimit(RLIMIT_NOFILE, &rl) != 0 || rl.rlim_cur == RLIM_INFINITY)
383 return std::nullopt;
384 s.limit = static_cast<std::uint64_t>(rl.rlim_cur);
385#if BOOST_OS_LINUX
386 constexpr char const* kFdDir = "/proc/self/fd";
387#else
388 constexpr char const* kFdDir = "/dev/fd";
389#endif
390 if (DIR* d = ::opendir(kFdDir))
391 {
392 std::uint64_t cnt = 0;
393 while (::readdir(d) != nullptr)
394 ++cnt;
395 ::closedir(d);
396 // readdir counts '.', '..', and the DIR* itself shows in the list
397 s.used = (cnt >= 3) ? (cnt - 3) : 0;
398 return s;
399 }
400 return std::nullopt;
401#endif
402}

404template <class Handler>
405bool
406Door<Handler>::should_throttle_for_fds()
407{
408#if BOOST_OS_WINDOWS
409 return false;
410#else
411 auto const stats = query_fd_stats();
412 if (!stats || stats->limit == 0)
413 return false;
414
415 auto const& s = *stats;
416 auto const free = (s.limit > s.used) ? (s.limit - s.used) : 0ull;
417 double const free_ratio = static_cast<double>(free) / static_cast<double>(s.limit);
418 if (free_ratio < FREE_FD_THRESHOLD)
419 {
420 return true;
421 }
422 return false;
423#endif
424}
