27NotTEC
28LoanPay::preflight(PreflightContext const& ctx)
29{
30 if (ctx.tx[sfLoanID] == beast::zero)
31 return temINVALID;
32
33 if (ctx.tx[sfAmount] <= beast::zero)
34 return temBAD_AMOUNT;
35
36 // The loan payment flags are all mutually exclusive. If more than one is
37 // set, the tx is malformed.
38 static_assert(
39 (tfLoanLatePayment | tfLoanFullPayment | tfLoanOverpayment) ==
40 ~(tfLoanPayMask | tfUniversal));
41 auto const flagsSet = ctx.tx.getFlags() & ~(tfLoanPayMask | tfUniversal);
42 if (std::popcount(flagsSet) > 1)
43 {
44 JLOG(ctx.j.warn()) << "Only one LoanPay flag can be set per tx. "
45 << flagsSet << " is too many.";
46 return temINVALID_FLAG;
47 }
48
49 return tesSUCCESS;
50}

52XRPAmount
53LoanPay::calculateBaseFee(ReadView const& view, STTx const& tx)
54{
55 using namespace Lending;
56
57 auto const normalCost = Transactor::calculateBaseFee(view, tx);
58
59 if (tx.isFlag(tfLoanFullPayment) || tx.isFlag(tfLoanLatePayment))
60 // The loan will be making one set of calculations for one full or late
61 // payment
62 return normalCost;
63
64 // The fee is based on the potential number of payments, unless the loan is
65 // being fully paid off.
66 auto const amount = tx[sfAmount];
67 auto const loanID = tx[sfLoanID];
68
69 auto const loanSle = view.read(keylet::loan(loanID));
70 if (!loanSle)
71 // Let preclaim worry about the error for this
72 return normalCost;
73
74 if (loanSle->at(sfPaymentRemaining) <= loanPaymentsPerFeeIncrement)
75 {
76 // If there are fewer than loanPaymentsPerFeeIncrement payments left to
77 // pay, we can skip the computations.
78 return normalCost;
79 }
80
81 if (hasExpired(view, loanSle->at(sfNextPaymentDueDate)))
82 // If the payment is late, and the late payment flag is not set, it'll
83 // fail
84 return normalCost;
85
86 auto const brokerSle =
87 view.read(keylet::loanbroker(loanSle->at(sfLoanBrokerID)));
88 if (!brokerSle)
89 // Let preclaim worry about the error for this
90 return normalCost;
91 auto const vaultSle = view.read(keylet::vault(brokerSle->at(sfVaultID)));
92 if (!vaultSle)
93 // Let preclaim worry about the error for this
94 return normalCost;
95
96 auto const asset = vaultSle->at(sfAsset);
97
98 if (asset != amount.asset())
99 // Let preclaim worry about the error for this
100 return normalCost;
101
102 auto const scale = loanSle->at(sfLoanScale);
103
104 auto const regularPayment =
105 roundPeriodicPayment(asset, loanSle->at(sfPeriodicPayment), scale) +
106 loanSle->at(sfLoanServiceFee);
107
108 // If making an overpayment, count it as a full payment because it will do
109 // about the same amount of work, if not more.
110 NumberRoundModeGuard mg(
111 tx.isFlag(tfLoanOverpayment) ? Number::upward : Number::downward);
112 // Estimate how many payments will be made
113 Number const numPaymentEstimate =
114 static_cast<std::int64_t>(amount / regularPayment);
115
116 // Charge one base fee per paymentsPerFeeIncrement payments, rounding up.
117 Number::setround(Number::upward);
118 auto const feeIncrements = std::max(
119 std::int64_t(1),
120 static_cast<std::int64_t>(
121 numPaymentEstimate / loanPaymentsPerFeeIncrement));
122
123 return feeIncrements * normalCost;
124}

126TER
127LoanPay::preclaim(PreclaimContext const& ctx)
128{
129 auto const& tx = ctx.tx;
130
131 auto const account = tx[sfAccount];
132 auto const loanID = tx[sfLoanID];
133 auto const amount = tx[sfAmount];
134
135 auto const loanSle = ctx.view.read(keylet::loan(loanID));
136 if (!loanSle)
137 {
138 JLOG(ctx.j.warn()) << "Loan does not exist.";
139 return tecNO_ENTRY;
140 }
141
142 if (loanSle->at(sfBorrower) != account)
143 {
144 JLOG(ctx.j.warn()) << "Loan does not belong to the account.";
145 return tecNO_PERMISSION;
146 }
147
148 if (tx.isFlag(tfLoanOverpayment) && !loanSle->isFlag(lsfLoanOverpayment))
149 {
150 JLOG(ctx.j.warn())
151 << "Requested overpayment on a loan that doesn't allow it";
152 return temINVALID_FLAG;
153 }
154
155 auto const principalOutstanding = loanSle->at(sfPrincipalOutstanding);
156 auto const paymentRemaining = loanSle->at(sfPaymentRemaining);
157
158 if (paymentRemaining == 0 || principalOutstanding == 0)
159 {
160 JLOG(ctx.j.warn()) << "Loan is already paid off.";
161 return tecKILLED;
162 }
163
164 auto const loanBrokerID = loanSle->at(sfLoanBrokerID);
165 auto const loanBrokerSle = ctx.view.read(keylet::loanbroker(loanBrokerID));
166 if (!loanBrokerSle)
167 {
168 // This should be impossible
169 // LCOV_EXCL_START
170 JLOG(ctx.j.fatal()) << "LoanBroker does not exist.";
171 return tefBAD_LEDGER;
172 // LCOV_EXCL_STOP
173 }
174 auto const vaultID = loanBrokerSle->at(sfVaultID);
175 auto const vaultSle = ctx.view.read(keylet::vault(vaultID));
176 if (!vaultSle)
177 {
178 // This should be impossible
179 // LCOV_EXCL_START
180 JLOG(ctx.j.fatal()) << "Vault does not exist.";
181 return tefBAD_LEDGER;
182 // LCOV_EXCL_STOP
183 }
184 auto const asset = vaultSle->at(sfAsset);
185 auto const vaultPseudoAccount = vaultSle->at(sfAccount);
186
187 if (amount.asset() != asset)
188 {
189 JLOG(ctx.j.warn()) << "Loan amount does not match the Vault asset.";
190 return tecWRONG_ASSET;
191 }
192
193 if (auto const ret = checkFrozen(ctx.view, account, asset))
194 {
195 JLOG(ctx.j.warn()) << "Borrower account is frozen.";
196 return ret;
197 }
198 if (auto const ret = checkDeepFrozen(ctx.view, vaultPseudoAccount, asset))
199 {
200 JLOG(ctx.j.warn())
201 << "Vault pseudo-account can not receive funds (deep frozen).";
202 return ret;
203 }
204 if (auto const ret = requireAuth(ctx.view, asset, account))
205 {
206 JLOG(ctx.j.warn()) << "Borrower account is not authorized.";
207 return ret;
208 }
209 // Make sure the borrower has enough funds to make the payment!
210 // Do not support "partial payments" - if the transaction says to pay X,
211 // then the account must have X available, even if the loan payment takes
212 // less.
213 if (auto const balance = accountHolds(
214 ctx.view,
215 account,
216 asset,
217 fhZERO_IF_FROZEN,
218 ahZERO_IF_UNAUTHORIZED,
219 ctx.j,
220 SpendableHandling::shFULL_BALANCE);
221 balance < amount)
222 {
223 JLOG(ctx.j.warn()) << "Payment amount too large. Amount: "
224 << to_string(amount.getJson())
225 << ". Balance: " << to_string(balance.getJson());
226 return tecINSUFFICIENT_FUNDS;
227 }
228
229 return tesSUCCESS;
230}

232TER
233LoanPay::doApply()
234{
235 auto const& tx = ctx_.tx;
236 auto& view = ctx_.view();
237
238 auto const amount = tx[sfAmount];
239
240 auto const loanID = tx[sfLoanID];
241 auto const loanSle = view.peek(keylet::loan(loanID));
242 if (!loanSle)
243 return tefBAD_LEDGER; // LCOV_EXCL_LINE
244 std::int32_t const loanScale = loanSle->at(sfLoanScale);
245
246 auto const brokerID = loanSle->at(sfLoanBrokerID);
247 auto const brokerSle = view.peek(keylet::loanbroker(brokerID));
248 if (!brokerSle)
249 return tefBAD_LEDGER; // LCOV_EXCL_LINE
250 auto const brokerOwner = brokerSle->at(sfOwner);
251 auto const brokerPseudoAccount = brokerSle->at(sfAccount);
252 auto const vaultID = brokerSle->at(sfVaultID);
253 auto const vaultSle = view.peek(keylet::vault(vaultID));
254 if (!vaultSle)
255 return tefBAD_LEDGER; // LCOV_EXCL_LINE
256 auto const vaultPseudoAccount = vaultSle->at(sfAccount);
257 auto const asset = *vaultSle->at(sfAsset);
258
259 // Determine where to send the broker's fee
260 auto coverAvailableProxy = brokerSle->at(sfCoverAvailable);
261 TenthBips32 const coverRateMinimum{brokerSle->at(sfCoverRateMinimum)};
262 auto debtTotalProxy = brokerSle->at(sfDebtTotal);
263
264 // Send the broker fee to the owner if they have sufficient cover available,
265 // _and_ if the owner can receive funds
266 // _and_ if the broker is authorized to hold funds. If not, so as not to
267 // block the payment, add it to the cover balance (send it to the broker
268 // pseudo account).
269 //
270 // Normally freeze status is checked in preclaim, but we do it here to
271 // avoid duplicating the check. It'll claim a fee either way.
272 bool const sendBrokerFeeToOwner = [&]() {
273 // Round the minimum required cover up to be conservative. This ensures
274 // CoverAvailable never drops below the theoretical minimum, protecting
275 // the broker's solvency.
276 NumberRoundModeGuard mg(Number::upward);
277 return coverAvailableProxy >=
278 roundToAsset(
279 asset,
280 tenthBipsOfValue(debtTotalProxy.value(), coverRateMinimum),
281 loanScale) &&
282 !isDeepFrozen(view, brokerOwner, asset) &&
283 !requireAuth(view, asset, brokerOwner, AuthType::StrongAuth);
284 }();
285
286 auto const brokerPayee =
287 sendBrokerFeeToOwner ? brokerOwner : brokerPseudoAccount;
288 auto const brokerPayeeSle = view.peek(keylet::account(brokerPayee));
289 if (!sendBrokerFeeToOwner)
290 {
291 // If we can't send the fee to the owner, and the pseudo-account is
292 // frozen, then we have to fail the payment.
293 if (auto const ret = checkDeepFrozen(view, brokerPayee, asset))
294 {
295 JLOG(j_.warn())
296 << "Both Loan Broker and Loan Broker pseudo-account "
297 "can not receive funds (deep frozen).";
298 return ret;
299 }
300 }
301
302 //------------------------------------------------------
303 // Loan object state changes
304
305 // Unimpair the loan if it was impaired. Do this before the payment is
306 // attempted, so the original values can be used. If the payment fails, this
307 // change will be discarded.
308 if (loanSle->isFlag(lsfLoanImpaired))
309 {
310 if (auto const ret =
311 LoanManage::unimpairLoan(view, loanSle, vaultSle, asset, j_))
312 {
313 JLOG(j_.fatal()) << "Failed to unimpair loan before payment.";
314 return ret; // LCOV_EXCL_LINE
315 }
316 }
317
318 LoanPaymentType const paymentType = [&tx]() {
319 // preflight already checked that at most one flag is set.
320 if (tx.isFlag(tfLoanLatePayment))
321 return LoanPaymentType::late;
322 if (tx.isFlag(tfLoanFullPayment))
323 return LoanPaymentType::full;
324 if (tx.isFlag(tfLoanOverpayment))
325 return LoanPaymentType::overpayment;
326 return LoanPaymentType::regular;
327 }();
328
329 Expected<LoanPaymentParts, TER> const paymentParts = loanMakePayment(
330 asset, view, loanSle, brokerSle, amount, paymentType, j_);
331
332 if (!paymentParts)
333 {
334 XRPL_ASSERT_PARTS(
335 paymentParts.error(),
336 "xrpl::LoanPay::doApply",
337 "payment error is an error");
338 return paymentParts.error();
339 }
340
341 // If the payment computation completed without error, the loanSle object
342 // has been modified.
343 view.update(loanSle);
344
345 XRPL_ASSERT_PARTS(
346 // It is possible to pay 0 principal
347 paymentParts->principalPaid >= 0,
348 "xrpl::LoanPay::doApply",
349 "valid principal paid");
350 XRPL_ASSERT_PARTS(
351 // It is possible to pay 0 interest
352 paymentParts->interestPaid >= 0,
353 "xrpl::LoanPay::doApply",
354 "valid interest paid");
355 XRPL_ASSERT_PARTS(
356 // It should not be possible to pay 0 total
357 paymentParts->principalPaid + paymentParts->interestPaid > 0,
358 "xrpl::LoanPay::doApply",
359 "valid total paid");
360 XRPL_ASSERT_PARTS(
361 paymentParts->feePaid >= 0, "xrpl::LoanPay::doApply", "valid fee paid");
362
363 if (paymentParts->principalPaid < 0 || paymentParts->interestPaid < 0 ||
364 paymentParts->feePaid < 0)
365 {
366 // LCOV_EXCL_START
367 JLOG(j_.fatal()) << "Loan payment computation returned invalid values.";
368 return tecLIMIT_EXCEEDED;
369 // LCOV_EXCL_STOP
370 }
371
372 JLOG(j_.debug()) << "Loan Pay: principal paid: "
373 << paymentParts->principalPaid
374 << ", interest paid: " << paymentParts->interestPaid
375 << ", fee paid: " << paymentParts->feePaid
376 << ", value change: " << paymentParts->valueChange;
377
378 //------------------------------------------------------
379 // LoanBroker object state changes
380 view.update(brokerSle);
381
382 auto assetsAvailableProxy = vaultSle->at(sfAssetsAvailable);
383 auto assetsTotalProxy = vaultSle->at(sfAssetsTotal);
384
385 // The vault may be at a different scale than the loan. Reduce rounding
386 // errors during the payment by rounding some of the values to that scale.
387 auto const vaultScale = getAssetsTotalScale(vaultSle);
388
389 auto const totalPaidToVaultRaw =
390 paymentParts->principalPaid + paymentParts->interestPaid;
391 auto const totalPaidToVaultRounded =
392 roundToAsset(asset, totalPaidToVaultRaw, vaultScale, Number::downward);
393 XRPL_ASSERT_PARTS(
394 !asset.integral() || totalPaidToVaultRaw == totalPaidToVaultRounded,
395 "xrpl::LoanPay::doApply",
396 "rounding does nothing for integral asset");
397 // Account for value changes when reducing the broker's debt:
398 // - Positive value change (from full/late/overpayments): Subtract from the
399 // amount credited toward debt to avoid over-reducing the debt.
400 // - Negative value change (from full/overpayments): Add to the amount
401 // credited toward debt,effectively increasing the debt reduction.
402 auto const totalPaidToVaultForDebt =
403 totalPaidToVaultRaw - paymentParts->valueChange;
404
405 auto const totalPaidToBroker = paymentParts->feePaid;
406
407 XRPL_ASSERT_PARTS(
408 (totalPaidToVaultRaw + totalPaidToBroker) ==
409 (paymentParts->principalPaid + paymentParts->interestPaid +
410 paymentParts->feePaid),
411 "xrpl::LoanPay::doApply",
412 "payments add up");
413
414 // Decrease LoanBroker Debt by the amount paid, add the Loan value change
415 // (which might be negative). totalPaidToVaultForDebt may be negative,
416 // increasing the debt
417 XRPL_ASSERT_PARTS(
418 isRounded(asset, totalPaidToVaultForDebt, loanScale),
419 "xrpl::LoanPay::doApply",
420 "totalPaidToVaultForDebt rounding good");
421 // Despite our best efforts, it's possible for rounding errors to accumulate
422 // in the loan broker's debt total. This is because the broker may have more
423 // than one loan with significantly different scales.
424 adjustImpreciseNumber(
425 debtTotalProxy, -totalPaidToVaultForDebt, asset, vaultScale);
426
427 //------------------------------------------------------
428 // Vault object state changes
429 view.update(vaultSle);
430
431#if !NDEBUG
432 {
433 Number const assetsAvailableBefore = *assetsAvailableProxy;
434 Number const pseudoAccountBalanceBefore = accountHolds(
435 view,
436 vaultPseudoAccount,
437 asset,
438 FreezeHandling::fhIGNORE_FREEZE,
439 AuthHandling::ahIGNORE_AUTH,
440 j_);
441
442 XRPL_ASSERT_PARTS(
443 assetsAvailableBefore == pseudoAccountBalanceBefore,
444 "xrpl::LoanPay::doApply",
445 "vault pseudo balance agrees before");
446 }
447#endif
448
449 assetsAvailableProxy += totalPaidToVaultRounded;
450 assetsTotalProxy += paymentParts->valueChange;
451
452 XRPL_ASSERT_PARTS(
453 *assetsAvailableProxy <= *assetsTotalProxy,
454 "xrpl::LoanPay::doApply",
455 "assets available must not be greater than assets outstanding");
456
457 if (*assetsAvailableProxy > *assetsTotalProxy)
458 {
459 // LCOV_EXCL_START
460 JLOG(j_.fatal()) << "Vault assets available must not be greater "
461 "than assets outstanding. Available: "
462 << *assetsAvailableProxy
463 << ", Total: " << *assetsTotalProxy;
464 return tecINTERNAL;
465 // LCOV_EXCL_STOP
466 }
467
468 JLOG(j_.debug()) << "total paid to vault raw: " << totalPaidToVaultRaw
469 << ", total paid to vault rounded: "
470 << totalPaidToVaultRounded
471 << ", total paid to broker: " << totalPaidToBroker
472 << ", amount from transaction: " << amount;
473
474 // Move funds
475 XRPL_ASSERT_PARTS(
476 totalPaidToVaultRounded + totalPaidToBroker <= amount,
477 "xrpl::LoanPay::doApply",
478 "amount is sufficient");
479
480 if (!sendBrokerFeeToOwner)
481 {
482 // If there is not enough first-loss capital, add the fee to First Loss
483 // Cover Pool. Note that this moves the entire fee - it does not attempt
484 // to split it. The broker can Withdraw it later if they want, or leave
485 // it for future needs.
486 coverAvailableProxy += totalPaidToBroker;
487 }
488
489 associateAsset(*loanSle, asset);
490 associateAsset(*brokerSle, asset);
491 associateAsset(*vaultSle, asset);
492
493 // Duplicate some checks after rounding
494 XRPL_ASSERT_PARTS(
495 *assetsAvailableProxy <= *assetsTotalProxy,
496 "xrpl::LoanPay::doApply",
497 "assets available must not be greater than assets outstanding");
498
499#if !NDEBUG
500 auto const accountBalanceBefore = accountHolds(
501 view,
502 account_,
503 asset,
504 fhIGNORE_FREEZE,
505 ahIGNORE_AUTH,
506 j_,
507 SpendableHandling::shFULL_BALANCE);
508 auto const vaultBalanceBefore = account_ == vaultPseudoAccount
509 ? STAmount{asset, 0}
510 : accountHolds(
511 view,
512 vaultPseudoAccount,
513 asset,
514 fhIGNORE_FREEZE,
515 ahIGNORE_AUTH,
516 j_,
517 SpendableHandling::shFULL_BALANCE);
518 auto const brokerBalanceBefore = account_ == brokerPayee
519 ? STAmount{asset, 0}
520 : accountHolds(
521 view,
522 brokerPayee,
523 asset,
524 fhIGNORE_FREEZE,
525 ahIGNORE_AUTH,
526 j_,
527 SpendableHandling::shFULL_BALANCE);
528#endif
529
530 if (totalPaidToVaultRounded != beast::zero)
531 {
532 if (auto const ter = requireAuth(
533 view, asset, vaultPseudoAccount, AuthType::StrongAuth))
534 return ter;
535 }
536
537 if (totalPaidToBroker != beast::zero)
538 {
539 if (brokerPayee == account_)
540 {
541 // The broker may have deleted their holding. Recreate it if needed
542 if (auto const ter = addEmptyHolding(
543 view,
544 brokerPayee,
545 brokerPayeeSle->at(sfBalance).value().xrp(),
546 asset,
547 j_);
548 ter && ter != tecDUPLICATE)
549 // ignore tecDUPLICATE. That means the holding already exists,
550 // and is fine here
551 return ter;
552 }
553 if (auto const ter =
554 requireAuth(view, asset, brokerPayee, AuthType::StrongAuth))
555 return ter;
556 }
557
558 if (auto const ter = accountSendMulti(
559 view,
560 account_,
561 asset,
562 {{vaultPseudoAccount, totalPaidToVaultRounded},
563 {brokerPayee, totalPaidToBroker}},
564 j_,
565 WaiveTransferFee::Yes))
566 return ter;
567
568#if !NDEBUG
569 Number const assetsAvailableAfter = *assetsAvailableProxy;
570 Number const pseudoAccountBalanceAfter = accountHolds(
571 view,
572 vaultPseudoAccount,
573 asset,
574 FreezeHandling::fhIGNORE_FREEZE,
575 AuthHandling::ahIGNORE_AUTH,
576 j_);
577 XRPL_ASSERT_PARTS(
578 assetsAvailableAfter == pseudoAccountBalanceAfter,
579 "xrpl::LoanPay::doApply",
580 "vault pseudo balance agrees after");
581
582 auto const accountBalanceAfter = accountHolds(
583 view,
584 account_,
585 asset,
586 fhIGNORE_FREEZE,
587 ahIGNORE_AUTH,
588 j_,
589 SpendableHandling::shFULL_BALANCE);
590 auto const vaultBalanceAfter = account_ == vaultPseudoAccount
591 ? STAmount{asset, 0}
592 : accountHolds(
593 view,
594 vaultPseudoAccount,
595 asset,
596 fhIGNORE_FREEZE,
597 ahIGNORE_AUTH,
598 j_,
599 SpendableHandling::shFULL_BALANCE);
600 auto const brokerBalanceAfter = account_ == brokerPayee
601 ? STAmount{asset, 0}
602 : accountHolds(
603 view,
604 brokerPayee,
605 asset,
606 fhIGNORE_FREEZE,
607 ahIGNORE_AUTH,
608 j_,
609 SpendableHandling::shFULL_BALANCE);
610
611 XRPL_ASSERT_PARTS(
612 accountBalanceBefore + vaultBalanceBefore + brokerBalanceBefore ==
613 accountBalanceAfter + vaultBalanceAfter + brokerBalanceAfter,
614 "xrpl::LoanPay::doApply",
615 "funds are conserved (with rounding)");
616 XRPL_ASSERT_PARTS(
617 accountBalanceAfter >= beast::zero,
618 "xrpl::LoanPay::doApply",
619 "positive account balance");
620 XRPL_ASSERT_PARTS(
621 accountBalanceAfter < accountBalanceBefore ||
622 account_ == asset.getIssuer(),
623 "xrpl::LoanPay::doApply",
624 "account balance decreased");
625 XRPL_ASSERT_PARTS(
626 vaultBalanceAfter >= beast::zero && brokerBalanceAfter >= beast::zero,
627 "xrpl::LoanPay::doApply",
628 "positive vault and broker balances");
629 XRPL_ASSERT_PARTS(
630 vaultBalanceAfter >= vaultBalanceBefore,
631 "xrpl::LoanPay::doApply",
632 "vault balance did not decrease");
633 XRPL_ASSERT_PARTS(
634 brokerBalanceAfter >= brokerBalanceBefore,
635 "xrpl::LoanPay::doApply",
636 "broker balance did not decrease");
637 XRPL_ASSERT_PARTS(
638 vaultBalanceAfter > vaultBalanceBefore ||
639 brokerBalanceAfter > brokerBalanceBefore,
640 "xrpl::LoanPay::doApply",
641 "vault and/or broker balance increased");
642#endif
643
644 return tesSUCCESS;
645}
