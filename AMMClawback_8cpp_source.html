22NotTEC
23AMMClawback::preflight(PreflightContext const& ctx)
24{
25 AccountID const issuer = ctx.tx[sfAccount];
26 AccountID const holder = ctx.tx[sfHolder];
27
28 if (issuer == holder)
29 {
30 JLOG(ctx.j.trace()) << "AMMClawback: holder cannot be the same as issuer.";
31 return temMALFORMED;
32 }
33
34 std::optional<STAmount> const clawAmount = ctx.tx[~sfAmount];
35 auto const asset = ctx.tx[sfAsset].get<Issue>();
36 auto const asset2 = ctx.tx[sfAsset2].get<Issue>();
37
38 if (isXRP(asset))
39 return temMALFORMED;
40
41 auto const flags = ctx.tx.getFlags();
42
43 if (flags & tfClawTwoAssets && asset.account != asset2.account)
44 {
45 JLOG(ctx.j.trace()) << "AMMClawback: tfClawTwoAssets can only be enabled when two "
46 "assets in the AMM pool are both issued by the issuer";
47 return temINVALID_FLAG;
48 }
49
50 if (asset.account != issuer)
51 {
52 JLOG(ctx.j.trace()) << "AMMClawback: Asset's account does not "
53 "match Account field.";
54 return temMALFORMED;
55 }
56
57 if (clawAmount && clawAmount->get<Issue>() != asset)
58 {
59 JLOG(ctx.j.trace()) << "AMMClawback: Amount's issuer/currency subfield "
60 "does not match Asset field";
61 return temBAD_AMOUNT;
62 }
63
64 if (clawAmount && *clawAmount <= beast::zero)
65 return temBAD_AMOUNT;
66
67 return tesSUCCESS;
68}

71AMMClawback::preclaim(PreclaimContext const& ctx)
72{
73 auto const asset = ctx.tx[sfAsset].get<Issue>();
74 auto const asset2 = ctx.tx[sfAsset2].get<Issue>();
75 auto const sleIssuer = ctx.view.read(keylet::account(ctx.tx[sfAccount]));
76 if (!sleIssuer)
77 return terNO_ACCOUNT; // LCOV_EXCL_LINE
78
79 if (!ctx.view.read(keylet::account(ctx.tx[sfHolder])))
80 return terNO_ACCOUNT;
81
82 auto const ammSle = ctx.view.read(keylet::amm(asset, asset2));
83 if (!ammSle)
84 {
85 JLOG(ctx.j.debug()) << "AMM Clawback: Invalid asset pair.";
86 return terNO_AMM;
87 }
88
89 std::uint32_t const issuerFlagsIn = sleIssuer->getFieldU32(sfFlags);
90
91 // If AllowTrustLineClawback is not set or NoFreeze is set, return no
92 // permission
93 if (!(issuerFlagsIn & lsfAllowTrustLineClawback) || (issuerFlagsIn & lsfNoFreeze))
94 return tecNO_PERMISSION;
95
96 return tesSUCCESS;
97}

111TER
112AMMClawback::applyGuts(Sandbox& sb)
113{
114 std::optional<STAmount> const clawAmount = ctx_.tx[~sfAmount];
115 AccountID const issuer = ctx_.tx[sfAccount];
116 AccountID const holder = ctx_.tx[sfHolder];
117 Issue const asset = ctx_.tx[sfAsset].get<Issue>();
118 Issue const asset2 = ctx_.tx[sfAsset2].get<Issue>();
119
120 auto ammSle = sb.peek(keylet::amm(asset, asset2));
121 if (!ammSle)
122 return tecINTERNAL; // LCOV_EXCL_LINE
123
124 auto const ammAccount = (*ammSle)[sfAccount];
125 auto const accountSle = sb.read(keylet::account(ammAccount));
126 if (!accountSle)
127 return tecINTERNAL; // LCOV_EXCL_LINE
128
129 if (sb.rules().enabled(fixAMMClawbackRounding))
130 {
131 // retrieve LP token balance inside the amendment gate to avoid
132 // inconsistent error behavior
133 auto const lpTokenBalance = ammLPHolds(sb, *ammSle, holder, j_);
134 if (lpTokenBalance == beast::zero)
135 return tecAMM_BALANCE;
136
137 if (auto const res = verifyAndAdjustLPTokenBalance(sb, lpTokenBalance, ammSle, holder); !res)
138 return res.error(); // LCOV_EXCL_LINE
139 }
140
141 auto const expected = ammHolds(sb, *ammSle, asset, asset2, FreezeHandling::fhIGNORE_FREEZE, ctx_.journal);
142
143 if (!expected)
144 return expected.error(); // LCOV_EXCL_LINE
145 auto const [amountBalance, amount2Balance, lptAMMBalance] = *expected;
146
147 TER result;
148 STAmount newLPTokenBalance;
149 STAmount amountWithdraw;
150 std::optional<STAmount> amount2Withdraw;
151
152 auto const holdLPtokens = ammLPHolds(sb, *ammSle, holder, j_);
153 if (holdLPtokens == beast::zero)
154 return tecAMM_BALANCE;
155
156 if (!clawAmount)
157 // Because we are doing a two-asset withdrawal,
158 // tfee is actually not used, so pass tfee as 0.
159 std::tie(result, newLPTokenBalance, amountWithdraw, amount2Withdraw) = AMMWithdraw::equalWithdrawTokens(
160 sb,
161 *ammSle,
162 holder,
163 ammAccount,
164 amountBalance,
165 amount2Balance,
166 lptAMMBalance,
167 holdLPtokens,
168 holdLPtokens,
169 0,
170 FreezeHandling::fhIGNORE_FREEZE,
171 WithdrawAll::Yes,
172 mPriorBalance,
173 ctx_.journal);
174 else
175 std::tie(result, newLPTokenBalance, amountWithdraw, amount2Withdraw) = equalWithdrawMatchingOneAmount(
176 sb, *ammSle, holder, ammAccount, amountBalance, amount2Balance, lptAMMBalance, holdLPtokens, *clawAmount);
177
178 if (result != tesSUCCESS)
179 return result; // LCOV_EXCL_LINE
180
181 auto const res = AMMWithdraw::deleteAMMAccountIfEmpty(sb, ammSle, newLPTokenBalance, asset, asset2, j_);
182 if (!res.second)
183 return res.first; // LCOV_EXCL_LINE
184
185 JLOG(ctx_.journal.trace()) << "AMM Withdraw during AMMClawback: lptoken new balance: "
186 << to_string(newLPTokenBalance.iou())
187 << " old balance: " << to_string(lptAMMBalance.iou());
188
189 auto const ter = rippleCredit(sb, holder, issuer, amountWithdraw, true, j_);
190 if (ter != tesSUCCESS)
191 return ter; // LCOV_EXCL_LINE
192
193 // if the issuer issues both assets and sets flag tfClawTwoAssets, we
194 // will claw the paired asset as well. We already checked if
195 // tfClawTwoAssets is enabled, the two assets have to be issued by the
196 // same issuer.
197 if (!amount2Withdraw)
198 return tecINTERNAL; // LCOV_EXCL_LINE
199
200 auto const flags = ctx_.tx.getFlags();
201 if (flags & tfClawTwoAssets)
202 return rippleCredit(sb, holder, issuer, *amount2Withdraw, true, j_);
203
204 return tesSUCCESS;
205}

207std::tuple<TER, STAmount, STAmount, std::optional<STAmount>>
208AMMClawback::equalWithdrawMatchingOneAmount(
209 Sandbox& sb,
210 SLE const& ammSle,
211 AccountID const& holder,
212 AccountID const& ammAccount,
213 STAmount const& amountBalance,
214 STAmount const& amount2Balance,
215 STAmount const& lptAMMBalance,
216 STAmount const& holdLPtokens,
217 STAmount const& amount)
218{
219 auto frac = Number{amount} / amountBalance;
220 auto amount2Withdraw = amount2Balance * frac;
221
222 auto const lpTokensWithdraw = toSTAmount(lptAMMBalance.issue(), lptAMMBalance * frac);
223
224 if (lpTokensWithdraw > holdLPtokens)
225 // if lptoken balance less than what the issuer intended to clawback,
226 // clawback all the tokens. Because we are doing a two-asset withdrawal,
227 // tfee is actually not used, so pass tfee as 0.
228 return AMMWithdraw::equalWithdrawTokens(
229 sb,
230 ammSle,
231 holder,
232 ammAccount,
233 amountBalance,
234 amount2Balance,
235 lptAMMBalance,
236 holdLPtokens,
237 holdLPtokens,
238 0,
239 FreezeHandling::fhIGNORE_FREEZE,
240 WithdrawAll::Yes,
241 mPriorBalance,
242 ctx_.journal);
243
244 auto const& rules = sb.rules();
245 if (rules.enabled(fixAMMClawbackRounding))
246 {
247 auto tokensAdj = getRoundedLPTokens(rules, lptAMMBalance, frac, IsDeposit::No);
248
249 // LCOV_EXCL_START
250 if (tokensAdj == beast::zero)
251 return {tecAMM_INVALID_TOKENS, STAmount{}, STAmount{}, std::nullopt};
252 // LCOV_EXCL_STOP
253
254 frac = adjustFracByTokens(rules, lptAMMBalance, tokensAdj, frac);
255 auto amount2Rounded = getRoundedAsset(rules, amount2Balance, frac, IsDeposit::No);
256
257 auto amountRounded = getRoundedAsset(rules, amountBalance, frac, IsDeposit::No);
258
259 return AMMWithdraw::withdraw(
260 sb,
261 ammSle,
262 ammAccount,
263 holder,
264 amountBalance,
265 amountRounded,
266 amount2Rounded,
267 lptAMMBalance,
268 tokensAdj,
269 0,
270 FreezeHandling::fhIGNORE_FREEZE,
271 WithdrawAll::No,
272 mPriorBalance,
273 ctx_.journal);
274 }
275
276 // Because we are doing a two-asset withdrawal,
277 // tfee is actually not used, so pass tfee as 0.
278 return AMMWithdraw::withdraw(
279 sb,
280 ammSle,
281 ammAccount,
282 holder,
283 amountBalance,
284 amount,
285 toSTAmount(amount2Balance.issue(), amount2Withdraw),
286 lptAMMBalance,
287 toSTAmount(lptAMMBalance.issue(), lptAMMBalance * frac),
288 0,
289 FreezeHandling::fhIGNORE_FREEZE,
290 WithdrawAll::No,
291 mPriorBalance,
292 ctx_.journal);
293}
