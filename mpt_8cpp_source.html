126void
127MPTTester::create(MPTCreate const& arg)
128{
129 if (id_)
130 Throw<std::runtime_error>("MPT can't be reused");
131 id_ = makeMptID(env_.seq(issuer_), issuer_);
132 Json::Value jv = createJV(
133 {.issuer = issuer_,
134 .maxAmt = arg.maxAmt,
135 .assetScale = arg.assetScale,
136 .transferFee = arg.transferFee,
137 .metadata = arg.metadata,
138 .mutableFlags = arg.mutableFlags,
139 .domainID = arg.domainID});
140 if (submit(arg, jv) != tesSUCCESS)
141 {
142 // Verify issuance doesn't exist
143 env_.require(requireAny([&]() -> bool { return env_.le(keylet::mptIssuance(*id_)) == nullptr; }));
144
145 id_.reset();
146 }
147 else
148 {
149 env_.require(mptflags(*this, arg.flags.value_or(0)));
150 auto authAndPay = [&](auto const& accts, auto const&& getAcct) {
151 for (auto const& it : accts)
152 {
153 authorize({.account = getAcct(it)});
154 if ((arg.flags.value_or(0) & tfMPTRequireAuth) && arg.authHolder)
155 authorize({.account = issuer_, .holder = getAcct(it)});
156 if (arg.pay && arg.pay->first.empty())
157 pay(issuer_, getAcct(it), arg.pay->second);
158 }
159 if (arg.pay)
160 {
161 for (auto const& p : arg.pay->first)
162 pay(issuer_, p, arg.pay->second);
163 }
164 };
165 if (arg.authorize)
166 {
167 if (arg.authorize->empty())
168 authAndPay(holders_, [](auto const& it) { return it.second; });
169 else
170 authAndPay(*arg.authorize, [](auto const& it) { return it; });
171 }
172 else if (arg.pay)
173 {
174 if (arg.pay->first.empty())
175 authAndPay(holders_, [](auto const& it) { return it.second; });
176 else
177 authAndPay(arg.pay->first, [](auto const& it) { return it; });
178 }
179 }
180}

228void
229MPTTester::authorize(MPTAuthorize const& arg)
230{
231 if (!arg.id && !id_)
232 Throw<std::runtime_error>("MPT has not been created");
233 Json::Value jv = authorizeJV({
234 .account = arg.account ? arg.account : issuer_,
235 .holder = arg.holder,
236 .id = arg.id ? arg.id : id_,
237 });
238 if (auto const result = submit(arg, jv); result == tesSUCCESS)
239 {
240 // Issuer authorizes
241 if (!arg.account || *arg.account == issuer_)
242 {
243 auto const flags = getFlags(arg.holder);
244 // issuer un-authorizes the holder
245 if (arg.flags.value_or(0) == tfMPTUnauthorize)
246 env_.require(mptflags(*this, flags, arg.holder));
247 // issuer authorizes the holder
248 else
249 env_.require(mptflags(*this, flags | lsfMPTAuthorized, arg.holder));
250 }
251 // Holder authorizes
252 else if (arg.flags.value_or(0) != tfMPTUnauthorize)
253 {
254 auto const flags = getFlags(arg.account);
255 // holder creates a token
256 env_.require(mptflags(*this, flags, arg.account));
257 env_.require(mptbalance(*this, *arg.account, 0));
258 }
259 else
260 {
261 // Verify that the MPToken doesn't exist.
262 forObject([&](SLEP const& sle) { return env_.test.BEAST_EXPECT(!sle); }, arg.account);
263 }
264 }
265 else if (arg.account && *arg.account != issuer_ && arg.flags.value_or(0) != tfMPTUnauthorize && id_)
266 {
267 if (result == tecDUPLICATE)
268 {
269 // Verify that MPToken already exists
270 env_.require(
271 requireAny([&]() -> bool { return env_.le(keylet::mptoken(*id_, arg.account->id())) != nullptr; }));
272 }
273 else
274 {
275 // Verify MPToken doesn't exist if holder failed authorizing(unless
276 // it already exists)
277 env_.require(
278 requireAny([&]() -> bool { return env_.le(keylet::mptoken(*id_, arg.account->id())) == nullptr; }));
279 }
280 }

327void
328MPTTester::set(MPTSet const& arg)
329{
330 if (!arg.id && !id_)
331 Throw<std::runtime_error>("MPT has not been created");
332 Json::Value jv = setJV(
333 {.account = arg.account ? arg.account : issuer_,
334 .holder = arg.holder,
335 .id = arg.id ? arg.id : id_,
336 .mutableFlags = arg.mutableFlags,
337 .transferFee = arg.transferFee,
338 .metadata = arg.metadata,
339 .delegate = arg.delegate,
340 .domainID = arg.domainID});
341 if (submit(arg, jv) == tesSUCCESS && (arg.flags.value_or(0) || arg.mutableFlags))
342 {
343 auto require = [&](std::optional<Account> const& holder, bool unchanged) {
344 auto flags = getFlags(holder);
345 if (!unchanged)
346 {
347 if (arg.flags)
348 {
349 if (*arg.flags & tfMPTLock)
350 flags |= lsfMPTLocked;
351 else if (*arg.flags & tfMPTUnlock)
352 flags &= ~lsfMPTLocked;
353 }
354
355 if (arg.mutableFlags)
356 {
357 if (*arg.mutableFlags & tmfMPTSetCanLock)
358 flags |= lsfMPTCanLock;
359 else if (*arg.mutableFlags & tmfMPTClearCanLock)
360 flags &= ~lsfMPTCanLock;
361
362 if (*arg.mutableFlags & tmfMPTSetRequireAuth)
363 flags |= lsfMPTRequireAuth;
364 else if (*arg.mutableFlags & tmfMPTClearRequireAuth)
365 flags &= ~lsfMPTRequireAuth;
366
367 if (*arg.mutableFlags & tmfMPTSetCanEscrow)
368 flags |= lsfMPTCanEscrow;
369 else if (*arg.mutableFlags & tmfMPTClearCanEscrow)
370 flags &= ~lsfMPTCanEscrow;
371
372 if (*arg.mutableFlags & tmfMPTSetCanClawback)
373 flags |= lsfMPTCanClawback;
374 else if (*arg.mutableFlags & tmfMPTClearCanClawback)
375 flags &= ~lsfMPTCanClawback;
376
377 if (*arg.mutableFlags & tmfMPTSetCanTrade)
378 flags |= lsfMPTCanTrade;
379 else if (*arg.mutableFlags & tmfMPTClearCanTrade)
380 flags &= ~lsfMPTCanTrade;
381
382 if (*arg.mutableFlags & tmfMPTSetCanTransfer)
383 flags |= lsfMPTCanTransfer;
384 else if (*arg.mutableFlags & tmfMPTClearCanTransfer)
385 flags &= ~lsfMPTCanTransfer;
386 }
387 }
388 env_.require(mptflags(*this, flags, holder));
389 };
390 if (arg.account)
391 require(std::nullopt, arg.holder.has_value());
392 if (auto const account = (arg.holder ? std::get_if<Account>(&(*arg.holder)) : nullptr))
393 require(*account, false);
394 }

468void
469MPTTester::pay(
470 Account const& src,
471 Account const& dest,
472 std::int64_t amount,
473 std::optional<TER> err,
474 std::optional<std::vector<std::string>> credentials)
475{
476 if (!id_)
477 Throw<std::runtime_error>("MPT has not been created");
478 auto const srcAmt = getBalance(src);
479 auto const destAmt = getBalance(dest);
480 auto const outstandingAmt = getBalance(issuer_);
481
482 if (credentials)
483 env_(jtx::pay(src, dest, mpt(amount)), ter(err.value_or(tesSUCCESS)), credentials::ids(*credentials));
484 else
485 env_(jtx::pay(src, dest, mpt(amount)), ter(err.value_or(tesSUCCESS)));
486
487 if (env_.ter() != tesSUCCESS)
488 amount = 0;
489 if (close_)
490 env_.close();
491 if (src == issuer_)
492 {
493 env_.require(mptbalance(*this, src, srcAmt + amount));
494 env_.require(mptbalance(*this, dest, destAmt + amount));
495 }
496 else if (dest == issuer_)
497 {
498 env_.require(mptbalance(*this, src, srcAmt - amount));
499 env_.require(mptbalance(*this, dest, destAmt - amount));
500 }
501 else
502 {
503 STAmount const saAmount = {*id_, amount};
504 auto const actual = multiply(saAmount, transferRate(*env_.current(), *id_)).mpt().value();
505 // Sender pays the transfer fee if any
506 env_.require(mptbalance(*this, src, srcAmt - actual));
507 env_.require(mptbalance(*this, dest, destAmt + amount));
508 // Outstanding amount is reduced by the transfer fee if any
509 env_.require(mptbalance(*this, issuer_, outstandingAmt - (actual - amount)));
510 }
511}

514MPTTester::claw(Account const& issuer, Account const& holder, std::int64_t amount, std::optional<TER> err)
515{

520 env_(jtx::claw(issuer, mpt(amount), holder), ter(err.value_or(tesSUCCESS)));
521 if (env_.ter() != tesSUCCESS)

523 if (close_)
