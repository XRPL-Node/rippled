37TER
38VaultWithdraw::preclaim(PreclaimContext const& ctx)
39{
40 auto const vault = ctx.view.read(keylet::vault(ctx.tx[sfVaultID]));
41 if (!vault)
42 return tecNO_ENTRY;
43
44 auto const assets = ctx.tx[sfAmount];
45 auto const vaultAsset = vault->at(sfAsset);
46 auto const vaultShare = vault->at(sfShareMPTID);
47 if (assets.asset() != vaultAsset && assets.asset() != vaultShare)
48 return tecWRONG_ASSET;
49
50 auto const& vaultAccount = vault->at(sfAccount);
51 auto const& account = ctx.tx[sfAccount];
52 auto const& dstAcct = ctx.tx[~sfDestination].value_or(account);
53 if (auto ter = canTransfer(ctx.view, vaultAsset, vaultAccount, dstAcct); !isTesSuccess(ter))
54 {
55 JLOG(ctx.j.debug()) << "VaultWithdraw: vault assets are non-transferable.";
56 return ter;
57 }
58
59 // Enforce valid withdrawal policy
60 if (vault->at(sfWithdrawalPolicy) != vaultStrategyFirstComeFirstServe)
61 {
62 // LCOV_EXCL_START
63 JLOG(ctx.j.error()) << "VaultWithdraw: invalid withdrawal policy.";
64 return tefINTERNAL;
65 // LCOV_EXCL_STOP
66 }
67
68 if (auto const ret = canWithdraw(ctx.view, ctx.tx))
69 return ret;
70
71 // If sending to Account (i.e. not a transfer), we will also create (only
72 // if authorized) a trust line or MPToken as needed, in doApply().
73 // Destination MPToken or trust line must exist if _not_ sending to Account.
74 AuthType const authType = account == dstAcct ? AuthType::WeakAuth : AuthType::StrongAuth;
75 if (auto const ter = requireAuth(ctx.view, vaultAsset, dstAcct, authType); !isTesSuccess(ter))
76 return ter;
77
78 // Cannot withdraw from a Vault an Asset frozen for the destination account
79 if (auto const ret = checkFrozen(ctx.view, dstAcct, vaultAsset))
80 return ret;
81
82 // Cannot return shares to the vault, if the underlying asset was frozen for
83 // the submitter
84 if (auto const ret = checkFrozen(ctx.view, account, vaultShare))
85 return ret;
86
87 return tesSUCCESS;
88}

90TER
91VaultWithdraw::doApply()
92{
93 auto const vault = view().peek(keylet::vault(ctx_.tx[sfVaultID]));
94 if (!vault)
95 return tefINTERNAL; // LCOV_EXCL_LINE
96
97 auto const mptIssuanceID = *((*vault)[sfShareMPTID]);
98 auto const sleIssuance = view().read(keylet::mptIssuance(mptIssuanceID));
99 if (!sleIssuance)
100 {
101 // LCOV_EXCL_START
102 JLOG(j_.error()) << "VaultWithdraw: missing issuance of vault shares.";
103 return tefINTERNAL;
104 // LCOV_EXCL_STOP
105 }
106
107 // Note, we intentionally do not check lsfVaultPrivate flag on the Vault. If
108 // you have a share in the vault, it means you were at some point authorized
109 // to deposit into it, and this means you are also indefinitely authorized
110 // to withdraw from it.
111
112 auto const amount = ctx_.tx[sfAmount];
113 Asset const vaultAsset = vault->at(sfAsset);
114
115 MPTIssue const share{mptIssuanceID};
116 STAmount sharesRedeemed = {share};
117 STAmount assetsWithdrawn;
118 try
119 {
120 if (amount.asset() == vaultAsset)
121 {
122 // Fixed assets, variable shares.
123 {
124 auto const maybeShares = assetsToSharesWithdraw(vault, sleIssuance, amount);
125 if (!maybeShares)
126 return tecINTERNAL; // LCOV_EXCL_LINE
127 sharesRedeemed = *maybeShares;
128 }
129
130 if (sharesRedeemed == beast::zero)
131 return tecPRECISION_LOSS;
132 auto const maybeAssets = sharesToAssetsWithdraw(vault, sleIssuance, sharesRedeemed);
133 if (!maybeAssets)
134 return tecINTERNAL; // LCOV_EXCL_LINE
135 assetsWithdrawn = *maybeAssets;
136 }
137 else if (amount.asset() == share)
138 {
139 // Fixed shares, variable assets.
140 sharesRedeemed = amount;
141 auto const maybeAssets = sharesToAssetsWithdraw(vault, sleIssuance, sharesRedeemed);
142 if (!maybeAssets)
143 return tecINTERNAL; // LCOV_EXCL_LINE
144 assetsWithdrawn = *maybeAssets;
145 }
146 else
147 return tefINTERNAL; // LCOV_EXCL_LINE
148 }
149 catch (std::overflow_error const&)
150 {
151 // It's easy to hit this exception from Number with large enough Scale
152 // so we avoid spamming the log and only use debug here.
153 JLOG(j_.debug()) //
154 << "VaultWithdraw: overflow error with"
155 << " scale=" << (int)vault->at(sfScale).value() //
156 << ", assetsTotal=" << vault->at(sfAssetsTotal).value()
157 << ", sharesTotal=" << sleIssuance->at(sfOutstandingAmount)
158 << ", amount=" << amount.value();
159 return tecPATH_DRY;
160 }
161
162 if (accountHolds(
163 view(),
164 account_,
165 share,
166 FreezeHandling::fhZERO_IF_FROZEN,
167 AuthHandling::ahIGNORE_AUTH,
168 j_) < sharesRedeemed)
169 {
170 JLOG(j_.debug()) << "VaultWithdraw: account doesn't hold enough shares";
171 return tecINSUFFICIENT_FUNDS;
172 }
173
174 auto assetsAvailable = vault->at(sfAssetsAvailable);
175 auto assetsTotal = vault->at(sfAssetsTotal);
176 [[maybe_unused]] auto const lossUnrealized = vault->at(sfLossUnrealized);
177 XRPL_ASSERT(
178 lossUnrealized <= (assetsTotal - assetsAvailable),
179 "xrpl::VaultWithdraw::doApply : loss and assets do balance");
180
181 // The vault must have enough assets on hand. The vault may hold assets
182 // that it has already pledged. That is why we look at AssetAvailable
183 // instead of the pseudo-account balance.
184 if (*assetsAvailable < assetsWithdrawn)
185 {
186 JLOG(j_.debug()) << "VaultWithdraw: vault doesn't hold enough assets";
187 return tecINSUFFICIENT_FUNDS;
188 }
189
190 assetsTotal -= assetsWithdrawn;
191 assetsAvailable -= assetsWithdrawn;
192 view().update(vault);
193
194 auto const& vaultAccount = vault->at(sfAccount);
195 // Transfer shares from depositor to vault.
196 if (auto const ter =
197 accountSend(view(), account_, vaultAccount, sharesRedeemed, j_, WaiveTransferFee::Yes);
198 !isTesSuccess(ter))
199 return ter;
200
201 // Try to remove MPToken for shares, if the account balance is zero. Vault
202 // pseudo-account will never set lsfMPTAuthorized, so we ignore flags.
203 // Keep MPToken if holder is the vault owner.
204 if (account_ != vault->at(sfOwner))
205 {
206 if (auto const ter = removeEmptyHolding(view(), account_, sharesRedeemed.asset(), j_);
207 isTesSuccess(ter))
208 {
209 JLOG(j_.debug()) //
210 << "VaultWithdraw: removed empty MPToken for vault shares"
211 << " MPTID=" << to_string(mptIssuanceID) //
212 << " account=" << toBase58(account_);
213 }
214 else if (ter != tecHAS_OBLIGATIONS)
215 {
216 // LCOV_EXCL_START
217 JLOG(j_.error()) //
218 << "VaultWithdraw: failed to remove MPToken for vault shares"
219 << " MPTID=" << to_string(mptIssuanceID) //
220 << " account=" << toBase58(account_) //
221 << " with result: " << transToken(ter);
222 return ter;
223 // LCOV_EXCL_STOP
224 }
225 // else quietly ignore, account balance is not zero
226 }
227
228 auto const dstAcct = ctx_.tx[~sfDestination].value_or(account_);
229
230 associateAsset(*vault, vaultAsset);
231
232 return doWithdraw(
233 view(), ctx_.tx, account_, dstAcct, vaultAccount, mPriorBalance, assetsWithdrawn, j_);
234}
