227bool
228transResultInfo(TER code, std::string& token, std::string& text)
229{
230 auto& results = transResults();
231
232 auto const r = results.find(TERtoInt(code));
233
234 if (r == results.end())
235 return false;
236
237 token = r->second.first;
238 text = r->second.second;
239 return true;
240}

260std::optional<TER>
261transCode(std::string const& token)
262{
263 static auto const results = [] {
264 auto& byTer = transResults();
265 auto range = boost::make_iterator_range(byTer.begin(), byTer.end());
266 auto tRange = boost::adaptors::transform(
267 range, [](auto const& r) { return std::make_pair(r.second.first, r.first); });
268 std::unordered_map<std::string, TERUnderlyingType> const byToken(
269 tRange.begin(), tRange.end());
270 return byToken;
271 }();
272
273 auto const r = results.find(token);
274
275 if (r == results.end())
276 return std::nullopt;
277
278 return TER::fromInt(r->second);
279}
