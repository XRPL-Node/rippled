12CreateCheck::preflight(PreflightContext const& ctx)
13{
14 if (ctx.tx[sfAccount] == ctx.tx[sfDestination])
15 {
16 // They wrote a check to themselves.
17 JLOG(ctx.j.warn()) << "Malformed transaction: Check to self.";
18 return temREDUNDANT;
19 }
20
21 {
22 STAmount const sendMax{ctx.tx.getFieldAmount(sfSendMax)};
23 if (!isLegalNet(sendMax) || sendMax.signum() <= 0)
24 {
25 JLOG(ctx.j.warn()) << "Malformed transaction: bad sendMax amount: " << sendMax.getFullText();
26 return temBAD_AMOUNT;
27 }
28
29 if (badCurrency() == sendMax.getCurrency())
30 {
31 JLOG(ctx.j.warn()) << "Malformed transaction: Bad currency.";
32 return temBAD_CURRENCY;
33 }
34 }
35
36 if (auto const optExpiry = ctx.tx[~sfExpiration])
37 {
38 if (*optExpiry == 0)
39 {
40 JLOG(ctx.j.warn()) << "Malformed transaction: bad expiration";
41 return temBAD_EXPIRATION;
42 }
43 }
44
45 return tesSUCCESS;
46}

49CreateCheck::preclaim(PreclaimContext const& ctx)
50{
51 AccountID const dstId{ctx.tx[sfDestination]};
52 auto const sleDst = ctx.view.read(keylet::account(dstId));
53 if (!sleDst)
54 {
55 JLOG(ctx.j.warn()) << "Destination account does not exist.";
56 return tecNO_DST;
57 }
58
59 auto const flags = sleDst->getFlags();
60
61 // Check if the destination has disallowed incoming checks
62 if (flags & lsfDisallowIncomingCheck)
63 return tecNO_PERMISSION;
64
65 // Pseudo-accounts cannot cash checks. Note, this is not amendment-gated
66 // because all writes to pseudo-account discriminator fields **are**
67 // amendment gated, hence the behaviour of this check will always match the
68 // currently active amendments.
69 if (isPseudoAccount(sleDst))
70 return tecNO_PERMISSION;
71
72 if ((flags & lsfRequireDestTag) && !ctx.tx.isFieldPresent(sfDestinationTag))
73 {
74 // The tag is basically account-specific information we don't
75 // understand, but we can require someone to fill it in.
76 JLOG(ctx.j.warn()) << "Malformed transaction: DestinationTag required.";
77 return tecDST_TAG_NEEDED;
78 }
79
80 {
81 STAmount const sendMax{ctx.tx[sfSendMax]};
82 if (!sendMax.native())
83 {
84 // The currency may not be globally frozen
85 AccountID const& issuerId{sendMax.getIssuer()};
86 if (isGlobalFrozen(ctx.view, issuerId))
87 {
88 JLOG(ctx.j.warn()) << "Creating a check for frozen asset";
89 return tecFROZEN;
90 }
91 // If this account has a trustline for the currency, that
92 // trustline may not be frozen.
93 //
94 // Note that we DO allow create check for a currency that the
95 // account does not yet have a trustline to.
96 AccountID const srcId{ctx.tx.getAccountID(sfAccount)};
97 if (issuerId != srcId)
98 {
99 // Check if the issuer froze the line
100 auto const sleTrust = ctx.view.read(keylet::line(srcId, issuerId, sendMax.getCurrency()));
101 if (sleTrust && sleTrust->isFlag((issuerId > srcId) ? lsfHighFreeze : lsfLowFreeze))
102 {
103 JLOG(ctx.j.warn()) << "Creating a check for frozen trustline.";
104 return tecFROZEN;
105 }
106 }
107 if (issuerId != dstId)
108 {
109 // Check if dst froze the line.
110 auto const sleTrust = ctx.view.read(keylet::line(issuerId, dstId, sendMax.getCurrency()));
111 if (sleTrust && sleTrust->isFlag((dstId > issuerId) ? lsfHighFreeze : lsfLowFreeze))
112 {
113 JLOG(ctx.j.warn()) << "Creating a check for destination frozen trustline.";
114 return tecFROZEN;
115 }
116 }
117 }
118 }
119 if (hasExpired(ctx.view, ctx.tx[~sfExpiration]))
120 {
121 JLOG(ctx.j.warn()) << "Creating a check that has already expired.";
122 return tecEXPIRED;
123 }
124 return tesSUCCESS;
125}

127TER
128CreateCheck::doApply()
129{
130 auto const sle = view().peek(keylet::account(account_));
131 if (!sle)
132 return tefINTERNAL; // LCOV_EXCL_LINE
133
134 // A check counts against the reserve of the issuing account, but we
135 // check the starting balance because we want to allow dipping into the
136 // reserve to pay fees.
137 {
138 STAmount const reserve{view().fees().accountReserve(sle->getFieldU32(sfOwnerCount) + 1)};
139
140 if (mPriorBalance < reserve)
141 return tecINSUFFICIENT_RESERVE;
142 }
143
144 // Note that we use the value from the sequence or ticket as the
145 // Check sequence. For more explanation see comments in SeqProxy.h.
146 std::uint32_t const seq = ctx_.tx.getSeqValue();
147 Keylet const checkKeylet = keylet::check(account_, seq);
148 auto sleCheck = std::make_shared<SLE>(checkKeylet);
149
150 sleCheck->setAccountID(sfAccount, account_);
151 AccountID const dstAccountId = ctx_.tx[sfDestination];
152 sleCheck->setAccountID(sfDestination, dstAccountId);
153 sleCheck->setFieldU32(sfSequence, seq);
154 sleCheck->setFieldAmount(sfSendMax, ctx_.tx[sfSendMax]);
155 if (auto const srcTag = ctx_.tx[~sfSourceTag])
156 sleCheck->setFieldU32(sfSourceTag, *srcTag);
157 if (auto const dstTag = ctx_.tx[~sfDestinationTag])
158 sleCheck->setFieldU32(sfDestinationTag, *dstTag);
159 if (auto const invoiceId = ctx_.tx[~sfInvoiceID])
160 sleCheck->setFieldH256(sfInvoiceID, *invoiceId);
161 if (auto const expiry = ctx_.tx[~sfExpiration])
162 sleCheck->setFieldU32(sfExpiration, *expiry);
163
164 view().insert(sleCheck);
165
166 auto viewJ = ctx_.registry.journal("View");
167 // If it's not a self-send (and it shouldn't be), add Check to the
168 // destination's owner directory.
169 if (dstAccountId != account_)
170 {
171 auto const page = view().dirInsert(keylet::ownerDir(dstAccountId), checkKeylet, describeOwnerDir(dstAccountId));
172
173 JLOG(j_.trace()) << "Adding Check to destination directory " << to_string(checkKeylet.key) << ": "
174 << (page ? "success" : "failure");
175
176 if (!page)
177 return tecDIR_FULL; // LCOV_EXCL_LINE
178
179 sleCheck->setFieldU64(sfDestinationNode, *page);
180 }
181
182 {
183 auto const page = view().dirInsert(keylet::ownerDir(account_), checkKeylet, describeOwnerDir(account_));
184
185 JLOG(j_.trace()) << "Adding Check to owner directory " << to_string(checkKeylet.key) << ": "
186 << (page ? "success" : "failure");
187
188 if (!page)
189 return tecDIR_FULL; // LCOV_EXCL_LINE
190
191 sleCheck->setFieldU64(sfOwnerNode, *page);
192 }
193 // If we succeeded, the new entry counts against the creator's reserve.
194 adjustOwnerCount(view(), sle, 1, viewJ);
195 return tesSUCCESS;
196}
