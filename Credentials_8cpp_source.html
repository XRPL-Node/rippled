14/*
15 Credentials
16 ======
17
18 A verifiable credentials (VC
19 https://en.wikipedia.org/wiki/Verifiable_credentials), as defined by the W3C
20 specification (https://www.w3.org/TR/vc-data-model-2.0/), is a
21 secure and tamper-evident way to represent information about a subject, such
22 as an individual, organization, or even an IoT device. These credentials are
23 issued by a trusted entity and can be verified by third parties without
24 directly involving the issuer at all.
25*/

39CredentialCreate::preflight(PreflightContext const& ctx)
40{
41 auto const& tx = ctx.tx;
42 auto& j = ctx.j;
43
44 if (!tx[sfSubject])
45 {
46 JLOG(j.trace()) << "Malformed transaction: Invalid Subject";
47 return temMALFORMED;
48 }
49
50 auto const uri = tx[~sfURI];
51 if (uri && (uri->empty() || (uri->size() > maxCredentialURILength)))
52 {
53 JLOG(j.trace()) << "Malformed transaction: invalid size of URI.";
54 return temMALFORMED;
55 }
56
57 auto const credType = tx[sfCredentialType];
58 if (credType.empty() || (credType.size() > maxCredentialTypeLength))
59 {
60 JLOG(j.trace()) << "Malformed transaction: invalid size of CredentialType.";
61 return temMALFORMED;
62 }
63
64 return tesSUCCESS;
65}

88TER
89CredentialCreate::doApply()
90{
91 auto const subject = ctx_.tx[sfSubject];
92 auto const credType(ctx_.tx[sfCredentialType]);
93 Keylet const credentialKey = keylet::credential(subject, account_, credType);
94
95 auto const sleCred = std::make_shared<SLE>(credentialKey);
96 if (!sleCred)
97 return tefINTERNAL; // LCOV_EXCL_LINE
98
99 auto const optExp = ctx_.tx[~sfExpiration];
100 if (optExp)
101 {
102 std::uint32_t const closeTime = ctx_.view().header().parentCloseTime.time_since_epoch().count();
103
104 if (closeTime > *optExp)
105 {
106 JLOG(j_.trace()) << "Malformed transaction: "
107 "Expiration time is in the past.";
108 return tecEXPIRED;
109 }
110
111 sleCred->setFieldU32(sfExpiration, ctx_.tx.getFieldU32(sfExpiration));
112 }
113
114 auto const sleIssuer = view().peek(keylet::account(account_));
115 if (!sleIssuer)
116 return tefINTERNAL; // LCOV_EXCL_LINE
117
118 {
119 STAmount const reserve{view().fees().accountReserve(sleIssuer->getFieldU32(sfOwnerCount) + 1)};
120 if (mPriorBalance < reserve)
121 return tecINSUFFICIENT_RESERVE;
122 }
123
124 sleCred->setAccountID(sfSubject, subject);
125 sleCred->setAccountID(sfIssuer, account_);
126 sleCred->setFieldVL(sfCredentialType, credType);
127
128 if (ctx_.tx.isFieldPresent(sfURI))
129 sleCred->setFieldVL(sfURI, ctx_.tx.getFieldVL(sfURI));
130
131 {
132 auto const page = view().dirInsert(keylet::ownerDir(account_), credentialKey, describeOwnerDir(account_));
133 JLOG(j_.trace()) << "Adding Credential to owner directory " << to_string(credentialKey.key) << ": "
134 << (page ? "success" : "failure");
135 if (!page)
136 return tecDIR_FULL;
137 sleCred->setFieldU64(sfIssuerNode, *page);
138
139 adjustOwnerCount(view(), sleIssuer, 1, j_);
140 }
141
142 if (subject == account_)
143 {
144 sleCred->setFieldU32(sfFlags, lsfAccepted);
145 }
146 else
147 {
148 auto const page = view().dirInsert(keylet::ownerDir(subject), credentialKey, describeOwnerDir(subject));
149 JLOG(j_.trace()) << "Adding Credential to owner directory " << to_string(credentialKey.key) << ": "
150 << (page ? "success" : "failure");
151 if (!page)
152 return tecDIR_FULL;
153 sleCred->setFieldU64(sfSubjectNode, *page);
154 view().update(view().peek(keylet::account(subject)));
155 }
156
157 view().insert(sleCred);
158
159 return tesSUCCESS;
160}

172CredentialDelete::preflight(PreflightContext const& ctx)
173{
174 auto const subject = ctx.tx[~sfSubject];
175 auto const issuer = ctx.tx[~sfIssuer];
176
177 if (!subject && !issuer)
178 {
179 // Neither field is present, the transaction is malformed.
180 JLOG(ctx.j.trace()) << "Malformed transaction: "
181 "No Subject or Issuer fields.";
182 return temMALFORMED;
183 }
184
185 // Make sure that the passed account is valid.
186 if ((subject && subject->isZero()) || (issuer && issuer->isZero()))
187 {
188 JLOG(ctx.j.trace()) << "Malformed transaction: Subject or Issuer "
189 "field zeroed.";
190 return temINVALID_ACCOUNT_ID;
191 }
192
193 auto const credType = ctx.tx[sfCredentialType];
194 if (credType.empty() || (credType.size() > maxCredentialTypeLength))
195 {
196 JLOG(ctx.j.trace()) << "Malformed transaction: invalid size of CredentialType.";
197 return temMALFORMED;
198 }
199
200 return tesSUCCESS;

217TER
218CredentialDelete::doApply()
219{
220 auto const subject = ctx_.tx[~sfSubject].value_or(account_);
221 auto const issuer = ctx_.tx[~sfIssuer].value_or(account_);
222
223 auto const credType(ctx_.tx[sfCredentialType]);
224 auto const sleCred = view().peek(keylet::credential(subject, issuer, credType));
225 if (!sleCred)
226 return tefINTERNAL; // LCOV_EXCL_LINE
227
228 if ((subject != account_) && (issuer != account_) && !checkExpired(sleCred, ctx_.view().header().parentCloseTime))
229 {
230 JLOG(j_.trace()) << "Can't delete non-expired credential.";
231 return tecNO_PERMISSION;
232 }
233
234 return deleteSLE(view(), sleCred, j_);

265TER
266CredentialAccept::preclaim(PreclaimContext const& ctx)
267{
268 AccountID const subject = ctx.tx[sfAccount];
269 AccountID const issuer = ctx.tx[sfIssuer];
270 auto const credType(ctx.tx[sfCredentialType]);
271
272 if (!ctx.view.exists(keylet::account(issuer)))
273 {
274 JLOG(ctx.j.warn()) << "No issuer: " << to_string(issuer);
275 return tecNO_ISSUER;
276 }
277
278 auto const sleCred = ctx.view.read(keylet::credential(subject, issuer, credType));
279 if (!sleCred)
280 {
281 JLOG(ctx.j.warn()) << "No credential: " << to_string(subject) << ", " << to_string(issuer) << ", " << credType;
282 return tecNO_ENTRY;
283 }
284
285 if (sleCred->getFieldU32(sfFlags) & lsfAccepted)
286 {
287 JLOG(ctx.j.warn()) << "Credential already accepted: " << to_string(subject) << ", " << to_string(issuer) << ", "
288 << credType;
289 return tecDUPLICATE;
290 }
291
292 return tesSUCCESS;
293}

295TER
296CredentialAccept::doApply()
297{
298 AccountID const issuer{ctx_.tx[sfIssuer]};
299
300 // Both exist as credential object exist itself (checked in preclaim)
301 auto const sleSubject = view().peek(keylet::account(account_));
302 auto const sleIssuer = view().peek(keylet::account(issuer));
303
304 if (!sleSubject || !sleIssuer)
305 return tefINTERNAL; // LCOV_EXCL_LINE
306
307 {
308 STAmount const reserve{view().fees().accountReserve(sleSubject->getFieldU32(sfOwnerCount) + 1)};
309 if (mPriorBalance < reserve)
310 return tecINSUFFICIENT_RESERVE;
311 }
312
313 auto const credType(ctx_.tx[sfCredentialType]);
314 Keylet const credentialKey = keylet::credential(account_, issuer, credType);
315 auto const sleCred = view().peek(credentialKey); // Checked in preclaim()
316
317 if (checkExpired(sleCred, view().header().parentCloseTime))
318 {
319 JLOG(j_.trace()) << "Credential is expired: " << sleCred->getText();
320 // delete expired credentials even if the transaction failed
321 auto const err = credentials::deleteSLE(view(), sleCred, j_);
322 return isTesSuccess(err) ? tecEXPIRED : err;
323 }
324
325 sleCred->setFieldU32(sfFlags, lsfAccepted);
326 view().update(sleCred);
327
328 adjustOwnerCount(view(), sleIssuer, -1, j_);
329 adjustOwnerCount(view(), sleSubject, 1, j_);
330
331 return tesSUCCESS;
332}
