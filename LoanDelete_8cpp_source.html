62TER
63LoanDelete::doApply()
64{
65 auto const& tx = ctx_.tx;
66 auto& view = ctx_.view();
67
68 auto const loanID = tx[sfLoanID];
69 auto const loanSle = view.peek(keylet::loan(loanID));
70 if (!loanSle)
71 return tefBAD_LEDGER; // LCOV_EXCL_LINE
72 auto const borrower = loanSle->at(sfBorrower);
73 auto const borrowerSle = view.peek(keylet::account(borrower));
74 if (!borrowerSle)
75 return tefBAD_LEDGER; // LCOV_EXCL_LINE
76
77 auto const brokerID = loanSle->at(sfLoanBrokerID);
78 auto const brokerSle = view.peek(keylet::loanbroker(brokerID));
79 if (!brokerSle)
80 return tefBAD_LEDGER; // LCOV_EXCL_LINE
81 auto const brokerPseudoAccount = brokerSle->at(sfAccount);
82
83 auto const vaultSle = view.peek(keylet::vault(brokerSle->at(sfVaultID)));
84 if (!vaultSle)
85 return tefBAD_LEDGER; // LCOV_EXCL_LINE
86 auto const vaultAsset = vaultSle->at(sfAsset);
87
88 // Remove LoanID from Directory of the LoanBroker pseudo-account.
89 if (!view.dirRemove(
90 keylet::ownerDir(brokerPseudoAccount),
91 loanSle->at(sfLoanBrokerNode),
92 loanID,
93 false))
94 return tefBAD_LEDGER; // LCOV_EXCL_LINE
95 // Remove LoanID from Directory of the Borrower.
96 if (!view.dirRemove(
97 keylet::ownerDir(borrower),
98 loanSle->at(sfOwnerNode),
99 loanID,
100 false))
101 return tefBAD_LEDGER; // LCOV_EXCL_LINE
102
103 // Delete the Loan object
104 view.erase(loanSle);
105
106 // Decrement the LoanBroker's owner count.
107 // The broker's owner count is solely for the number of outstanding loans,
108 // and is distinct from the broker's pseudo-account's owner count
109 adjustOwnerCount(view, brokerSle, -1, j_);
110 // If there are no loans left, then any remaining debt must be forgiven,
111 // because there is no other way to pay it back.
112 if (brokerSle->at(sfOwnerCount) == 0)
113 {
114 auto debtTotalProxy = brokerSle->at(sfDebtTotal);
115 if (*debtTotalProxy != beast::zero)
116 {
117 XRPL_ASSERT_PARTS(
118 roundToAsset(
119 vaultSle->at(sfAsset),
120 debtTotalProxy,
121 getAssetsTotalScale(vaultSle),
122 Number::towards_zero) == beast::zero,
123 "xrpl::LoanDelete::doApply",
124 "last loan, remaining debt rounds to zero");
125 debtTotalProxy = 0;
126 }
127 }
128 // Decrement the borrower's owner count
129 adjustOwnerCount(view, borrowerSle, -1, j_);
130
131 // These associations shouldn't do anything, but do them just to be safe
132 associateAsset(*loanSle, vaultAsset);
133 associateAsset(*brokerSle, vaultAsset);
134 associateAsset(*vaultSle, vaultAsset);
135
136 return tesSUCCESS;
137}
