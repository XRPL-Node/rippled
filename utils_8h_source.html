69inline Preamble
70parsePreamble(Slice& s, std::error_code& ec)
71{
72 Preamble p;
73
74 if (s.size() < 2)
75 {
76 ec = error::short_preamble;
77 return p;
78 }
79
80 p.type = s[0] & 0xE0;
81 p.tag = s[0] & 0x1F;
82
83 s += 1;
84
85 if (p.tag == 0x1F)
86 { // Long tag form, which we do not support:
87 ec = error::long_tag;
88 return p;
89 }
90
91 p.length = s[0];
92 s += 1;
93
94 if (p.length & 0x80)
95 { // Long form length:
96 std::size_t const cnt = p.length & 0x7F;
97
98 if (cnt == 0)
99 {
100 ec = error::malformed_encoding;
101 return p;
102 }
103
104 if (cnt > sizeof(std::size_t))
105 {
106 ec = error::large_size;
107 return p;
108 }
109
110 if (cnt > s.size())
111 {
112 ec = error::short_preamble;
113 return p;
114 }
115
116 p.length = 0;
117
118 for (std::size_t i = 0; i != cnt; ++i)
119 p.length = (p.length << 8) + s[i];
120
121 s += cnt;
122
123 if (p.length == 0)
124 {
125 ec = error::malformed_encoding;
126 return p;
127 }
128 }
129
130 return p;
131}

153template <class Integer>
154Integer
155parseInteger(Slice& s, std::size_t count, std::error_code& ec)
156{
157 Integer v{0};
158
159 if (s.empty())
160 {
161 // can never have zero sized integers
162 ec = error::malformed_encoding;
163 return v;
164 }
165
166 if (count > s.size())
167 {
168 ec = error::buffer_underfull;
169 return v;
170 }
171
172 bool const isSigned = std::numeric_limits<Integer>::is_signed;
173 // unsigned types may have a leading zero octet
174 size_t const maxLength = isSigned ? sizeof(Integer) : sizeof(Integer) + 1;
175 if (count > maxLength)
176 {
177 ec = error::large_size;
178 return v;
179 }
180
181 if (!isSigned && (s[0] & (1 << 7)))
182 {
183 // trying to decode a negative number into a positive value
184 ec = error::malformed_encoding;
185 return v;
186 }
187
188 if (!isSigned && count == sizeof(Integer) + 1 && s[0])
189 {
190 // since integers are coded as two's complement, the first byte may
191 // be zero for unsigned reps
192 ec = error::malformed_encoding;
193 return v;
194 }
195
196 v = 0;
197 for (size_t i = 0; i < count; ++i)
198 v = (v << 8) | (s[i] & 0xff);
199
200 if (isSigned && (s[0] & (1 << 7)))
201 {
202 for (int i = count; i < sizeof(Integer); ++i)
203 v |= (Integer(0xff) << (8 * i));
204 }
205 s += count;
206 return v;
207}
