136 getInteger(Integer& number, int offset)
137 {
138 static auto const bytes = sizeof(Integer);
139 if ((offset + bytes) > mData.size())
140 return false;
141 number = 0;
142
143 auto ptr = &mData[offset];
144 for (auto i = 0; i < bytes; ++i)
145 {
146 if (i)
147 number <<= 8;
148 number |= *ptr++;
149 }
150 return true;
151 }

431template <std::size_t Bits, class Tag>
432base_uint<Bits, Tag>
433SerialIter::getBitString()
434{
435 auto const n = Bits / 8;
436
437 if (remain_ < n)

439
440 auto const x = p_;

442 p_ += n;
