96 void
97 sendThread()
98 {
99 Json::Value jvEvent;
100 bool bSend;
101
102 do
103 {
104 {
105 // Obtain the lock to manipulate the queue and change sending.
106 std::lock_guard sl(mLock);
107
108 if (mDeque.empty())
109 {
110 mSending = false;
111 bSend = false;
112 }
113 else
114 {
115 auto const [seq, env] = mDeque.front();
116
117 mDeque.pop_front();
118
119 jvEvent = env;
120 jvEvent["seq"] = seq;
121
122 bSend = true;
123 }
124 }
125

127 if (bSend)

139 mPassword,
140 mPath,

142 jvEvent,
143 mSSL,
144 true,
145 logs_);
146 }
147 catch (std::exception const& e)

149 JLOG(j_.info()) << "RPCCall::fromNetwork exception: " << e.what();
