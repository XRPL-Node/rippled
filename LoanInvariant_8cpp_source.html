14ValidLoanBroker::visitEntry(
15 bool isDelete,
16 std::shared_ptr<SLE const> const& before,
17 std::shared_ptr<SLE const> const& after)
18{
19 if (after)
20 {
21 if (after->getType() == ltLOAN_BROKER)
22 {
23 auto& broker = brokers_[after->key()];
24 broker.brokerBefore = before;
25 broker.brokerAfter = after;
26 }
27 else if (after->getType() == ltACCOUNT_ROOT && after->isFieldPresent(sfLoanBrokerID))
28 {
29 auto const& loanBrokerID = after->at(sfLoanBrokerID);
30 // create an entry if one doesn't already exist
31 brokers_.emplace(loanBrokerID, BrokerInfo{});
32 }
33 else if (after->getType() == ltRIPPLE_STATE)
34 {
35 lines_.emplace_back(after);
36 }
37 else if (after->getType() == ltMPTOKEN)
38 {
39 mpts_.emplace_back(after);
40 }
41 }

44bool
45ValidLoanBroker::goodZeroDirectory(
46 ReadView const& view,
47 SLE::const_ref dir,
48 beast::Journal const& j) const
49{
50 auto const next = dir->at(~sfIndexNext);
51 auto const prev = dir->at(~sfIndexPrevious);
52 if ((prev && *prev) || (next && *next))
53 {
54 JLOG(j.fatal()) << "Invariant failed: Loan Broker with zero "
55 "OwnerCount has multiple directory pages";
56 return false;
57 }
58 auto indexes = dir->getFieldV256(sfIndexes);
59 if (indexes.size() > 1)
60 {
61 JLOG(j.fatal()) << "Invariant failed: Loan Broker with zero "
62 "OwnerCount has multiple indexes in the Directory root";
63 return false;
64 }
65 if (indexes.size() == 1)
66 {
67 auto const index = indexes.value().front();
68 auto const sle = view.read(keylet::unchecked(index));
69 if (!sle)
70 {
71 JLOG(j.fatal()) << "Invariant failed: Loan Broker directory corrupt";
72 return false;
73 }
74 if (sle->getType() != ltRIPPLE_STATE && sle->getType() != ltMPTOKEN)
75 {
76 JLOG(j.fatal()) << "Invariant failed: Loan Broker with zero "
77 "OwnerCount has an unexpected entry in the directory";
78 return false;
79 }
80 }
81
82 return true;
83}

85bool
86ValidLoanBroker::finalize(
87 STTx const& tx,
88 TER const,
89 XRPAmount const,
90 ReadView const& view,
91 beast::Journal const& j)
92{
93 // Loan Brokers will not exist on ledger if the Lending Protocol amendment
94 // is not enabled, so there's no need to check it.
95
96 for (auto const& line : lines_)
97 {
98 for (auto const& field : {&sfLowLimit, &sfHighLimit})
99 {
100 auto const account = view.read(keylet::account(line->at(*field).getIssuer()));
101 // This Invariant doesn't know about the rules for Trust Lines, so
102 // if the account is missing, don't treat it as an error. This
103 // loop is only concerned with finding Broker pseudo-accounts
104 if (account && account->isFieldPresent(sfLoanBrokerID))
105 {
106 auto const& loanBrokerID = account->at(sfLoanBrokerID);
107 // create an entry if one doesn't already exist
108 brokers_.emplace(loanBrokerID, BrokerInfo{});
109 }
110 }
111 }
112 for (auto const& mpt : mpts_)
113 {
114 auto const account = view.read(keylet::account(mpt->at(sfAccount)));
115 // This Invariant doesn't know about the rules for MPTokens, so
116 // if the account is missing, don't treat is as an error. This
117 // loop is only concerned with finding Broker pseudo-accounts
118 if (account && account->isFieldPresent(sfLoanBrokerID))
119 {
120 auto const& loanBrokerID = account->at(sfLoanBrokerID);
121 // create an entry if one doesn't already exist
122 brokers_.emplace(loanBrokerID, BrokerInfo{});
123 }
124 }
125
126 for (auto const& [brokerID, broker] : brokers_)
127 {
128 auto const& after =
129 broker.brokerAfter ? broker.brokerAfter : view.read(keylet::loanbroker(brokerID));
130
131 if (!after)
132 {
133 JLOG(j.fatal()) << "Invariant failed: Loan Broker missing";
134 return false;
135 }
136
137 auto const& before = broker.brokerBefore;
138
139 // https://github.com/Tapanito/XRPL-Standards/blob/xls-66-lending-protocol/XLS-0066d-lending-protocol/README.md#3123-invariants
140 // If `LoanBroker.OwnerCount = 0` the `DirectoryNode` will have at most
141 // one node (the root), which will only hold entries for `RippleState`
142 // or `MPToken` objects.
143 if (after->at(sfOwnerCount) == 0)
144 {
145 auto const dir = view.read(keylet::ownerDir(after->at(sfAccount)));
146 if (dir)
147 {
148 if (!goodZeroDirectory(view, dir, j))
149 {
150 return false;
151 }
152 }
153 }
154 if (before && before->at(sfLoanSequence) > after->at(sfLoanSequence))
155 {
156 JLOG(j.fatal()) << "Invariant failed: Loan Broker sequence number "
157 "decreased";
158 return false;
159 }
160 if (after->at(sfDebtTotal) < 0)
161 {
162 JLOG(j.fatal()) << "Invariant failed: Loan Broker debt total is negative";
163 return false;
164 }
165 if (after->at(sfCoverAvailable) < 0)
166 {
167 JLOG(j.fatal()) << "Invariant failed: Loan Broker cover available is negative";
168 return false;
169 }
170 auto const vault = view.read(keylet::vault(after->at(sfVaultID)));
171 if (!vault)
172 {
173 JLOG(j.fatal()) << "Invariant failed: Loan Broker vault ID is invalid";
174 return false;
175 }
176 auto const& vaultAsset = vault->at(sfAsset);
177 if (after->at(sfCoverAvailable) < accountHolds(
178 view,
179 after->at(sfAccount),
180 vaultAsset,
181 FreezeHandling::fhIGNORE_FREEZE,
182 AuthHandling::ahIGNORE_AUTH,
183 j))
184 {
185 JLOG(j.fatal()) << "Invariant failed: Loan Broker cover available "
186 "is less than pseudo-account asset balance";
187 return false;
188 }
189 }
190 return true;
191}

207bool
208ValidLoan::finalize(
209 STTx const& tx,
210 TER const,
211 XRPAmount const,
212 ReadView const& view,
213 beast::Journal const& j)
214{
215 // Loans will not exist on ledger if the Lending Protocol amendment
216 // is not enabled, so there's no need to check it.
217
218 for (auto const& [before, after] : loans_)
219 {
220 // https://github.com/Tapanito/XRPL-Standards/blob/xls-66-lending-protocol/XLS-0066d-lending-protocol/README.md#3223-invariants
221 // If `Loan.PaymentRemaining = 0` then the loan MUST be fully paid off
222 if (after->at(sfPaymentRemaining) == 0 &&
223 (after->at(sfTotalValueOutstanding) != beast::zero ||
224 after->at(sfPrincipalOutstanding) != beast::zero ||
225 after->at(sfManagementFeeOutstanding) != beast::zero))
226 {
227 JLOG(j.fatal()) << "Invariant failed: Loan with zero payments "
228 "remaining has not been paid off";
229 return false;
230 }
231 // If `Loan.PaymentRemaining != 0` then the loan MUST NOT be fully paid
232 // off
233 if (after->at(sfPaymentRemaining) != 0 &&
234 after->at(sfTotalValueOutstanding) == beast::zero &&
235 after->at(sfPrincipalOutstanding) == beast::zero &&
236 after->at(sfManagementFeeOutstanding) == beast::zero)
237 {
238 JLOG(j.fatal()) << "Invariant failed: Loan with zero payments "
239 "remaining has not been paid off";
240 return false;
241 }
242 if (before && (before->isFlag(lsfLoanOverpayment) != after->isFlag(lsfLoanOverpayment)))
243 {
244 JLOG(j.fatal()) << "Invariant failed: Loan Overpayment flag changed";
245 return false;
246 }
247 // Must not be negative - STNumber
248 for (auto const field :
249 {&sfLoanServiceFee,
250 &sfLatePaymentFee,
251 &sfClosePaymentFee,
252 &sfPrincipalOutstanding,
253 &sfTotalValueOutstanding,
254 &sfManagementFeeOutstanding})
255 {
256 if (after->at(*field) < 0)
257 {
258 JLOG(j.fatal()) << "Invariant failed: " << field->getName() << " is negative ";
259 return false;
260 }
261 }
262 // Must be positive - STNumber
263 for (auto const field : {
264 &sfPeriodicPayment,
265 })
266 {
267 if (after->at(*field) <= 0)
268 {
269 JLOG(j.fatal()) << "Invariant failed: " << field->getName()
270 << " is zero or negative ";
271 return false;
272 }
273 }
274 }
275 return true;
276}
