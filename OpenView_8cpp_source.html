191auto
192OpenView::txRead(key_type const& key) const -> tx_type
193{
194 auto const iter = txs_.find(key);
195 if (iter == txs_.end())
196 return base_->txRead(key);
197 auto const& item = iter->second;
198 auto stx = std::make_shared<STTx const>(SerialIter{item.txn->slice()});
199 decltype(tx_type::second) sto;
200 if (item.meta)
201 sto = std::make_shared<STObject const>(SerialIter{item.meta->slice()}, sfMetadata);
202 else
203 sto = nullptr;
204 return {std::move(stx), std::move(sto)};
205}
