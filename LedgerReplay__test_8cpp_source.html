488 {
489 int fundedAccounts = accounts.size();
490 assert(fundedAccounts >= newTxes);
491 std::unordered_set<int> senders;
492
493 // somewhat random but reproducible
494 int r = ledgerMaster.getClosedLedger()->seq() * 7;
495 int fromIdx = 0;
496 int toIdx = 0;
497 auto updateIdx = [&]() {
498 assert(fundedAccounts > senders.size());
499 fromIdx = (fromIdx + r) % fundedAccounts;
500 while (senders.count(fromIdx) != 0)
501 fromIdx = (fromIdx + 1) % fundedAccounts;
502 senders.insert(fromIdx);
503 toIdx = (toIdx + r * 2) % fundedAccounts;
504 if (toIdx == fromIdx)
505 toIdx = (toIdx + 1) % fundedAccounts;
506 };
507
508 for (int i = 0; i < newTxes; ++i)
509 {
510 updateIdx();
511 env(pay(accounts[fromIdx],
512 accounts[toIdx],
513 jtx::drops(ledgerMaster.getClosedLedger()->fees().base) +
514 jtx::XRP(param.txAmount)),
515 jtx::seq(jtx::autofill),
516 jtx::fee(jtx::autofill),
517 jtx::sig(jtx::autofill));
518 }
519 env.close();
520 }
