47NotTEC
48CreateOffer::preflight(PreflightContext const& ctx)
49{
50 auto& tx = ctx.tx;
51 auto& j = ctx.j;
52
53 std::uint32_t const uTxFlags = tx.getFlags();
54
55 if (tx.isFlag(tfHybrid) && !tx.isFieldPresent(sfDomainID))
56 return temINVALID_FLAG;
57
58 bool const bImmediateOrCancel(uTxFlags & tfImmediateOrCancel);
59 bool const bFillOrKill(uTxFlags & tfFillOrKill);
60
61 if (bImmediateOrCancel && bFillOrKill)
62 {
63 JLOG(j.debug()) << "Malformed transaction: both IoC and FoK set.";
64 return temINVALID_FLAG;
65 }
66
67 bool const bHaveExpiration(tx.isFieldPresent(sfExpiration));
68
69 if (bHaveExpiration && (tx.getFieldU32(sfExpiration) == 0))
70 {
71 JLOG(j.debug()) << "Malformed offer: bad expiration";
72 return temBAD_EXPIRATION;
73 }
74
75 if (auto const cancelSequence = tx[~sfOfferSequence]; cancelSequence && *cancelSequence == 0)
76 {
77 JLOG(j.debug()) << "Malformed offer: bad cancel sequence";
78 return temBAD_SEQUENCE;
79 }
80
81 STAmount saTakerPays = tx[sfTakerPays];
82 STAmount saTakerGets = tx[sfTakerGets];
83
84 if (!isLegalNet(saTakerPays) || !isLegalNet(saTakerGets))
85 return temBAD_AMOUNT;
86
87 if (saTakerPays.native() && saTakerGets.native())
88 {
89 JLOG(j.debug()) << "Malformed offer: redundant (XRP for XRP)";
90 return temBAD_OFFER;
91 }
92 if (saTakerPays <= beast::zero || saTakerGets <= beast::zero)
93 {
94 JLOG(j.debug()) << "Malformed offer: bad amount";
95 return temBAD_OFFER;
96 }
97
98 auto const& uPaysIssuerID = saTakerPays.getIssuer();
99 auto const& uPaysCurrency = saTakerPays.getCurrency();
100
101 auto const& uGetsIssuerID = saTakerGets.getIssuer();
102 auto const& uGetsCurrency = saTakerGets.getCurrency();
103
104 if (uPaysCurrency == uGetsCurrency && uPaysIssuerID == uGetsIssuerID)
105 {
106 JLOG(j.debug()) << "Malformed offer: redundant (IOU for IOU)";
107 return temREDUNDANT;
108 }
109 // We don't allow a non-native currency to use the currency code XRP.
110 if (badCurrency() == uPaysCurrency || badCurrency() == uGetsCurrency)
111 {
112 JLOG(j.debug()) << "Malformed offer: bad currency";
113 return temBAD_CURRENCY;
114 }
115
116 if (saTakerPays.native() != !uPaysIssuerID || saTakerGets.native() != !uGetsIssuerID)
117 {
118 JLOG(j.debug()) << "Malformed offer: bad issuer";
119 return temBAD_ISSUER;
120 }
121
122 return tesSUCCESS;
123}

125TER
126CreateOffer::preclaim(PreclaimContext const& ctx)
127{
128 auto const id = ctx.tx[sfAccount];
129
130 auto saTakerPays = ctx.tx[sfTakerPays];
131 auto saTakerGets = ctx.tx[sfTakerGets];
132
133 auto const& uPaysIssuerID = saTakerPays.getIssuer();
134 auto const& uPaysCurrency = saTakerPays.getCurrency();
135
136 auto const& uGetsIssuerID = saTakerGets.getIssuer();
137
138 auto const cancelSequence = ctx.tx[~sfOfferSequence];
139
140 auto const sleCreator = ctx.view.read(keylet::account(id));
141 if (!sleCreator)
142 return terNO_ACCOUNT;
143
144 std::uint32_t const uAccountSequence = sleCreator->getFieldU32(sfSequence);
145
146 auto viewJ = ctx.registry.journal("View");
147
148 if (isGlobalFrozen(ctx.view, uPaysIssuerID) || isGlobalFrozen(ctx.view, uGetsIssuerID))
149 {
150 JLOG(ctx.j.debug()) << "Offer involves frozen asset";
151 return tecFROZEN;
152 }
153
154 if (accountFunds(ctx.view, id, saTakerGets, fhZERO_IF_FROZEN, viewJ) <= beast::zero)
155 {
156 JLOG(ctx.j.debug()) << "delay: Offers must be at least partially funded.";
157 return tecUNFUNDED_OFFER;
158 }
159
160 // This can probably be simplified to make sure that you cancel sequences
161 // before the transaction sequence number.
162 if (cancelSequence && (uAccountSequence <= *cancelSequence))
163 {
164 JLOG(ctx.j.debug()) << "uAccountSequenceNext=" << uAccountSequence << " uOfferSequence=" << *cancelSequence;
165 return temBAD_SEQUENCE;
166 }
167
168 if (hasExpired(ctx.view, ctx.tx[~sfExpiration]))
169 {
170 // Note that this will get checked again in applyGuts, but it saves
171 // us a call to checkAcceptAsset and possible false negative.
172 return tecEXPIRED;
173 }
174
175 // Make sure that we are authorized to hold what the taker will pay us.
176 if (!saTakerPays.native())
177 {
178 auto result = checkAcceptAsset(ctx.view, ctx.flags, id, ctx.j, Issue(uPaysCurrency, uPaysIssuerID));
179 if (result != tesSUCCESS)
180 return result;
181 }
182
183 // if domain is specified, make sure that domain exists and the offer create
184 // is part of the domain
185 if (ctx.tx.isFieldPresent(sfDomainID))
186 {
187 if (!permissioned_dex::accountInDomain(ctx.view, id, ctx.tx[sfDomainID]))
188 return tecNO_PERMISSION;
189 }
190
191 return tesSUCCESS;
192}

194TER
195CreateOffer::checkAcceptAsset(
196 ReadView const& view,
197 ApplyFlags const flags,
198 AccountID const id,
199 beast::Journal const j,
200 Issue const& issue)
201{
202 // Only valid for custom currencies
203 XRPL_ASSERT(!isXRP(issue.currency), "xrpl::CreateOffer::checkAcceptAsset : input is not XRP");
204
205 auto const issuerAccount = view.read(keylet::account(issue.account));
206
207 if (!issuerAccount)
208 {
209 JLOG(j.debug()) << "delay: can't receive IOUs from non-existent issuer: " << to_string(issue.account);
210
211 return (flags & tapRETRY) ? TER{terNO_ACCOUNT} : TER{tecNO_ISSUER};
212 }
213
214 if (issue.account == id)
215 // An account can always accept its own issuance.
216 return tesSUCCESS;
217
218 if ((*issuerAccount)[sfFlags] & lsfRequireAuth)
219 {
220 auto const trustLine = view.read(keylet::line(id, issue.account, issue.currency));
221
222 if (!trustLine)
223 {
224 return (flags & tapRETRY) ? TER{terNO_LINE} : TER{tecNO_LINE};
225 }
226
227 // Entries have a canonical representation, determined by a
228 // lexicographical "greater than" comparison employing strict weak
229 // ordering. Determine which entry we need to access.
230 bool const canonical_gt(id > issue.account);
231
232 bool const is_authorized((*trustLine)[sfFlags] & (canonical_gt ? lsfLowAuth : lsfHighAuth));
233
234 if (!is_authorized)
235 {
236 JLOG(j.debug()) << "delay: can't receive IOUs from issuer without auth.";
237
238 return (flags & tapRETRY) ? TER{terNO_AUTH} : TER{tecNO_AUTH};
239 }
240 }
241
242 // An account can not create a trustline to itself, so no line can exist
243 // to be frozen. Additionally, an issuer can always accept its own
244 // issuance.
245 if (issue.account == id)
246 {
247 return tesSUCCESS;
248 }
249
250 auto const trustLine = view.read(keylet::line(id, issue.account, issue.currency));
251
252 if (!trustLine)
253 {
254 return tesSUCCESS;
255 }
256
257 // There's no difference which side enacted deep freeze, accepting
258 // tokens shouldn't be possible.
259 bool const deepFrozen = (*trustLine)[sfFlags] & (lsfLowDeepFreeze | lsfHighDeepFreeze);
260
261 if (deepFrozen)
262 {
263 return tecFROZEN;
264 }
265
266 return tesSUCCESS;
267}

269std::pair<TER, Amounts>
270CreateOffer::flowCross(
271 PaymentSandbox& psb,
272 PaymentSandbox& psbCancel,
273 Amounts const& takerAmount,
274 std::optional<uint256> const& domainID)
275{
276 try
277 {
278 // If the taker is unfunded before we begin crossing there's nothing
279 // to do - just return an error.
280 //
281 // We check this in preclaim, but when selling XRP charged fees can
282 // cause a user's available balance to go to 0 (by causing it to dip
283 // below the reserve) so we check this case again.
284 STAmount const inStartBalance = accountFunds(psb, account_, takerAmount.in, fhZERO_IF_FROZEN, j_);
285 if (inStartBalance <= beast::zero)
286 {
287 // The account balance can't cover even part of the offer.
288 JLOG(j_.debug()) << "Not crossing: taker is unfunded.";
289 return {tecUNFUNDED_OFFER, takerAmount};
290 }
291
292 // If the gateway has a transfer rate, accommodate that. The
293 // gateway takes its cut without any special consent from the
294 // offer taker. Set sendMax to allow for the gateway's cut.
295 Rate gatewayXferRate{QUALITY_ONE};
296 STAmount sendMax = takerAmount.in;
297 if (!sendMax.native() && (account_ != sendMax.getIssuer()))
298 {
299 gatewayXferRate = transferRate(psb, sendMax.getIssuer());
300 if (gatewayXferRate.value != QUALITY_ONE)
301 {
302 sendMax = multiplyRound(takerAmount.in, gatewayXferRate, takerAmount.in.issue(), true);
303 }
304 }
305
306 // Payment flow code compares quality after the transfer rate is
307 // included. Since transfer rate is incorporated compute threshold.
308 Quality threshold{takerAmount.out, sendMax};
309
310 // If we're creating a passive offer adjust the threshold so we only
311 // cross offers that have a better quality than this one.
312 std::uint32_t const txFlags = ctx_.tx.getFlags();
313 if (txFlags & tfPassive)
314 ++threshold;
315
316 // Don't send more than our balance.
317 if (sendMax > inStartBalance)
318 sendMax = inStartBalance;
319
320 // Always invoke flow() with the default path. However if neither
321 // of the takerAmount currencies are XRP then we cross through an
322 // additional path with XRP as the intermediate between two books.
323 // This second path we have to build ourselves.
324 STPathSet paths;
325 if (!takerAmount.in.native() && !takerAmount.out.native())
326 {
327 STPath path;
328 path.emplace_back(std::nullopt, xrpCurrency(), std::nullopt);
329 paths.emplace_back(std::move(path));
330 }
331 // Special handling for the tfSell flag.
332 STAmount deliver = takerAmount.out;
333 OfferCrossing offerCrossing = OfferCrossing::yes;
334 if (txFlags & tfSell)
335 {
336 offerCrossing = OfferCrossing::sell;
337 // We are selling, so we will accept *more* than the offer
338 // specified. Since we don't know how much they might offer,
339 // we allow delivery of the largest possible amount.
340 if (deliver.native())
341 deliver = STAmount{STAmount::cMaxNative};
342 else
343 // We can't use the maximum possible currency here because
344 // there might be a gateway transfer rate to account for.
345 // Since the transfer rate cannot exceed 200%, we use 1/2
346 // maxValue for our limit.
347 deliver = STAmount{takerAmount.out.issue(), STAmount::cMaxValue / 2, STAmount::cMaxOffset};
348 }
349
350 // Call the payment engine's flow() to do the actual work.
351 auto const result = flow(
352 psb,
353 deliver,
354 account_,
355 account_,
356 paths,
357 true, // default path
358 !(txFlags & tfFillOrKill), // partial payment
359 true, // owner pays transfer fee
360 offerCrossing,
361 threshold,
362 sendMax,
363 domainID,
364 j_);
365
366 // If stale offers were found remove them.
367 for (auto const& toRemove : result.removableOffers)
368 {
369 if (auto otr = psb.peek(keylet::offer(toRemove)))
370 offerDelete(psb, otr, j_);
371 if (auto otr = psbCancel.peek(keylet::offer(toRemove)))
372 offerDelete(psbCancel, otr, j_);
373 }
374
375 // Determine the size of the final offer after crossing.
376 auto afterCross = takerAmount; // If !tesSUCCESS offer unchanged
377 if (isTesSuccess(result.result()))
378 {
379 STAmount const takerInBalance = accountFunds(psb, account_, takerAmount.in, fhZERO_IF_FROZEN, j_);
380
381 if (takerInBalance <= beast::zero)
382 {
383 // If offer crossing exhausted the account's funds don't
384 // create the offer.
385 afterCross.in.clear();
386 afterCross.out.clear();
387 }
388 else
389 {
390 STAmount const rate{Quality{takerAmount.out, takerAmount.in}.rate()};
391
392 if (txFlags & tfSell)
393 {
394 // If selling then scale the new out amount based on how
395 // much we sold during crossing. This preserves the offer
396 // Quality,
397
398 // Reduce the offer that is placed by the crossed amount.
399 // Note that we must ignore the portion of the
400 // actualAmountIn that may have been consumed by a
401 // gateway's transfer rate.
402 STAmount nonGatewayAmountIn = result.actualAmountIn;
403 if (gatewayXferRate.value != QUALITY_ONE)
404 nonGatewayAmountIn =
405 divideRound(result.actualAmountIn, gatewayXferRate, takerAmount.in.issue(), true);
406
407 afterCross.in -= nonGatewayAmountIn;
408
409 // It's possible that the divRound will cause our subtract
410 // to go slightly negative. So limit afterCross.in to zero.
411 if (afterCross.in < beast::zero)
412 // We should verify that the difference *is* small, but
413 // what is a good threshold to check?
414 afterCross.in.clear();
415
416 afterCross.out = divRoundStrict(afterCross.in, rate, takerAmount.out.issue(), false);
417 }
418 else
419 {
420 // If not selling, we scale the input based on the
421 // remaining output. This too preserves the offer
422 // Quality.
423 afterCross.out -= result.actualAmountOut;
424 XRPL_ASSERT(afterCross.out >= beast::zero, "xrpl::CreateOffer::flowCross : minimum offer");
425 if (afterCross.out < beast::zero)
426 afterCross.out.clear();
427 afterCross.in = mulRound(afterCross.out, rate, takerAmount.in.issue(), true);
428 }
429 }
430 }
431
432 // Return how much of the offer is left.
433 return {tesSUCCESS, afterCross};
434 }
435 catch (std::exception const& e)
436 {
437 JLOG(j_.error()) << "Exception during offer crossing: " << e.what();
438 }
439 return {tecINTERNAL, takerAmount};
440}

451TER
452CreateOffer::applyHybrid(
453 Sandbox& sb,
454 std::shared_ptr<STLedgerEntry> sleOffer,
455 Keylet const& offerKey,
456 STAmount const& saTakerPays,
457 STAmount const& saTakerGets,
458 std::function<void(SLE::ref, std::optional<uint256>)> const& setDir)
459{
460 if (!sleOffer->isFieldPresent(sfDomainID))
461 return tecINTERNAL; // LCOV_EXCL_LINE
462
463 // set hybrid flag
464 sleOffer->setFlag(lsfHybrid);
465
466 // if offer is hybrid, need to also place into open offer dir
467 Book const book{saTakerPays.issue(), saTakerGets.issue(), std::nullopt};
468
469 auto dir = keylet::quality(keylet::book(book), getRate(saTakerGets, saTakerPays));
470 bool const bookExists = sb.exists(dir);
471
472 auto const bookNode = sb.dirAppend(dir, offerKey, [&](SLE::ref sle) {
473 // don't set domainID on the directory object since this directory is
474 // for open book
475 setDir(sle, std::nullopt);
476 });
477
478 if (!bookNode)
479 {
480 JLOG(j_.debug()) << "final result: failed to add hybrid offer to open book";
481 return tecDIR_FULL; // LCOV_EXCL_LINE
482 }
483
484 STArray bookArr(sfAdditionalBooks, 1);
485 auto bookInfo = STObject::makeInnerObject(sfBook);
486 bookInfo.setFieldH256(sfBookDirectory, dir.key);
487 bookInfo.setFieldU64(sfBookNode, *bookNode);
488 bookArr.push_back(std::move(bookInfo));
489
490 if (!bookExists)
491 ctx_.registry.getOrderBookDB().addOrderBook(book);
492
493 sleOffer->setFieldArray(sfAdditionalBooks, bookArr);
494 return tesSUCCESS;
495}

497std::pair<TER, bool>
498CreateOffer::applyGuts(Sandbox& sb, Sandbox& sbCancel)
499{
500 using beast::zero;
501
502 std::uint32_t const uTxFlags = ctx_.tx.getFlags();
503
504 bool const bPassive(uTxFlags & tfPassive);
505 bool const bImmediateOrCancel(uTxFlags & tfImmediateOrCancel);
506 bool const bFillOrKill(uTxFlags & tfFillOrKill);
507 bool const bSell(uTxFlags & tfSell);
508 bool const bHybrid(uTxFlags & tfHybrid);
509
510 auto saTakerPays = ctx_.tx[sfTakerPays];
511 auto saTakerGets = ctx_.tx[sfTakerGets];
512 auto const domainID = ctx_.tx[~sfDomainID];
513
514 auto const cancelSequence = ctx_.tx[~sfOfferSequence];
515
516 // Note that we we use the value from the sequence or ticket as the
517 // offer sequence. For more explanation see comments in SeqProxy.h.
518 auto const offerSequence = ctx_.tx.getSeqValue();
519
520 // This is the original rate of the offer, and is the rate at which
521 // it will be placed, even if crossing offers change the amounts that
522 // end up on the books.
523 auto uRate = getRate(saTakerGets, saTakerPays);
524
525 auto viewJ = ctx_.registry.journal("View");
526
527 TER result = tesSUCCESS;
528
529 // Process a cancellation request that's passed along with an offer.
530 if (cancelSequence)
531 {
532 auto const sleCancel = sb.peek(keylet::offer(account_, *cancelSequence));
533
534 // It's not an error to not find the offer to cancel: it might have
535 // been consumed or removed. If it is found, however, it's an error
536 // to fail to delete it.
537 if (sleCancel)
538 {
539 JLOG(j_.debug()) << "Create cancels order " << *cancelSequence;
540 result = offerDelete(sb, sleCancel, viewJ);
541 }
542 }
543
544 auto const expiration = ctx_.tx[~sfExpiration];
545
546 if (hasExpired(sb, expiration))
547 {
548 // If the offer has expired, the transaction has successfully
549 // done nothing, so short circuit from here.
550 return {tecEXPIRED, true};
551 }
552
553 bool const bOpenLedger = sb.open();
554 bool crossed = false;
555
556 if (result == tesSUCCESS)
557 {
558 // If a tick size applies, round the offer to the tick size
559 auto const& uPaysIssuerID = saTakerPays.getIssuer();
560 auto const& uGetsIssuerID = saTakerGets.getIssuer();
561
562 std::uint8_t uTickSize = Quality::maxTickSize;
563 if (!isXRP(uPaysIssuerID))
564 {
565 auto const sle = sb.read(keylet::account(uPaysIssuerID));
566 if (sle && sle->isFieldPresent(sfTickSize))
567 uTickSize = std::min(uTickSize, (*sle)[sfTickSize]);
568 }
569 if (!isXRP(uGetsIssuerID))
570 {
571 auto const sle = sb.read(keylet::account(uGetsIssuerID));
572 if (sle && sle->isFieldPresent(sfTickSize))
573 uTickSize = std::min(uTickSize, (*sle)[sfTickSize]);
574 }
575 if (uTickSize < Quality::maxTickSize)
576 {
577 auto const rate = Quality{saTakerGets, saTakerPays}.round(uTickSize).rate();
578
579 // We round the side that's not exact,
580 // just as if the offer happened to execute
581 // at a slightly better (for the placer) rate
582 if (bSell)
583 {
584 // this is a sell, round taker pays
585 saTakerPays = multiply(saTakerGets, rate, saTakerPays.issue());
586 }
587 else
588 {
589 // this is a buy, round taker gets
590 saTakerGets = divide(saTakerPays, rate, saTakerGets.issue());
591 }
592 if (!saTakerGets || !saTakerPays)
593 {
594 JLOG(j_.debug()) << "Offer rounded to zero";
595 return {result, true};
596 }
597
598 uRate = getRate(saTakerGets, saTakerPays);
599 }
600
601 // We reverse pays and gets because during crossing we are taking.
602 Amounts const takerAmount(saTakerGets, saTakerPays);
603
604 JLOG(j_.debug()) << "Attempting cross: " << to_string(takerAmount.in.issue()) << " -> "
605 << to_string(takerAmount.out.issue());
606
607 if (auto stream = j_.trace())
608 {
609 stream << " mode: " << (bPassive ? "passive " : "") << (bSell ? "sell" : "buy");
610 stream << " in: " << format_amount(takerAmount.in);
611 stream << " out: " << format_amount(takerAmount.out);
612 }
613
614 // The amount of the offer that is unfilled after crossing has been
615 // performed. It may be equal to the original amount (didn't cross),
616 // empty (fully crossed), or something in-between.
617 Amounts place_offer;
618 PaymentSandbox psbFlow{&sb};
619 PaymentSandbox psbCancelFlow{&sbCancel};
620
621 std::tie(result, place_offer) = flowCross(psbFlow, psbCancelFlow, takerAmount, domainID);
622 psbFlow.apply(sb);
623 psbCancelFlow.apply(sbCancel);
624
625 // We expect the implementation of cross to succeed
626 // or give a tec.
627 XRPL_ASSERT(
628 result == tesSUCCESS || isTecClaim(result),
629 "xrpl::CreateOffer::applyGuts : result is tesSUCCESS or "
630 "tecCLAIM");
631
632 if (auto stream = j_.trace())
633 {
634 stream << "Cross result: " << transToken(result);
635 stream << " in: " << format_amount(place_offer.in);
636 stream << " out: " << format_amount(place_offer.out);
637 }
638
639 if (result == tecFAILED_PROCESSING && bOpenLedger)
640 result = telFAILED_PROCESSING;
641
642 if (result != tesSUCCESS)
643 {
644 JLOG(j_.debug()) << "final result: " << transToken(result);
645 return {result, true};
646 }
647
648 XRPL_ASSERT(
649 saTakerGets.issue() == place_offer.in.issue(), "xrpl::CreateOffer::applyGuts : taker gets issue match");
650 XRPL_ASSERT(
651 saTakerPays.issue() == place_offer.out.issue(), "xrpl::CreateOffer::applyGuts : taker pays issue match");
652
653 if (takerAmount != place_offer)
654 crossed = true;
655
656 // The offer that we need to place after offer crossing should
657 // never be negative. If it is, something went very very wrong.
658 if (place_offer.in < zero || place_offer.out < zero)
659 {
660 JLOG(j_.fatal()) << "Cross left offer negative!"
661 << " in: " << format_amount(place_offer.in)
662 << " out: " << format_amount(place_offer.out);
663 return {tefINTERNAL, true};
664 }
665
666 if (place_offer.in == zero || place_offer.out == zero)
667 {
668 JLOG(j_.debug()) << "Offer fully crossed!";
669 return {result, true};
670 }
671
672 // We now need to adjust the offer to reflect the amount left after
673 // crossing. We reverse in and out here, since during crossing we
674 // were the taker.
675 saTakerPays = place_offer.out;
676 saTakerGets = place_offer.in;
677 }
678
679 XRPL_ASSERT(
680 saTakerPays > zero && saTakerGets > zero, "xrpl::CreateOffer::applyGuts : taker pays and gets positive");
681
682 if (result != tesSUCCESS)
683 {
684 JLOG(j_.debug()) << "final result: " << transToken(result);
685 return {result, true};
686 }
687
688 if (auto stream = j_.trace())
689 {
690 stream << "Place" << (crossed ? " remaining " : " ") << "offer:";
691 stream << " Pays: " << saTakerPays.getFullText();
692 stream << " Gets: " << saTakerGets.getFullText();
693 }
694
695 // For 'fill or kill' offers, failure to fully cross means that the
696 // entire operation should be aborted, with only fees paid.
697 if (bFillOrKill)
698 {
699 JLOG(j_.trace()) << "Fill or Kill: offer killed";
700 return {tecKILLED, false};
701 }
702
703 // For 'immediate or cancel' offers, the amount remaining doesn't get
704 // placed - it gets canceled and the operation succeeds.
705 if (bImmediateOrCancel)
706 {
707 JLOG(j_.trace()) << "Immediate or cancel: offer canceled";
708 if (!crossed)
709 // Any ImmediateOrCancel offer that transfers absolutely no funds
710 // returns tecKILLED rather than tesSUCCESS. Motivation for the
711 // change is here: https://github.com/ripple/rippled/issues/4115
712 return {tecKILLED, false};
713 return {tesSUCCESS, true};
714 }
715
716 auto const sleCreator = sb.peek(keylet::account(account_));
717 if (!sleCreator)
718 return {tefINTERNAL, false};
719
720 {
721 XRPAmount reserve = sb.fees().accountReserve(sleCreator->getFieldU32(sfOwnerCount) + 1);
722
723 if (mPriorBalance < reserve)
724 {
725 // If we are here, the signing account had an insufficient reserve
726 // *prior* to our processing. If something actually crossed, then
727 // we allow this; otherwise, we just claim a fee.
728 if (!crossed)
729 result = tecINSUF_RESERVE_OFFER;
730
731 if (result != tesSUCCESS)
732 {
733 JLOG(j_.debug()) << "final result: " << transToken(result);
734 }
735
736 return {result, true};
737 }
738 }
739
740 // We need to place the remainder of the offer into its order book.
741 auto const offer_index = keylet::offer(account_, offerSequence);
742
743 // Add offer to owner's directory.
744 auto const ownerNode = sb.dirInsert(keylet::ownerDir(account_), offer_index, describeOwnerDir(account_));
745
746 if (!ownerNode)
747 {
748 // LCOV_EXCL_START
749 JLOG(j_.debug()) << "final result: failed to add offer to owner's directory";
750 return {tecDIR_FULL, true};
751 // LCOV_EXCL_STOP
752 }
753
754 // Update owner count.
755 adjustOwnerCount(sb, sleCreator, 1, viewJ);
756
757 JLOG(j_.trace()) << "adding to book: " << to_string(saTakerPays.issue()) << " : " << to_string(saTakerGets.issue())
758 << (domainID ? (" : " + to_string(*domainID)) : "");
759
760 Book const book{saTakerPays.issue(), saTakerGets.issue(), domainID};
761
762 // Add offer to order book, using the original rate
763 // before any crossing occurred.
764 //
765 // Regular offer - BookDirectory points to open directory
766 //
767 // Domain offer (w/o hybrid) - BookDirectory points to domain
768 // directory
769 //
770 // Hybrid domain offer - BookDirectory points to domain directory,
771 // and AdditionalBooks field stores one entry that points to the open
772 // directory
773 auto dir = keylet::quality(keylet::book(book), uRate);
774 bool const bookExisted = static_cast<bool>(sb.peek(dir));
775
776 auto setBookDir = [&](SLE::ref sle, std::optional<uint256> const& maybeDomain) {
777 sle->setFieldH160(sfTakerPaysCurrency, saTakerPays.issue().currency);
778 sle->setFieldH160(sfTakerPaysIssuer, saTakerPays.issue().account);
779 sle->setFieldH160(sfTakerGetsCurrency, saTakerGets.issue().currency);
780 sle->setFieldH160(sfTakerGetsIssuer, saTakerGets.issue().account);
781 sle->setFieldU64(sfExchangeRate, uRate);
782 if (maybeDomain)
783 sle->setFieldH256(sfDomainID, *maybeDomain);
784 };
785
786 auto const bookNode = sb.dirAppend(dir, offer_index, [&](SLE::ref sle) {
787 // sets domainID on book directory if it's a domain offer
788 setBookDir(sle, domainID);
789 });
790
791 if (!bookNode)
792 {
793 // LCOV_EXCL_START
794 JLOG(j_.debug()) << "final result: failed to add offer to book";
795 return {tecDIR_FULL, true};
796 // LCOV_EXCL_STOP
797 }
798
799 auto sleOffer = std::make_shared<SLE>(offer_index);
800 sleOffer->setAccountID(sfAccount, account_);
801 sleOffer->setFieldU32(sfSequence, offerSequence);
802 sleOffer->setFieldH256(sfBookDirectory, dir.key);
803 sleOffer->setFieldAmount(sfTakerPays, saTakerPays);
804 sleOffer->setFieldAmount(sfTakerGets, saTakerGets);
805 sleOffer->setFieldU64(sfOwnerNode, *ownerNode);
806 sleOffer->setFieldU64(sfBookNode, *bookNode);
807 if (expiration)
808 sleOffer->setFieldU32(sfExpiration, *expiration);
809 if (bPassive)
810 sleOffer->setFlag(lsfPassive);
811 if (bSell)
812 sleOffer->setFlag(lsfSell);
813 if (domainID)
814 sleOffer->setFieldH256(sfDomainID, *domainID);
815
816 // if it's a hybrid offer, set hybrid flag, and create an open dir
817 if (bHybrid)
818 {
819 auto const res = applyHybrid(sb, sleOffer, offer_index, saTakerPays, saTakerGets, setBookDir);
820 if (res != tesSUCCESS)
821 return {res, true}; // LCOV_EXCL_LINE
822 }
823
824 sb.insert(sleOffer);
825
826 if (!bookExisted)
827 ctx_.registry.getOrderBookDB().addOrderBook(book);
828
829 JLOG(j_.debug()) << "final result: success";
830
831 return {tesSUCCESS, true};
832}
833
834TER
835CreateOffer::doApply()
836{
837 // This is the ledger view that we work against. Transactions are applied
838 // as we go on processing transactions.
839 Sandbox sb(&ctx_.view());
840
841 // This is a ledger with just the fees paid and any unfunded or expired
842 // offers we encounter removed. It's used when handling Fill-or-Kill offers,
843 // if the order isn't going to be placed, to avoid wasting the work we did.
844 Sandbox sbCancel(&ctx_.view());
845
846 auto const result = applyGuts(sb, sbCancel);
847 if (result.second)
848 sb.apply(ctx_.rawView());
849 else
850 sbCancel.apply(ctx_.rawView());
851 return result.first;
852}
