11void
12ValidPermissionedDomain::visitEntry(
13 bool isDel,
14 std::shared_ptr<SLE const> const& before,
15 std::shared_ptr<SLE const> const& after)
16{
17 if (before && before->getType() != ltPERMISSIONED_DOMAIN)
18 return;
19 if (after && after->getType() != ltPERMISSIONED_DOMAIN)
20 return;
21
22 auto check = [isDel](std::vector<SleStatus>& sleStatus, std::shared_ptr<SLE const> const& sle) {
23 auto const& credentials = sle->getFieldArray(sfAcceptedCredentials);
24 auto const sorted = credentials::makeSorted(credentials);
25
26 SleStatus ss{credentials.size(), false, !sorted.empty(), isDel};
27
28 // If array have duplicates then all the other checks are invalid
29 if (ss.isUnique_)
30 {
31 unsigned i = 0;
32 for (auto const& cred : sorted)
33 {
34 auto const& credTx = credentials[i++];
35 ss.isSorted_ =
36 (cred.first == credTx[sfIssuer]) && (cred.second == credTx[sfCredentialType]);
37 if (!ss.isSorted_)
38 break;
39 }
40 }
41 sleStatus.emplace_back(std::move(ss));
42 };
43
44 if (after)
45 check(sleStatus_, after);
46}

48bool
49ValidPermissionedDomain::finalize(
50 STTx const& tx,
51 TER const result,
52 XRPAmount const,
53 ReadView const& view,
54 beast::Journal const& j)
55{
56 auto check = [](SleStatus const& sleStatus, beast::Journal const& j) {
57 if (!sleStatus.credentialsSize_)
58 {
59 JLOG(j.fatal()) << "Invariant failed: permissioned domain with "
60 "no rules.";
61 return false;
62 }
63
64 if (sleStatus.credentialsSize_ > maxPermissionedDomainCredentialsArraySize)
65 {
66 JLOG(j.fatal()) << "Invariant failed: permissioned domain bad "
67 "credentials size "
68 << sleStatus.credentialsSize_;
69 return false;
70 }
71
72 if (!sleStatus.isUnique_)
73 {
74 JLOG(j.fatal()) << "Invariant failed: permissioned domain credentials "
75 "aren't unique";
76 return false;
77 }
78
79 if (!sleStatus.isSorted_)
80 {
81 JLOG(j.fatal()) << "Invariant failed: permissioned domain credentials "
82 "aren't sorted";
83 return false;
84 }
85
86 return true;
87 };
88
89 if (view.rules().enabled(fixPermissionedDomainInvariant))
90 {
91 // No permissioned domains should be affected if the transaction failed
92 if (result != tesSUCCESS)
93 // If nothing changed, all is good. If there were changes, that's
94 // bad.
95 return sleStatus_.empty();
96
97 if (sleStatus_.size() > 1)
98 {
99 JLOG(j.fatal()) << "Invariant failed: transaction affected more "
100 "than 1 permissioned domain entry.";
101 return false;
102 }
103
104 switch (tx.getTxnType())
105 {
106 case ttPERMISSIONED_DOMAIN_SET: {
107 if (sleStatus_.empty())
108 {
109 JLOG(j.fatal()) << "Invariant failed: no domain objects affected by "
110 "PermissionedDomainSet";
111 return false;
112 }
113
114 auto const& sleStatus = sleStatus_[0];
115 if (sleStatus.isDelete_)
116 {
117 JLOG(j.fatal()) << "Invariant failed: domain object "
118 "deleted by PermissionedDomainSet";
119 return false;
120 }
121 return check(sleStatus, j);
122 }
123 case ttPERMISSIONED_DOMAIN_DELETE: {
124 if (sleStatus_.empty())
125 {
126 JLOG(j.fatal()) << "Invariant failed: no domain objects affected by "
127 "PermissionedDomainDelete";
128 return false;
129 }
130
131 if (!sleStatus_[0].isDelete_)
132 {
133 JLOG(j.fatal()) << "Invariant failed: domain object "
134 "modified, but not deleted by "
135 "PermissionedDomainDelete";
136 return false;
137 }
138 return true;
139 }
140 default: {
141 if (!sleStatus_.empty())
142 {
143 JLOG(j.fatal()) << "Invariant failed: " << sleStatus_.size()
144 << " domain object(s) affected by an "
145 "unauthorized transaction. "
146 << tx.getTxnType();
147 return false;
148 }
149 return true;
150 }
151 }
152 }
153 else
154 {
155 if (tx.getTxnType() != ttPERMISSIONED_DOMAIN_SET || result != tesSUCCESS ||
156 sleStatus_.empty())
157 return true;
158 return check(sleStatus_[0], j);
159 }
160}
