169template <class>
170bool
171runner::run(suite_info const& s)
172{
173 // Enable 'default' testcase
174 default_ = true;
175 failed_ = false;
176 on_suite_begin(s);
177 s.run(*this);
178 // Forgot to call pass or fail.
179 BOOST_ASSERT(cond_);
180 on_case_end();
181 on_suite_end();
182 return failed_;
183}

185template <class FwdIter>

187runner::run(FwdIter first, FwdIter last)
188{
189 bool failed(false);
190 for (; first != last; ++first)

227template <class>
228void
229runner::testcase(std::string const& name)
230{
231 std::lock_guard lock(mutex_);
232 // Name may not be empty
233 BOOST_ASSERT(default_ || !name.empty());
234 // Forgot to call pass or fail
235 BOOST_ASSERT(default_ || cond_);
236 if (!default_)
237 on_case_end();
238 default_ = false;
239 cond_ = false;
240 on_case_begin(name);
241}
