165bool
166RawStateTable::exists(ReadView const& base, Keylet const& k) const
167{
168 XRPL_ASSERT(k.key.isNonZero(), "xrpl::detail::RawStateTable::exists : nonzero key");
169 auto const iter = items_.find(k.key);
170 if (iter == items_.end())
171 return base.exists(k);
172 auto const& item = iter->second;
173 if (item.action == Action::erase)
174 return false;
175 if (!k.check(*item.sle))
176 return false;
177 return true;
178}

180/* This works by first calculating succ() on the parent,
181 then calculating succ() our internal list, and taking
182 the lower of the two.
183*/
184auto
185RawStateTable::succ(ReadView const& base, key_type const& key, std::optional<key_type> const& last) const
186 -> std::optional<key_type>
187{
188 std::optional<key_type> next = key;
189 items_t::const_iterator iter;
190 // Find base successor that is
191 // not also deleted in our list
192 do
193 {
194 next = base.succ(*next, last);
195 if (!next)
196 break;
197 iter = items_.find(*next);
198 } while (iter != items_.end() && iter->second.action == Action::erase);
199 // Find non-deleted successor in our list
200 for (iter = items_.upper_bound(key); iter != items_.end(); ++iter)
201 {
202 if (iter->second.action != Action::erase)
203 {
204 // Found both, return the lower key
205 if (!next || next > iter->first)
206 next = iter->first;
207 break;
208 }
209 }
210 // Nothing in our list, return
211 // what we got from the parent.
212 if (last && next >= last)
213 return std::nullopt;
214 return next;
215}

218RawStateTable::erase(std::shared_ptr<SLE> const& sle)
219{
220 // The base invariant is checked during apply
221 auto const result = items_.emplace(

223 if (result.second)
224 return;
225 auto& item = result.first->second;

229 LogicError("RawStateTable::erase: already erased");
230 break;
231 case Action::insert:

233 break;
234 case Action::replace:
235 item.action = Action::erase;
236 item.sle = sle;
237 break;
238 }
