22Json::Value
23doChannelAuthorize(RPC::JsonContext& context)
24{
25 if (context.role != Role::ADMIN && !context.app.config().canSign())
26 {
27 return RPC::make_error(rpcNOT_SUPPORTED, "Signing is not supported by this server.");
28 }
29
30 auto const& params(context.params);
31 for (auto const& p : {jss::channel_id, jss::amount})
32 if (!params.isMember(p))
33 return RPC::missing_field_error(p);
34
35 // Compatibility if a key type isn't specified. If it is, the
36 // keypairForSignature code will validate parameters and return
37 // the appropriate error.
38 if (!params.isMember(jss::key_type) && !params.isMember(jss::secret))
39 return RPC::missing_field_error(jss::secret);
40
41 Json::Value result;
42 std::optional<std::pair<PublicKey, SecretKey>> const keyPair =
43 RPC::keypairForSignature(params, result, context.apiVersion);
44
45 XRPL_ASSERT(keyPair || RPC::contains_error(result), "xrpl::doChannelAuthorize : valid keyPair or an error");
46 if (!keyPair || RPC::contains_error(result))
47 return result;
48
49 PublicKey const& pk = keyPair->first;
50 SecretKey const& sk = keyPair->second;
51
52 uint256 channelId;
53 if (!channelId.parseHex(params[jss::channel_id].asString()))
54 return rpcError(rpcCHANNEL_MALFORMED);
55
56 std::optional<std::uint64_t> const optDrops =
57 params[jss::amount].isString() ? to_uint64(params[jss::amount].asString()) : std::nullopt;
58
59 if (!optDrops)
60 return rpcError(rpcCHANNEL_AMT_MALFORMED);
61
62 std::uint64_t const drops = *optDrops;
63
64 Serializer msg;
65 serializePayChanAuthorization(msg, channelId, XRPAmount(drops));
66
67 try
68 {

70 result[jss::signature] = strHex(buf);
71 }
72 catch (std::exception const& ex)
73 {
74 // LCOV_EXCL_START

76 // LCOV_EXCL_STOP

100 if (!pk)

103 if (!pkHex)
104 return rpcError(rpcPUBLIC_MALFORMED);
105 auto const pkType = publicKeyType(makeSlice(*pkHex));
106 if (!pkType)
107 return rpcError(rpcPUBLIC_MALFORMED);
108 pk.emplace(makeSlice(*pkHex));
109 }
110 }

114 return rpcError(rpcCHANNEL_MALFORMED);
