18textTime(
19 std::string& text,
20 UptimeClock::time_point& seconds,
21 char const* unitName,
22 std::chrono::seconds unitVal)
23{
24 auto i = seconds.time_since_epoch() / unitVal;
25
26 if (i == 0)
27 return;
28
29 seconds -= unitVal * i;
30
31 if (!text.empty())
32 text += ", ";
