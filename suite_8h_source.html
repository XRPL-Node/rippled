23template <class String>
24static std::string
25make_reason(String const& reason, char const* file, int line)
26{
27 std::string s(reason);
28 if (!s.empty())
29 s.append(": ");
30 namespace fs = boost::filesystem;
31 s.append(fs::path{file}.filename().string());
32 s.append("(");
33 s.append(boost::lexical_cast<std::string>(line));
34 s.append(")");
35 return s;
36}

529template <class>
530void
531suite::run(runner& r)
532{
533 runner_ = &r;
534
535 try
536 {
537 run();
538 }
539 catch (abort_exception const&)
540 {
541 // ends the suite
542 }
543 catch (std::exception const& e)
544 {
545 runner_->fail("unhandled exception: " + std::string(e.what()));
546 }
547 catch (...)
548 {
549 runner_->fail("unhandled exception");
550 }
551}
