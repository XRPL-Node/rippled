97bool
98STValidation::isValid() const noexcept
99{
100 if (!valid_)
101 {
102 XRPL_ASSERT(
103 publicKeyType(getSignerPublic()) == KeyType::secp256k1, "xrpl::STValidation::isValid : valid key type");
104
105 valid_ = verifyDigest(
106 getSignerPublic(), getSigningHash(), makeSlice(getFieldVL(sfSignature)), getFlags() & vfFullyCanonicalSig);
