290 PrettyAmount
291 operator()(Number v) const
292 {
293 auto const c = dropsPerXRP.drops();
294 auto const d = std::int64_t(v * c);
295 if (Number(d) / c != v)
296 Throw<std::domain_error>("unrepresentable");
297 return {d};
298 }

300 PrettyAmount
301 operator()(double v) const
302 {
303 auto const c = dropsPerXRP.drops();
304 if (v >= 0)
305 {
306 auto const d = std::uint64_t(std::round(v * c));
307 if (double(d) / c != v)
308 Throw<std::domain_error>("unrepresentable");
309 return {d};
310 }
311 auto const d = std::int64_t(std::round(v * c));
312 if (double(d) / c != v)

314 return {d};
