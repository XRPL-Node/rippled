56// Represents a span of ancestry of a ledger
57template <class Ledger>
58class Span
59{
60 using Seq = typename Ledger::Seq;
61 using ID = typename Ledger::ID;
62
63 // The span is the half-open interval [start,end) of ledger_
64 Seq start_{0};
65 Seq end_{1};
66 Ledger ledger_;
67
68public:
69 Span() : ledger_{typename Ledger::MakeGenesis{}}
70 {
71 // Require default ledger to be genesis seq
72 XRPL_ASSERT(ledger_.seq() == start_, "xrpl::Span::Span : ledger is genesis");
73 }
74
75 Span(Ledger ledger) : start_{0}, end_{ledger.seq() + Seq{1}}, ledger_{std::move(ledger)}
76 {
77 }
78
79 Span(Span const& s) = default;
80 Span(Span&& s) = default;
81 Span&
82 operator=(Span const&) = default;
83 Span&
84 operator=(Span&&) = default;
85
86 Seq
87 start() const
88 {
89 return start_;
90 }
91
92 Seq
93 end() const
94 {
95 return end_;
96 }
97
98 // Return the Span from [spot,end_) or none if no such valid span
99 std::optional<Span>
100 from(Seq spot) const
101 {
102 return sub(spot, end_);
103 }
104
105 // Return the Span from [start_,spot) or none if no such valid span
106 std::optional<Span>
107 before(Seq spot) const
108 {
109 return sub(start_, spot);
110 }
111
112 // Return the ID of the ledger that starts this span
113 ID
114 startID() const
115 {
116 return ledger_[start_];
117 }
118
119 // Return the ledger sequence number of the first possible difference
120 // between this span and a given ledger.
121 Seq
122 diff(Ledger const& o) const
123 {
124 return clamp(mismatch(ledger_, o));
125 }
126
127 // The tip of this span
128 SpanTip<Ledger>
129 tip() const
130 {
131 Seq tipSeq{end_ - Seq{1}};
132 return SpanTip<Ledger>{tipSeq, ledger_[tipSeq], ledger_};
133 }
134
135private:
136 Span(Seq start, Seq end, Ledger const& l) : start_{start}, end_{end}, ledger_{l}
137 {
138 // Spans cannot be empty
139 XRPL_ASSERT(start < end, "xrpl::Span::Span : non-empty span input");
140 }
141
142 Seq
143 clamp(Seq val) const
144 {
145 return std::min(std::max(start_, val), end_);
146 }
147
148 // Return a span of this over the half-open interval [from,to)
149 std::optional<Span>
150 sub(Seq from, Seq to) const
151 {
152 Seq newFrom = clamp(from);
153 Seq newTo = clamp(to);
154 if (newFrom < newTo)
155 return Span(newFrom, newTo, ledger_);
156 return std::nullopt;
157 }
158
159 friend std::ostream&
160 operator<<(std::ostream& o, Span const& s)
161 {
162 return o << s.tip().id << "[" << s.start_ << "," << s.end_ << ")";
163 }
164
165 friend Span
166 merge(Span const& a, Span const& b)
167 {
168 // Return combined span, using ledger_ from higher sequence span
169 if (a.end_ < b.end_)
170 return Span(std::min(a.start_, b.start_), b.end_, b.ledger_);
171
172 return Span(std::min(a.start_, b.start_), a.end_, a.ledger_);
173 }

331 // The root of the trie. The root is allowed to break the no-single child
332 // invariant.
333 std::unique_ptr<Node> root;
334

344 std::pair<Node*, Seq>
345 find(Ledger const& ledger) const
346 {
347 Node* curr = root.get();
348
349 // Root is always defined and is in common with all ledgers
350 XRPL_ASSERT(curr, "xrpl::LedgerTrie::find : non-null root");
351 Seq pos = curr->span.diff(ledger);
352
353 bool done = false;
354
355 // Continue searching for a better span as long as the current position
356 // matches the entire span
357 while (!done && pos == curr->span.end())
358 {
359 done = true;
360 // Find the child with the longest ancestry match
361 for (std::unique_ptr<Node> const& child : curr->children)
362 {
363 auto const childPos = child->span.diff(ledger);
364 if (childPos > pos)
365 {
366 done = false;
367 pos = childPos;
368 curr = child.get();
369 break;
370 }
371 }
372 }
373 return std::make_pair(curr, pos);
374 }
375
382 Node*
383 findByLedgerID(Ledger const& ledger, Node* parent = nullptr) const
384 {
385 if (!parent)
386 parent = root.get();
387 if (ledger.id() == parent->span.tip().id)
388 return parent;
389 for (auto const& child : parent->children)
390 {
391 auto cl = findByLedgerID(ledger, child.get());
392 if (cl)
393 return cl;
394 }
395 return nullptr;
396 }

424 void
425 insert(Ledger const& ledger, std::uint32_t count = 1)
426 {
427 auto const [loc, diffSeq] = find(ledger);
428
429 // There is always a place to insert
430 XRPL_ASSERT(loc, "xrpl::LedgerTrie::insert : valid input ledger");
431
432 // Node from which to start incrementing branchSupport
433 Node* incNode = loc;
434
435 // loc->span has the longest common prefix with Span{ledger} of all
436 // existing nodes in the trie. The optional<Span>'s below represent
437 // the possible common suffixes between loc->span and Span{ledger}.
438 //
439 // loc->span
440 // a b c | d e f
441 // prefix | oldSuffix
442 //
443 // Span{ledger}
444 // a b c | g h i
445 // prefix | newSuffix
446
447 std::optional<Span> prefix = loc->span.before(diffSeq);
448 std::optional<Span> oldSuffix = loc->span.from(diffSeq);
449 std::optional<Span> newSuffix = Span{ledger}.from(diffSeq);
450
451 if (oldSuffix)
452 {
453 // Have
454 // abcdef -> ....
455 // Inserting
456 // abc
457 // Becomes
458 // abc -> def -> ...
459
460 // Create oldSuffix node that takes over loc
461 auto newNode = std::make_unique<Node>(*oldSuffix);
462 newNode->tipSupport = loc->tipSupport;
463 newNode->branchSupport = loc->branchSupport;
464 newNode->children = std::move(loc->children);
465 XRPL_ASSERT(loc->children.empty(), "xrpl::LedgerTrie::insert : moved-from children");
466 for (std::unique_ptr<Node>& child : newNode->children)
467 child->parent = newNode.get();
468
469 // Loc truncates to prefix and newNode is its child
470 XRPL_ASSERT(prefix, "xrpl::LedgerTrie::insert : prefix is set");
471 loc->span = *prefix;
472 newNode->parent = loc;
473 loc->children.emplace_back(std::move(newNode));
474 loc->tipSupport = 0;
475 }
476 if (newSuffix)
477 {
478 // Have
479 // abc -> ...
480 // Inserting
481 // abcdef-> ...
482 // Becomes
483 // abc -> ...
484 // \-> def
485
486 auto newNode = std::make_unique<Node>(*newSuffix);
487 newNode->parent = loc;
488 // increment support starting from the new node
489 incNode = newNode.get();
490 loc->children.push_back(std::move(newNode));
491 }
492
493 incNode->tipSupport += count;
494 while (incNode)
495 {
496 incNode->branchSupport += count;
497 incNode = incNode->parent;
498 }
499
500 seqSupport[ledger.seq()] += count;
501 }

510 bool
511 remove(Ledger const& ledger, std::uint32_t count = 1)
512 {
513 Node* loc = findByLedgerID(ledger);
514 // Must be exact match with tip support
515 if (!loc || loc->tipSupport == 0)
516 return false;
517
518 // found our node, remove it
519 count = std::min(count, loc->tipSupport);
520 loc->tipSupport -= count;
521
522 auto const it = seqSupport.find(ledger.seq());
523 XRPL_ASSERT(
524 it != seqSupport.end() && it->second >= count,
525 "xrpl::LedgerTrie::remove : valid input ledger");
526 it->second -= count;
527 if (it->second == 0)
528 seqSupport.erase(it->first);
529
530 Node* decNode = loc;
531 while (decNode)
532 {
533 decNode->branchSupport -= count;
534 decNode = decNode->parent;
535 }
536
537 while (loc->tipSupport == 0 && loc != root.get())
538 {
539 Node* parent = loc->parent;
540 if (loc->children.empty())
541 {
542 // this node can be erased
543 parent->erase(loc);
544 }
545 else if (loc->children.size() == 1)
546 {
547 // This node can be combined with its child
548 std::unique_ptr<Node> child = std::move(loc->children.front());
549 child->span = merge(loc->span, child->span);
550 child->parent = parent;
551 parent->children.emplace_back(std::move(child));
552 parent->erase(loc);
553 }
554 else
555 break;
556 loc = parent;
557 }
558 return true;
559 }

580 std::uint32_t
581 branchSupport(Ledger const& ledger) const
582 {
583 Node const* loc = findByLedgerID(ledger);
584 if (!loc)
585 {
586 Seq diffSeq;
587 std::tie(loc, diffSeq) = find(ledger);
588 // Check that ledger is a proper prefix of loc
589 if (!(diffSeq > ledger.seq() && ledger.seq() < loc->span.end()))
590 loc = nullptr;
591 }
592 return loc ? loc->branchSupport : 0;
593 }

654 std::optional<SpanTip<Ledger>>
655 getPreferred(Seq const largestIssued) const
656 {
657 if (empty())
658 return std::nullopt;
659
660 Node* curr = root.get();
661
662 bool done = false;
663
664 std::uint32_t uncommitted = 0;
665 auto uncommittedIt = seqSupport.begin();
666
667 while (curr && !done)
668 {
669 // Within a single span, the preferred by branch strategy is simply
670 // to continue along the span as long as the branch support of
671 // the next ledger exceeds the uncommitted support for that ledger.
672 {
673 // Add any initial uncommitted support prior for ledgers
674 // earlier than nextSeq or earlier than largestIssued
675 Seq nextSeq = curr->span.start() + Seq{1};
676 while (uncommittedIt != seqSupport.end() &&
677 uncommittedIt->first < std::max(nextSeq, largestIssued))
678 {
679 uncommitted += uncommittedIt->second;
680 uncommittedIt++;
681 }
682
683 // Advance nextSeq along the span
684 while (nextSeq < curr->span.end() && curr->branchSupport > uncommitted)
685 {
686 // Jump to the next seqSupport change
687 if (uncommittedIt != seqSupport.end() &&
688 uncommittedIt->first < curr->span.end())
689 {
690 nextSeq = uncommittedIt->first + Seq{1};
691 uncommitted += uncommittedIt->second;
692 uncommittedIt++;
693 }
694 else // otherwise we jump to the end of the span
695 nextSeq = curr->span.end();
696 }
697 // We did not consume the entire span, so we have found the
698 // preferred ledger
699 if (nextSeq < curr->span.end())
700 return curr->span.before(nextSeq)->tip();
701 }
702
703 // We have reached the end of the current span, so we need to
704 // find the best child
705 Node* best = nullptr;
706 std::uint32_t margin = 0;
707 if (curr->children.size() == 1)
708 {
709 best = curr->children[0].get();
710 margin = best->branchSupport;
711 }
712 else if (!curr->children.empty())
713 {
714 // Sort placing children with largest branch support in the
715 // front, breaking ties with the span's starting ID
716 std::partial_sort(
717 curr->children.begin(),
718 curr->children.begin() + 2,
719 curr->children.end(),
720 [](std::unique_ptr<Node> const& a, std::unique_ptr<Node> const& b) {
721 return std::make_tuple(a->branchSupport, a->span.startID()) >
722 std::make_tuple(b->branchSupport, b->span.startID());
723 });
724
725 best = curr->children[0].get();
726 margin = curr->children[0]->branchSupport - curr->children[1]->branchSupport;
727
728 // If best holds the tie-breaker, gets one larger margin
729 // since the second best needs additional branchSupport
730 // to overcome the tie
731 if (best->span.startID() > curr->children[1]->span.startID())
732 margin++;
733 }
734
735 // If the best child has margin exceeding the uncommitted support,
736 // continue from that child, otherwise we are done
737 if (best && ((margin > uncommitted) || (uncommitted == 0)))
738 curr = best;
739 else // current is the best
740 done = true;
741 }
742 return curr->span.tip();
743 }

776 bool
777 checkInvariants() const
778 {
779 std::map<Seq, std::uint32_t> expectedSeqSupport;
780
781 std::stack<Node const*> nodes;
782 nodes.push(root.get());
783 while (!nodes.empty())
784 {
785 Node const* curr = nodes.top();
786 nodes.pop();
787 if (!curr)
788 continue;
789
790 // Node with 0 tip support must have multiple children
791 // unless it is the root node
792 if (curr != root.get() && curr->tipSupport == 0 && curr->children.size() < 2)
793 return false;
794
795 // branchSupport = tipSupport + sum(child->branchSupport)
796 std::size_t support = curr->tipSupport;
797 if (curr->tipSupport != 0)
798 expectedSeqSupport[curr->span.end() - Seq{1}] += curr->tipSupport;
799
800 for (auto const& child : curr->children)
801 {
802 if (child->parent != curr)
803 return false;
804
805 support += child->branchSupport;
806 nodes.push(child.get());
807 }
808 if (support != curr->branchSupport)
809 return false;
810 }
811 return expectedSeqSupport == seqSupport;
812 }
