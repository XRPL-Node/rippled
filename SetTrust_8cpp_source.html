14uint32_t
15computeFreezeFlags(
16 uint32_t uFlags,
17 bool bHigh,
18 bool bNoFreeze,
19 bool bSetFreeze,
20 bool bClearFreeze,
21 bool bSetDeepFreeze,
22 bool bClearDeepFreeze)
23{
24 if (bSetFreeze && !bClearFreeze && !bNoFreeze)
25 {
26 uFlags |= (bHigh ? xrpl::lsfHighFreeze : xrpl::lsfLowFreeze);
27 }
28 else if (bClearFreeze && !bSetFreeze)
29 {
30 uFlags &= ~(bHigh ? xrpl::lsfHighFreeze : xrpl::lsfLowFreeze);
31 }
32 if (bSetDeepFreeze && !bClearDeepFreeze && !bNoFreeze)
33 {
34 uFlags |= (bHigh ? xrpl::lsfHighDeepFreeze : xrpl::lsfLowDeepFreeze);
35 }
36 else if (bClearDeepFreeze && !bSetDeepFreeze)
37 {
38 uFlags &= ~(bHigh ? xrpl::lsfHighDeepFreeze : xrpl::lsfLowDeepFreeze);
39 }
40
41 return uFlags;
42}

54NotTEC
55SetTrust::preflight(PreflightContext const& ctx)
56{
57 auto& tx = ctx.tx;
58 auto& j = ctx.j;
59
60 std::uint32_t const uTxFlags = tx.getFlags();
61
62 if (!ctx.rules.enabled(featureDeepFreeze))
63 {
64 // Even though the deep freeze flags are included in the
65 // `tfTrustSetMask`, they are not valid if the amendment is not enabled.
66 if (uTxFlags & (tfSetDeepFreeze | tfClearDeepFreeze))
67 {
68 return temINVALID_FLAG;
69 }
70 }
71
72 STAmount const saLimitAmount(tx.getFieldAmount(sfLimitAmount));
73
74 if (!isLegalNet(saLimitAmount))
75 return temBAD_AMOUNT;
76
77 if (saLimitAmount.native())
78 {
79 JLOG(j.trace()) << "Malformed transaction: specifies native limit " << saLimitAmount.getFullText();
80 return temBAD_LIMIT;
81 }
82
83 if (badCurrency() == saLimitAmount.getCurrency())
84 {
85 JLOG(j.trace()) << "Malformed transaction: specifies XRP as IOU";
86 return temBAD_CURRENCY;
87 }
88
89 if (saLimitAmount < beast::zero)
90 {
91 JLOG(j.trace()) << "Malformed transaction: Negative credit limit.";
92 return temBAD_LIMIT;
93 }
94
95 // Check if destination makes sense.
96 auto const& issuer = saLimitAmount.getIssuer();
97
98 if (!issuer || issuer == noAccount())
99 {
100 JLOG(j.trace()) << "Malformed transaction: no destination account.";
101 return temDST_NEEDED;
102 }
103
104 return tesSUCCESS;
105}

107NotTEC
108SetTrust::checkPermission(ReadView const& view, STTx const& tx)
109{
110 auto const delegate = tx[~sfDelegate];
111 if (!delegate)
112 return tesSUCCESS;
113
114 auto const delegateKey = keylet::delegate(tx[sfAccount], *delegate);
115 auto const sle = view.read(delegateKey);
116
117 if (!sle)
118 return terNO_DELEGATE_PERMISSION;
119
120 if (checkTxPermission(sle, tx) == tesSUCCESS)
121 return tesSUCCESS;
122
123 std::uint32_t const txFlags = tx.getFlags();
124
125 // Currently we only support TrustlineAuthorize, TrustlineFreeze and
126 // TrustlineUnfreeze granular permission. Setting other flags returns
127 // error.
128 if (txFlags & tfTrustSetPermissionMask)
129 return terNO_DELEGATE_PERMISSION;
130
131 if (tx.isFieldPresent(sfQualityIn) || tx.isFieldPresent(sfQualityOut))
132 return terNO_DELEGATE_PERMISSION;
133
134 auto const saLimitAmount = tx.getFieldAmount(sfLimitAmount);
135 auto const sleRippleState =
136 view.read(keylet::line(tx[sfAccount], saLimitAmount.getIssuer(), saLimitAmount.getCurrency()));
137
138 // if the trustline does not exist, granular permissions are
139 // not allowed to create trustline
140 if (!sleRippleState)
141 return terNO_DELEGATE_PERMISSION;
142
143 std::unordered_set<GranularPermissionType> granularPermissions;
144 loadGranularPermission(sle, ttTRUST_SET, granularPermissions);
145
146 if (txFlags & tfSetfAuth && !granularPermissions.contains(TrustlineAuthorize))
147 return terNO_DELEGATE_PERMISSION;
148 if (txFlags & tfSetFreeze && !granularPermissions.contains(TrustlineFreeze))
149 return terNO_DELEGATE_PERMISSION;
150 if (txFlags & tfClearFreeze && !granularPermissions.contains(TrustlineUnfreeze))
151 return terNO_DELEGATE_PERMISSION;
152
153 // updating LimitAmount is not allowed only with granular permissions,
154 // unless there's a new granular permission for this in the future.
155 auto const curLimit = tx[sfAccount] > saLimitAmount.getIssuer() ? sleRippleState->getFieldAmount(sfHighLimit)
156 : sleRippleState->getFieldAmount(sfLowLimit);
157
158 STAmount saLimitAllow = saLimitAmount;
159 saLimitAllow.setIssuer(tx[sfAccount]);
160
161 if (curLimit != saLimitAllow)
162 return terNO_DELEGATE_PERMISSION;
163
164 return tesSUCCESS;
165}

167TER
168SetTrust::preclaim(PreclaimContext const& ctx)
169{
170 auto const id = ctx.tx[sfAccount];
171
172 auto const sle = ctx.view.read(keylet::account(id));
173 if (!sle)
174 return terNO_ACCOUNT;
175
176 std::uint32_t const uTxFlags = ctx.tx.getFlags();
177
178 bool const bSetAuth = (uTxFlags & tfSetfAuth);
179
180 if (bSetAuth && !(sle->getFieldU32(sfFlags) & lsfRequireAuth))
181 {
182 JLOG(ctx.j.trace()) << "Retry: Auth not required.";
183 return tefNO_AUTH_REQUIRED;
184 }
185
186 auto const saLimitAmount = ctx.tx[sfLimitAmount];
187
188 auto const currency = saLimitAmount.getCurrency();
189 auto const uDstAccountID = saLimitAmount.getIssuer();
190
191 if (id == uDstAccountID)
192 return temDST_IS_SRC;
193
194 // This might be nullptr
195 auto const sleDst = ctx.view.read(keylet::account(uDstAccountID));
196 if ((ammEnabled(ctx.view.rules()) || ctx.view.rules().enabled(featureSingleAssetVault)) && sleDst == nullptr)
197 return tecNO_DST;
198
199 // If the destination has opted to disallow incoming trustlines
200 // then honour that flag
201 if (sleDst->getFlags() & lsfDisallowIncomingTrustline)
202 {
203 // The original implementation of featureDisallowIncoming was
204 // too restrictive. If
205 // o fixDisallowIncomingV1 is enabled and
206 // o The trust line already exists
207 // Then allow the TrustSet.
208 if (ctx.view.rules().enabled(fixDisallowIncomingV1) &&
209 ctx.view.exists(keylet::line(id, uDstAccountID, currency)))
210 {
211 // pass
212 }
213 else
214 return tecNO_PERMISSION;
215 }
216
217 // In general, trust lines to pseudo accounts are not permitted, unless
218 // enabled in the code section below, for specific cases. This block is not
219 // amendment-gated because sleDst will not have a pseudo-account designator
220 // field populated, unless the appropriate amendment was already enabled.
221 if (sleDst && isPseudoAccount(sleDst))
222 {
223 // If destination is AMM and the trustline doesn't exist then only allow
224 // SetTrust if the asset is AMM LP token and AMM is not in empty state.
225 if (sleDst->isFieldPresent(sfAMMID))
226 {
227 if (ctx.view.exists(keylet::line(id, uDstAccountID, currency)))
228 {
229 // pass
230 }
231 else if (auto const ammSle = ctx.view.read({ltAMM, sleDst->getFieldH256(sfAMMID)}))
232 {
233 if (auto const lpTokens = ammSle->getFieldAmount(sfLPTokenBalance); lpTokens == beast::zero)
234 return tecAMM_EMPTY;
235 else if (lpTokens.getCurrency() != saLimitAmount.getCurrency())
236 return tecNO_PERMISSION;
237 }
238 else
239 return tecINTERNAL; // LCOV_EXCL_LINE
240 }
241 else if (sleDst->isFieldPresent(sfVaultID) || sleDst->isFieldPresent(sfLoanBrokerID))
242 {
243 if (!ctx.view.exists(keylet::line(id, uDstAccountID, currency)))
244 return tecNO_PERMISSION;
245 // else pass
246 }
247 else
248 return tecPSEUDO_ACCOUNT;
249 }
250
251 // Checking all freeze/deep freeze flag invariants.
252 if (ctx.view.rules().enabled(featureDeepFreeze))
253 {
254 bool const bNoFreeze = sle->isFlag(lsfNoFreeze);
255 bool const bSetFreeze = (uTxFlags & tfSetFreeze);
256 bool const bSetDeepFreeze = (uTxFlags & tfSetDeepFreeze);
257
258 if (bNoFreeze && (bSetFreeze || bSetDeepFreeze))
259 {
260 // Cannot freeze the trust line if NoFreeze is set
261 return tecNO_PERMISSION;
262 }
263
264 bool const bClearFreeze = (uTxFlags & tfClearFreeze);
265 bool const bClearDeepFreeze = (uTxFlags & tfClearDeepFreeze);
266 if ((bSetFreeze || bSetDeepFreeze) && (bClearFreeze || bClearDeepFreeze))
267 {
268 // Freezing and unfreezing in the same transaction should be
269 // illegal
270 return tecNO_PERMISSION;
271 }
272
273 bool const bHigh = id > uDstAccountID;
274 // Fetching current state of trust line
275 auto const sleRippleState = ctx.view.read(keylet::line(id, uDstAccountID, currency));
276 std::uint32_t uFlags = sleRippleState ? sleRippleState->getFieldU32(sfFlags) : 0u;
277 // Computing expected trust line state
278 uFlags =
279 computeFreezeFlags(uFlags, bHigh, bNoFreeze, bSetFreeze, bClearFreeze, bSetDeepFreeze, bClearDeepFreeze);
280
281 auto const frozen = uFlags & (bHigh ? lsfHighFreeze : lsfLowFreeze);
282 auto const deepFrozen = uFlags & (bHigh ? lsfHighDeepFreeze : lsfLowDeepFreeze);
283
284 // Trying to set deep freeze on not already frozen trust line must
285 // fail. This also checks that clearing normal freeze while deep
286 // frozen must not work
287 if (deepFrozen && !frozen)
288 {
289 return tecNO_PERMISSION;
290 }
291 }
292
293 return tesSUCCESS;
294}

296TER
297SetTrust::doApply()
298{
299 TER terResult = tesSUCCESS;
300
301 STAmount const saLimitAmount(ctx_.tx.getFieldAmount(sfLimitAmount));
302 bool const bQualityIn(ctx_.tx.isFieldPresent(sfQualityIn));
303 bool const bQualityOut(ctx_.tx.isFieldPresent(sfQualityOut));
304
305 Currency const currency(saLimitAmount.getCurrency());
306 AccountID uDstAccountID(saLimitAmount.getIssuer());
307
308 // true, if current is high account.
309 bool const bHigh = account_ > uDstAccountID;
310
311 auto const sle = view().peek(keylet::account(account_));
312 if (!sle)
313 return tefINTERNAL; // LCOV_EXCL_LINE
314
315 std::uint32_t const uOwnerCount = sle->getFieldU32(sfOwnerCount);
316
317 // The reserve that is required to create the line. Note
318 // that although the reserve increases with every item
319 // an account owns, in the case of trust lines we only
320 // *enforce* a reserve if the user owns more than two
321 // items.
322 //
323 // We do this because being able to exchange currencies,
324 // which needs trust lines, is a powerful Ripple feature.
325 // So we want to make it easy for a gateway to fund the
326 // accounts of its users without fear of being tricked.
327 //
328 // Without this logic, a gateway that wanted to have a
329 // new user use its services, would have to give that
330 // user enough XRP to cover not only the account reserve
331 // but the incremental reserve for the trust line as
332 // well. A person with no intention of using the gateway
333 // could use the extra XRP for their own purposes.
334
335 XRPAmount const reserveCreate(
336 (uOwnerCount < 2) ? XRPAmount(beast::zero) : view().fees().accountReserve(uOwnerCount + 1));
337
338 std::uint32_t uQualityIn(bQualityIn ? ctx_.tx.getFieldU32(sfQualityIn) : 0);
339 std::uint32_t uQualityOut(bQualityOut ? ctx_.tx.getFieldU32(sfQualityOut) : 0);
340
341 if (bQualityOut && QUALITY_ONE == uQualityOut)
342 uQualityOut = 0;
343
344 std::uint32_t const uTxFlags = ctx_.tx.getFlags();
345
346 bool const bSetAuth = (uTxFlags & tfSetfAuth);
347 bool const bSetNoRipple = (uTxFlags & tfSetNoRipple);
348 bool const bClearNoRipple = (uTxFlags & tfClearNoRipple);
349 bool const bSetFreeze = (uTxFlags & tfSetFreeze);
350 bool const bClearFreeze = (uTxFlags & tfClearFreeze);
351 bool const bSetDeepFreeze = (uTxFlags & tfSetDeepFreeze);
352 bool const bClearDeepFreeze = (uTxFlags & tfClearDeepFreeze);
353
354 auto viewJ = ctx_.registry.journal("View");
355
356 SLE::pointer sleDst = view().peek(keylet::account(uDstAccountID));
357
358 if (!sleDst)
359 {
360 JLOG(j_.trace()) << "Delay transaction: Destination account does not exist.";
361 return tecNO_DST;
362 }
363
364 STAmount saLimitAllow = saLimitAmount;
365 saLimitAllow.setIssuer(account_);
366
367 SLE::pointer sleRippleState = view().peek(keylet::line(account_, uDstAccountID, currency));
368
369 if (sleRippleState)
370 {
371 STAmount saLowBalance;
372 STAmount saLowLimit;
373 STAmount saHighBalance;
374 STAmount saHighLimit;
375 std::uint32_t uLowQualityIn;
376 std::uint32_t uLowQualityOut;
377 std::uint32_t uHighQualityIn;
378 std::uint32_t uHighQualityOut;
379 auto const& uLowAccountID = !bHigh ? account_ : uDstAccountID;
380 auto const& uHighAccountID = bHigh ? account_ : uDstAccountID;
381 SLE::ref sleLowAccount = !bHigh ? sle : sleDst;
382 SLE::ref sleHighAccount = bHigh ? sle : sleDst;
383
384 //
385 // Balances
386 //
387
388 saLowBalance = sleRippleState->getFieldAmount(sfBalance);
389 saHighBalance = -saLowBalance;
390
391 //
392 // Limits
393 //
394
395 sleRippleState->setFieldAmount(!bHigh ? sfLowLimit : sfHighLimit, saLimitAllow);
396
397 saLowLimit = !bHigh ? saLimitAllow : sleRippleState->getFieldAmount(sfLowLimit);
398 saHighLimit = bHigh ? saLimitAllow : sleRippleState->getFieldAmount(sfHighLimit);
399
400 //
401 // Quality in
402 //
403
404 if (!bQualityIn)
405 {
406 // Not setting. Just get it.
407
408 uLowQualityIn = sleRippleState->getFieldU32(sfLowQualityIn);
409 uHighQualityIn = sleRippleState->getFieldU32(sfHighQualityIn);
410 }
411 else if (uQualityIn)
412 {
413 // Setting.
414
415 sleRippleState->setFieldU32(!bHigh ? sfLowQualityIn : sfHighQualityIn, uQualityIn);
416
417 uLowQualityIn = !bHigh ? uQualityIn : sleRippleState->getFieldU32(sfLowQualityIn);
418 uHighQualityIn = bHigh ? uQualityIn : sleRippleState->getFieldU32(sfHighQualityIn);
419 }
420 else
421 {
422 // Clearing.
423
424 sleRippleState->makeFieldAbsent(!bHigh ? sfLowQualityIn : sfHighQualityIn);
425
426 uLowQualityIn = !bHigh ? 0 : sleRippleState->getFieldU32(sfLowQualityIn);
427 uHighQualityIn = bHigh ? 0 : sleRippleState->getFieldU32(sfHighQualityIn);
428 }
429
430 if (QUALITY_ONE == uLowQualityIn)
431 uLowQualityIn = 0;
432
433 if (QUALITY_ONE == uHighQualityIn)
434 uHighQualityIn = 0;
435
436 //
437 // Quality out
438 //
439
440 if (!bQualityOut)
441 {
442 // Not setting. Just get it.
443
444 uLowQualityOut = sleRippleState->getFieldU32(sfLowQualityOut);
445 uHighQualityOut = sleRippleState->getFieldU32(sfHighQualityOut);
446 }
447 else if (uQualityOut)
448 {
449 // Setting.
450
451 sleRippleState->setFieldU32(!bHigh ? sfLowQualityOut : sfHighQualityOut, uQualityOut);
452
453 uLowQualityOut = !bHigh ? uQualityOut : sleRippleState->getFieldU32(sfLowQualityOut);
454 uHighQualityOut = bHigh ? uQualityOut : sleRippleState->getFieldU32(sfHighQualityOut);
455 }
456 else
457 {
458 // Clearing.
459
460 sleRippleState->makeFieldAbsent(!bHigh ? sfLowQualityOut : sfHighQualityOut);
461
462 uLowQualityOut = !bHigh ? 0 : sleRippleState->getFieldU32(sfLowQualityOut);
463 uHighQualityOut = bHigh ? 0 : sleRippleState->getFieldU32(sfHighQualityOut);
464 }
465
466 std::uint32_t const uFlagsIn(sleRippleState->getFieldU32(sfFlags));
467 std::uint32_t uFlagsOut(uFlagsIn);
468
469 if (bSetNoRipple && !bClearNoRipple)
470 {
471 if ((bHigh ? saHighBalance : saLowBalance) >= beast::zero)
472 uFlagsOut |= (bHigh ? lsfHighNoRipple : lsfLowNoRipple);
473
474 else
475 // Cannot set noRipple on a negative balance.
476 return tecNO_PERMISSION;
477 }
478 else if (bClearNoRipple && !bSetNoRipple)
479 {
480 uFlagsOut &= ~(bHigh ? lsfHighNoRipple : lsfLowNoRipple);
481 }
482
483 // Have to use lsfNoFreeze to maintain pre-deep freeze behavior
484 bool const bNoFreeze = sle->isFlag(lsfNoFreeze);
485 uFlagsOut =
486 computeFreezeFlags(uFlagsOut, bHigh, bNoFreeze, bSetFreeze, bClearFreeze, bSetDeepFreeze, bClearDeepFreeze);
487
488 if (QUALITY_ONE == uLowQualityOut)
489 uLowQualityOut = 0;
490
491 if (QUALITY_ONE == uHighQualityOut)
492 uHighQualityOut = 0;
493
494 bool const bLowDefRipple = sleLowAccount->getFlags() & lsfDefaultRipple;
495 bool const bHighDefRipple = sleHighAccount->getFlags() & lsfDefaultRipple;
496
497 bool const bLowReserveSet = uLowQualityIn || uLowQualityOut ||
498 ((uFlagsOut & lsfLowNoRipple) == 0) != bLowDefRipple || (uFlagsOut & lsfLowFreeze) || saLowLimit ||
499 saLowBalance > beast::zero;
500 bool const bLowReserveClear = !bLowReserveSet;
501
502 bool const bHighReserveSet = uHighQualityIn || uHighQualityOut ||
503 ((uFlagsOut & lsfHighNoRipple) == 0) != bHighDefRipple || (uFlagsOut & lsfHighFreeze) || saHighLimit ||
504 saHighBalance > beast::zero;
505 bool const bHighReserveClear = !bHighReserveSet;
506
507 bool const bDefault = bLowReserveClear && bHighReserveClear;
508
509 bool const bLowReserved = (uFlagsIn & lsfLowReserve);
510 bool const bHighReserved = (uFlagsIn & lsfHighReserve);
511
512 bool bReserveIncrease = false;
513
514 if (bSetAuth)
515 {
516 uFlagsOut |= (bHigh ? lsfHighAuth : lsfLowAuth);
517 }
518
519 if (bLowReserveSet && !bLowReserved)
520 {
521 // Set reserve for low account.
522 adjustOwnerCount(view(), sleLowAccount, 1, viewJ);
523 uFlagsOut |= lsfLowReserve;
524
525 if (!bHigh)
526 bReserveIncrease = true;
527 }
528
529 if (bLowReserveClear && bLowReserved)
530 {
531 // Clear reserve for low account.
532 adjustOwnerCount(view(), sleLowAccount, -1, viewJ);
533 uFlagsOut &= ~lsfLowReserve;
534 }
535
536 if (bHighReserveSet && !bHighReserved)
537 {
538 // Set reserve for high account.
539 adjustOwnerCount(view(), sleHighAccount, 1, viewJ);
540 uFlagsOut |= lsfHighReserve;
541
542 if (bHigh)
543 bReserveIncrease = true;
544 }
545
546 if (bHighReserveClear && bHighReserved)
547 {
548 // Clear reserve for high account.
549 adjustOwnerCount(view(), sleHighAccount, -1, viewJ);
550 uFlagsOut &= ~lsfHighReserve;
551 }
552
553 if (uFlagsIn != uFlagsOut)
554 sleRippleState->setFieldU32(sfFlags, uFlagsOut);
555
556 if (bDefault || badCurrency() == currency)
557 {
558 // Delete.
559
560 terResult = trustDelete(view(), sleRippleState, uLowAccountID, uHighAccountID, viewJ);
561 }
562 // Reserve is not scaled by load.
563 else if (bReserveIncrease && mPriorBalance < reserveCreate)
564 {
565 JLOG(j_.trace()) << "Delay transaction: Insufficent reserve to "
566 "add trust line.";
567
568 // Another transaction could provide XRP to the account and then
569 // this transaction would succeed.
570 terResult = tecINSUF_RESERVE_LINE;
571 }
572 else
573 {
574 view().update(sleRippleState);
575
576 JLOG(j_.trace()) << "Modify ripple line";
577 }
578 }
579 // Line does not exist.
580 else if (
581 !saLimitAmount && // Setting default limit.
582 (!bQualityIn || !uQualityIn) && // Not setting quality in or
583 // setting default quality in.
584 (!bQualityOut || !uQualityOut) && // Not setting quality out or
585 // setting default quality out.
586 (!bSetAuth))
587 {
588 JLOG(j_.trace()) << "Redundant: Setting non-existent ripple line to defaults.";
589 return tecNO_LINE_REDUNDANT;
590 }
591 else if (mPriorBalance < reserveCreate) // Reserve is not scaled by
592 // load.
593 {
594 JLOG(j_.trace()) << "Delay transaction: Line does not exist. "
595 "Insufficent reserve to create line.";
596
597 // Another transaction could create the account and then this
598 // transaction would succeed.
599 terResult = tecNO_LINE_INSUF_RESERVE;
600 }
601 else
602 {
603 // Zero balance in currency.
604 STAmount saBalance(Issue{currency, noAccount()});
605
606 auto const k = keylet::line(account_, uDstAccountID, currency);
607
608 JLOG(j_.trace()) << "doTrustSet: Creating ripple line: " << to_string(k.key);
609
610 // Create a new ripple line.
611 terResult = trustCreate(
612 view(),
613 bHigh,
614 account_,
615 uDstAccountID,
616 k.key,
617 sle,
618 bSetAuth,
619 bSetNoRipple && !bClearNoRipple,
620 bSetFreeze && !bClearFreeze,
621 bSetDeepFreeze,
622 saBalance,
623 saLimitAllow, // Limit for who is being charged.
624 uQualityIn,
625 uQualityOut,
626 viewJ);
627 }
628
629 return terResult;
630}
