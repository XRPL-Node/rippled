100 //< Return nodes that fail the white-paper no-forking condition
101 std::vector<ForkInfo>
102 forkablePairs(double quorum) const
103 {
104 // Check the forking condition by looking at intersection
105 // of UNL between all pairs of nodes.
106
107 // TODO: Use the improved bound instead of the whitepaper bound.
108
109 using UNL = std::set<Peer>;
110 std::set<UNL> unique;
111 for (Peer const peer : graph_.outVertices())
112 {
113 unique.emplace(std::begin(trustedPeers(peer)), std::end(trustedPeers(peer)));
114 }
115
116 std::vector<UNL> uniqueUNLs(unique.begin(), unique.end());
117 std::vector<ForkInfo> res;
118
119 // Loop over all pairs of uniqueUNLs
120 for (int i = 0; i < uniqueUNLs.size(); ++i)
121 {
122 for (int j = (i + 1); j < uniqueUNLs.size(); ++j)
123 {
124 auto const& unlA = uniqueUNLs[i];
125 auto const& unlB = uniqueUNLs[j];
126 double rhs = 2.0 * (1. - quorum) * std::max(unlA.size(), unlB.size());
127
128 int intersectionSize =
129 std::count_if(unlA.begin(), unlA.end(), [&](Peer p) { return unlB.find(p) != unlB.end(); });
130
131 if (intersectionSize < rhs)
132 {
133 res.emplace_back(ForkInfo{unlA, unlB, intersectionSize, rhs});
134 }
135 }
136 }
137 return res;
138 }
