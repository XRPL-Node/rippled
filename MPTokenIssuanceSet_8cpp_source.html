23// Maps set/clear mutable flags in an MPTokenIssuanceSet transaction to the
24// corresponding ledger mutable flags that control whether the change is
25// allowed.
26struct MPTMutabilityFlags
27{
28 std::uint32_t setFlag;
29 std::uint32_t clearFlag;
30 std::uint32_t canMutateFlag;
31};
32
33static constexpr std::array<MPTMutabilityFlags, 6> mptMutabilityFlags = {
34 {{tmfMPTSetCanLock, tmfMPTClearCanLock, lsmfMPTCanMutateCanLock},
35 {tmfMPTSetRequireAuth, tmfMPTClearRequireAuth, lsmfMPTCanMutateRequireAuth},
36 {tmfMPTSetCanEscrow, tmfMPTClearCanEscrow, lsmfMPTCanMutateCanEscrow},
37 {tmfMPTSetCanTrade, tmfMPTClearCanTrade, lsmfMPTCanMutateCanTrade},
38 {tmfMPTSetCanTransfer, tmfMPTClearCanTransfer, lsmfMPTCanMutateCanTransfer},
39 {tmfMPTSetCanClawback, tmfMPTClearCanClawback, lsmfMPTCanMutateCanClawback}}};

41NotTEC
42MPTokenIssuanceSet::preflight(PreflightContext const& ctx)
43{
44 auto const mutableFlags = ctx.tx[~sfMutableFlags];
45 auto const metadata = ctx.tx[~sfMPTokenMetadata];
46 auto const transferFee = ctx.tx[~sfTransferFee];
47 auto const isMutate = mutableFlags || metadata || transferFee;
48
49 if (isMutate && !ctx.rules.enabled(featureDynamicMPT))
50 return temDISABLED;
51
52 if (ctx.tx.isFieldPresent(sfDomainID) && ctx.tx.isFieldPresent(sfHolder))
53 return temMALFORMED;
54
55 auto const txFlags = ctx.tx.getFlags();
56
57 // fails if both flags are set
58 if ((txFlags & tfMPTLock) && (txFlags & tfMPTUnlock))
59 return temINVALID_FLAG;
60
61 auto const accountID = ctx.tx[sfAccount];
62 auto const holderID = ctx.tx[~sfHolder];
63 if (holderID && accountID == holderID)
64 return temMALFORMED;
65
66 if (ctx.rules.enabled(featureSingleAssetVault) || ctx.rules.enabled(featureDynamicMPT))
67 {
68 // Is this transaction actually changing anything ?
69 if (txFlags == 0 && !ctx.tx.isFieldPresent(sfDomainID) && !isMutate)
70 return temMALFORMED;
71 }
72
73 if (ctx.rules.enabled(featureDynamicMPT))
74 {
75 // Holder field is not allowed when mutating MPTokenIssuance
76 if (isMutate && holderID)
77 return temMALFORMED;
78
79 // Can not set flags when mutating MPTokenIssuance
80 if (isMutate && (txFlags & tfUniversalMask))
81 return temMALFORMED;
82
83 if (transferFee && *transferFee > maxTransferFee)
84 return temBAD_TRANSFER_FEE;
85
86 if (metadata && metadata->length() > maxMPTokenMetadataLength)
87 return temMALFORMED;
88
89 if (mutableFlags)
90 {
91 if (!*mutableFlags || (*mutableFlags & tmfMPTokenIssuanceSetMutableMask))
92 return temINVALID_FLAG;
93
94 // Can not set and clear the same flag
95 if (std::any_of(
96 mptMutabilityFlags.begin(),
97 mptMutabilityFlags.end(),
98 [mutableFlags](auto const& f) {
99 return (*mutableFlags & f.setFlag) && (*mutableFlags & f.clearFlag);
100 }))
101 return temINVALID_FLAG;
102
103 // Trying to set a non-zero TransferFee and clear MPTCanTransfer
104 // in the same transaction is not allowed.
105 if (transferFee.value_or(0) && (*mutableFlags & tmfMPTClearCanTransfer))
106 return temMALFORMED;
107 }
108 }
109
110 return tesSUCCESS;
111}

113NotTEC
114MPTokenIssuanceSet::checkPermission(ReadView const& view, STTx const& tx)
115{
116 auto const delegate = tx[~sfDelegate];
117 if (!delegate)
118 return tesSUCCESS;
119
120 auto const delegateKey = keylet::delegate(tx[sfAccount], *delegate);
121 auto const sle = view.read(delegateKey);
122
123 if (!sle)
124 return terNO_DELEGATE_PERMISSION;
125
126 if (checkTxPermission(sle, tx) == tesSUCCESS)
127 return tesSUCCESS;
128
129 auto const txFlags = tx.getFlags();
130
131 // this is added in case more flags will be added for MPTokenIssuanceSet
132 // in the future. Currently unreachable.
133 if (txFlags & tfMPTokenIssuanceSetPermissionMask)
134 return terNO_DELEGATE_PERMISSION; // LCOV_EXCL_LINE
135
136 std::unordered_set<GranularPermissionType> granularPermissions;
137 loadGranularPermission(sle, ttMPTOKEN_ISSUANCE_SET, granularPermissions);
138
139 if (txFlags & tfMPTLock && !granularPermissions.contains(MPTokenIssuanceLock))
140 return terNO_DELEGATE_PERMISSION;
141
142 if (txFlags & tfMPTUnlock && !granularPermissions.contains(MPTokenIssuanceUnlock))
143 return terNO_DELEGATE_PERMISSION;
144
145 return tesSUCCESS;
146}

148TER
149MPTokenIssuanceSet::preclaim(PreclaimContext const& ctx)
150{
151 // ensure that issuance exists
152 auto const sleMptIssuance = ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
153 if (!sleMptIssuance)
154 return tecOBJECT_NOT_FOUND;
155
156 if (!sleMptIssuance->isFlag(lsfMPTCanLock))
157 {
158 // For readability two separate `if` rather than `||` of two conditions
159 if (!ctx.view.rules().enabled(featureSingleAssetVault) &&
160 !ctx.view.rules().enabled(featureDynamicMPT))
161 return tecNO_PERMISSION;
162 else if (ctx.tx.isFlag(tfMPTLock) || ctx.tx.isFlag(tfMPTUnlock))
163 return tecNO_PERMISSION;
164 }
165
166 // ensure it is issued by the tx submitter
167 if ((*sleMptIssuance)[sfIssuer] != ctx.tx[sfAccount])
168 return tecNO_PERMISSION;
169
170 if (auto const holderID = ctx.tx[~sfHolder])
171 {
172 // make sure holder account exists
173 if (!ctx.view.exists(keylet::account(*holderID)))
174 return tecNO_DST;
175
176 // the mptoken must exist
177 if (!ctx.view.exists(keylet::mptoken(ctx.tx[sfMPTokenIssuanceID], *holderID)))
178 return tecOBJECT_NOT_FOUND;
179 }
180
181 if (auto const domain = ctx.tx[~sfDomainID])
182 {
183 if (not sleMptIssuance->isFlag(lsfMPTRequireAuth))
184 return tecNO_PERMISSION;
185
186 if (*domain != beast::zero)
187 {
188 auto const sleDomain = ctx.view.read(keylet::permissionedDomain(*domain));
189 if (!sleDomain)
190 return tecOBJECT_NOT_FOUND;
191 }
192 }
193
194 // sfMutableFlags is soeDEFAULT, defaulting to 0 if not specified on
195 // the ledger.
196 auto const currentMutableFlags = sleMptIssuance->getFieldU32(sfMutableFlags);
197
198 auto isMutableFlag = [&](std::uint32_t mutableFlag) -> bool {
199 return currentMutableFlags & mutableFlag;
200 };
201
202 if (auto const mutableFlags = ctx.tx[~sfMutableFlags])
203 {
204 if (std::any_of(
205 mptMutabilityFlags.begin(),
206 mptMutabilityFlags.end(),
207 [mutableFlags, &isMutableFlag](auto const& f) {
208 return !isMutableFlag(f.canMutateFlag) &&
209 ((*mutableFlags & (f.setFlag | f.clearFlag)));
210 }))
211 return tecNO_PERMISSION;
212 }
213
214 if (!isMutableFlag(lsmfMPTCanMutateMetadata) && ctx.tx.isFieldPresent(sfMPTokenMetadata))
215 return tecNO_PERMISSION;
216
217 if (auto const fee = ctx.tx[~sfTransferFee])
218 {
219 // A non-zero TransferFee is only valid if the lsfMPTCanTransfer flag
220 // was previously enabled (at issuance or via a prior mutation). Setting
221 // it by tmfMPTSetCanTransfer in the current transaction does not meet
222 // this requirement.
223 if (fee > 0u && !sleMptIssuance->isFlag(lsfMPTCanTransfer))
224 return tecNO_PERMISSION;
225
226 if (!isMutableFlag(lsmfMPTCanMutateTransferFee))
227 return tecNO_PERMISSION;
228 }
229
230 return tesSUCCESS;
231}

233TER
234MPTokenIssuanceSet::doApply()
235{
236 auto const mptIssuanceID = ctx_.tx[sfMPTokenIssuanceID];
237 auto const txFlags = ctx_.tx.getFlags();
238 auto const holderID = ctx_.tx[~sfHolder];
239 auto const domainID = ctx_.tx[~sfDomainID];
240 std::shared_ptr<SLE> sle;
241
242 if (holderID)
243 sle = view().peek(keylet::mptoken(mptIssuanceID, *holderID));
244 else
245 sle = view().peek(keylet::mptIssuance(mptIssuanceID));
246
247 if (!sle)
248 return tecINTERNAL; // LCOV_EXCL_LINE
249
250 std::uint32_t const flagsIn = sle->getFieldU32(sfFlags);
251 std::uint32_t flagsOut = flagsIn;
252
253 if (txFlags & tfMPTLock)
254 flagsOut |= lsfMPTLocked;
255 else if (txFlags & tfMPTUnlock)
256 flagsOut &= ~lsfMPTLocked;
257
258 if (auto const mutableFlags = ctx_.tx[~sfMutableFlags].value_or(0))
259 {
260 for (auto const& f : mptMutabilityFlags)
261 {
262 if (mutableFlags & f.setFlag)
263 flagsOut |= f.canMutateFlag;
264 else if (mutableFlags & f.clearFlag)
265 flagsOut &= ~f.canMutateFlag;
266 }
267
268 if (mutableFlags & tmfMPTClearCanTransfer)
269 {
270 // If the lsfMPTCanTransfer flag is being cleared, then also clear
271 // the TransferFee field.
272 sle->makeFieldAbsent(sfTransferFee);
273 }
274 }
275
276 if (flagsIn != flagsOut)
277 sle->setFieldU32(sfFlags, flagsOut);
278
279 if (auto const transferFee = ctx_.tx[~sfTransferFee])
280 {
281 // TransferFee uses soeDEFAULT style:
282 // - If the field is absent, it is interpreted as 0.
283 // - If the field is present, it must be non-zero.
284 // Therefore, when TransferFee is 0, the field should be removed.
285 if (transferFee == 0)
286 sle->makeFieldAbsent(sfTransferFee);
287 else
288 sle->setFieldU16(sfTransferFee, *transferFee);
289 }
290
291 if (auto const metadata = ctx_.tx[~sfMPTokenMetadata])
292 {
293 if (metadata->empty())
294 sle->makeFieldAbsent(sfMPTokenMetadata);
295 else
296 sle->setFieldVL(sfMPTokenMetadata, *metadata);
297 }
298
299 if (domainID)
300 {
301 // This is enforced in preflight.
302 XRPL_ASSERT(
303 sle->getType() == ltMPTOKEN_ISSUANCE,
304 "MPTokenIssuanceSet::doApply : modifying MPTokenIssuance");
305
306 if (*domainID != beast::zero)
307 {
308 sle->setFieldH256(sfDomainID, *domainID);
309 }
310 else
311 {
312 if (sle->isFieldPresent(sfDomainID))
313 sle->makeFieldAbsent(sfDomainID);
314 }
315 }
316
317 view().update(sle);
318
319 return tesSUCCESS;
320}
