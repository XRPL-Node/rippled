22NotTEC
23LoanSet::preflight(PreflightContext const& ctx)
24{
25 using namespace Lending;
26
27 auto const& tx = ctx.tx;
28
29 // Special case for Batch inner transactions
30 if (tx.isFlag(tfInnerBatchTxn) && ctx.rules.enabled(featureBatch) &&
31 !tx.isFieldPresent(sfCounterparty))
32 {
33 auto const parentBatchId = ctx.parentBatchId.value_or(uint256{0});
34 JLOG(ctx.j.debug()) << "BatchTrace[" << parentBatchId << "]: "
35 << "no Counterparty for inner LoanSet transaction.";
36 return temBAD_SIGNER;
37 }
38
39 // These extra hoops are because STObjects cannot be Proxy'd from STObject.
40 auto const counterPartySig = [&tx]() -> std::optional<STObject const> {
41 if (tx.isFieldPresent(sfCounterpartySignature))
42 return tx.getFieldObject(sfCounterpartySignature);
43 return std::nullopt;
44 }();
45 if (!tx.isFlag(tfInnerBatchTxn) && !counterPartySig)
46 {
47 JLOG(ctx.j.warn())
48 << "LoanSet transaction must have a CounterpartySignature.";
49 return temBAD_SIGNER;
50 }
51
52 if (counterPartySig)
53 {
54 if (auto const ret =
55 xrpl::detail::preflightCheckSigningKey(*counterPartySig, ctx.j))
56 return ret;
57 }
58
59 if (auto const data = tx[~sfData]; data && !data->empty() &&
60 !validDataLength(tx[~sfData], maxDataPayloadLength))
61 return temINVALID;
62 for (auto const& field :
63 {&sfLoanServiceFee, &sfLatePaymentFee, &sfClosePaymentFee})
64 {
65 if (!validNumericMinimum(tx[~*field]))
66 return temINVALID;
67 }
68 // Principal Requested is required
69 if (auto const p = tx[sfPrincipalRequested]; p <= 0)
70 return temINVALID;
71 else if (!validNumericRange(tx[~sfLoanOriginationFee], p))
72 return temINVALID;
73 if (!validNumericRange(tx[~sfInterestRate], maxInterestRate))
74 return temINVALID;
75 if (!validNumericRange(tx[~sfOverpaymentFee], maxOverpaymentFee))
76 return temINVALID;
77 if (!validNumericRange(tx[~sfLateInterestRate], maxLateInterestRate))
78 return temINVALID;
79 if (!validNumericRange(tx[~sfCloseInterestRate], maxCloseInterestRate))
80 return temINVALID;
81 if (!validNumericRange(
82 tx[~sfOverpaymentInterestRate], maxOverpaymentInterestRate))
83 return temINVALID;
84
85 if (auto const paymentTotal = tx[~sfPaymentTotal];
86 paymentTotal && *paymentTotal <= 0)
87 return temINVALID;
88
89 if (auto const paymentInterval = tx[~sfPaymentInterval];
90 !validNumericMinimum(paymentInterval, LoanSet::minPaymentInterval))
91 return temINVALID;
92 // Grace period is between min default value and payment interval
93 else if (auto const gracePeriod = tx[~sfGracePeriod]; //
94 !validNumericRange(
95 gracePeriod,
96 paymentInterval.value_or(LoanSet::defaultPaymentInterval),
97 defaultGracePeriod))
98 return temINVALID;
99
100 // Copied from preflight2
101 if (counterPartySig)
102 {
103 if (auto const ret = xrpl::detail::preflightCheckSimulateKeys(
104 ctx.flags, *counterPartySig, ctx.j))
105 return *ret;
106 }
107
108 if (auto const brokerID = ctx.tx[~sfLoanBrokerID];
109 brokerID && *brokerID == beast::zero)
110 return temINVALID;
111
112 return tesSUCCESS;
113}

116LoanSet::checkSign(PreclaimContext const& ctx)
117{
118 if (auto ret = Transactor::checkSign(ctx))
119 return ret;
120
121 // Counter signer is optional. If it's not specified, it's assumed to be
122 // `LoanBroker.Owner`. Note that we have not checked whether the
123 // loanbroker exists at this point.
124 auto const counterSigner = [&]() -> std::optional<AccountID> {
125 if (auto const c = ctx.tx.at(~sfCounterparty))
126 return c;
127
128 if (auto const broker =
129 ctx.view.read(keylet::loanbroker(ctx.tx[sfLoanBrokerID])))
130 return broker->at(sfOwner);
131 return std::nullopt;
132 }();
133 if (!counterSigner)
134 return temBAD_SIGNER;
135
136 // Counterparty signature is optional. Presence is checked in preflight.
137 if (!ctx.tx.isFieldPresent(sfCounterpartySignature))
138 return tesSUCCESS;
139 auto const counterSig = ctx.tx.getFieldObject(sfCounterpartySignature);
140 return Transactor::checkSign(
141 ctx.view,
142 ctx.flags,
143 ctx.parentBatchId,
144 *counterSigner,
145 counterSig,
146 ctx.j);
147}

149XRPAmount
150LoanSet::calculateBaseFee(ReadView const& view, STTx const& tx)
151{
152 auto const normalCost = Transactor::calculateBaseFee(view, tx);
153
154 // Compute the additional cost of each signature in the
155 // CounterpartySignature, whether a single signature or a multisignature
156 XRPAmount const baseFee = view.fees().base;
157
158 // Counterparty signature is optional, but getFieldObject will return an
159 // empty object if it's not present.
160 auto const counterSig = tx.getFieldObject(sfCounterpartySignature);
161 // Each signer adds one more baseFee to the minimum required fee
162 // for the transaction. Note that unlike the base class, the single signer
163 // is counted if present. It will only be absent in a batch inner
164 // transaction.
165 std::size_t const signerCount = [&counterSig]() {
166 // Compute defensively. Assure that "tx" cannot be accessed and cause
167 // confusion or miscalculations.
168 return counterSig.isFieldPresent(sfSigners)
169 ? counterSig.getFieldArray(sfSigners).size()
170 : (counterSig.isFieldPresent(sfTxnSignature) ? 1 : 0);
171 }();
172
173 return normalCost + (signerCount * baseFee);
174}

197TER
198LoanSet::preclaim(PreclaimContext const& ctx)
199{
200 auto const& tx = ctx.tx;
201
202 {
203 // Check for numeric overflow of the schedule before we load any
204 // objects. The Grace Period for the last payment ends at:
205 // startDate + (paymentInterval * paymentTotal) + gracePeriod.
206 // If that value is larger than "maxTime", the value
207 // overflows, and we kill the transaction.
208 using timeType = decltype(sfNextPaymentDueDate)::type::value_type;
209 static_assert(std::is_same_v<timeType, std::uint32_t>);
210 timeType constexpr maxTime = std::numeric_limits<timeType>::max();
211 static_assert(maxTime == 4'294'967'295);
212
213 auto const timeAvailable = maxTime - getStartDate(ctx.view);
214
215 auto const interval =
216 ctx.tx.at(~sfPaymentInterval).value_or(defaultPaymentInterval);
217 auto const total =
218 ctx.tx.at(~sfPaymentTotal).value_or(defaultPaymentTotal);
219 auto const grace =
220 ctx.tx.at(~sfGracePeriod).value_or(defaultGracePeriod);
221
222 // The grace period can't be larger than the interval. Check it first,
223 // mostly so that unit tests can test that specific case.
224 if (grace > timeAvailable)
225 {
226 JLOG(ctx.j.warn()) << "Grace period exceeds protocol time limit.";
227 return tecKILLED;
228 }
229
230 if (interval > timeAvailable)
231 {
232 JLOG(ctx.j.warn())
233 << "Payment interval exceeds protocol time limit.";
234 return tecKILLED;
235 }
236
237 if (total > timeAvailable)
238 {
239 JLOG(ctx.j.warn()) << "Payment total exceeds protocol time limit.";
240 return tecKILLED;
241 }
242
243 auto const timeLastPayment = timeAvailable - grace;
244
245 if (timeLastPayment / interval < total)
246 {
247 JLOG(ctx.j.warn()) << "Last payment due date, or grace period for "
248 "last payment exceeds protocol time limit.";
249 return tecKILLED;
250 }
251 }
252
253 auto const account = tx[sfAccount];
254 auto const brokerID = tx[sfLoanBrokerID];
255
256 auto const brokerSle = ctx.view.read(keylet::loanbroker(brokerID));
257 if (!brokerSle)
258 {
259 // This can only be hit if there's a counterparty specified, otherwise
260 // it'll fail in the signature check
261 JLOG(ctx.j.warn()) << "LoanBroker does not exist.";
262 return tecNO_ENTRY;
263 }
264 auto const brokerOwner = brokerSle->at(sfOwner);
265 auto const counterparty = tx[~sfCounterparty].value_or(brokerOwner);
266 if (account != brokerOwner && counterparty != brokerOwner)
267 {
268 JLOG(ctx.j.warn()) << "Neither Account nor Counterparty are the owner "
269 "of the LoanBroker.";
270 return tecNO_PERMISSION;
271 }
272 auto const brokerPseudo = brokerSle->at(sfAccount);
273
274 auto const borrower = counterparty == brokerOwner ? account : counterparty;
275 if (auto const borrowerSle = ctx.view.read(keylet::account(borrower));
276 !borrowerSle)
277 {
278 // It may not be possible to hit this case, because it'll fail the
279 // signature check with terNO_ACCOUNT.
280 JLOG(ctx.j.warn()) << "Borrower does not exist.";
281 return terNO_ACCOUNT;
282 }
283
284 auto const vault = ctx.view.read(keylet::vault(brokerSle->at(sfVaultID)));
285 if (!vault)
286 // Should be impossible
287 return tefBAD_LEDGER; // LCOV_EXCL_LINE
288
289 if (vault->at(sfAssetsMaximum) != 0 &&
290 vault->at(sfAssetsTotal) >= vault->at(sfAssetsMaximum))
291 {
292 JLOG(ctx.j.warn())
293 << "Vault at maximum assets limit. Can't add another loan.";
294 return tecLIMIT_EXCEEDED;
295 }
296
297 Asset const asset = vault->at(sfAsset);
298
299 auto const vaultPseudo = vault->at(sfAccount);
300
301 // Check that relevant values can be represented as the vault asset type.
302 // This check is almost duplicated in doApply, but that check is done after
303 // the overall loan scale is known. This is mostly only relevant for
304 // integral (non-IOU) types
305 for (auto const& field : getValueFields())
306 {
307 if (auto const value = tx[field];
308 value && STAmount{asset, *value} != *value)
309 {
310 JLOG(ctx.j.warn()) << field.f->getName() << " (" << *value
311 << ") can not be represented as a(n) "
312 << to_string(asset) << ".";
313 return tecPRECISION_LOSS;
314 }
315 }
316
317 if (auto const ter = canAddHolding(ctx.view, asset))
318 return ter;
319
320 // vaultPseudo is going to send funds, so it can't be frozen.
321 if (auto const ret = checkFrozen(ctx.view, vaultPseudo, asset))
322 {
323 JLOG(ctx.j.warn()) << "Vault pseudo-account is frozen.";
324 return ret;
325 }
326
327 // brokerPseudo is the fallback account to receive LoanPay fees, even if the
328 // broker owner is unable to accept them. Don't create the loan if it is
329 // deep frozen.
330 if (auto const ret = checkDeepFrozen(ctx.view, brokerPseudo, asset))
331 {
332 JLOG(ctx.j.warn()) << "Broker pseudo-account is frozen.";
333 return ret;
334 }
335
336 // borrower is eventually going to have to pay back the loan, so it can't be
337 // frozen now. It is also going to receive funds, so it can't be deep
338 // frozen, but being frozen is a prerequisite for being deep frozen, so
339 // checking the one is sufficient.
340 if (auto const ret = checkFrozen(ctx.view, borrower, asset))
341 {
342 JLOG(ctx.j.warn()) << "Borrower account is frozen.";
343 return ret;
344 }
345 // brokerOwner is going to receive funds if there's an origination fee, so
346 // it can't be deep frozen
347 if (auto const ret = checkDeepFrozen(ctx.view, brokerOwner, asset))
348 {
349 JLOG(ctx.j.warn()) << "Broker owner account is frozen.";
350 return ret;
351 }
352
353 return tesSUCCESS;
354}

356TER
357LoanSet::doApply()
358{
359 auto const& tx = ctx_.tx;
360 auto& view = ctx_.view();
361
362 auto const brokerID = tx[sfLoanBrokerID];
363
364 auto const brokerSle = view.peek(keylet::loanbroker(brokerID));
365 if (!brokerSle)
366 return tefBAD_LEDGER; // LCOV_EXCL_LINE
367 auto const brokerOwner = brokerSle->at(sfOwner);
368 auto const brokerOwnerSle = view.peek(keylet::account(brokerOwner));
369 if (!brokerOwnerSle)
370 return tefBAD_LEDGER; // LCOV_EXCL_LINE
371
372 auto const vaultSle = view.peek(keylet ::vault(brokerSle->at(sfVaultID)));
373 if (!vaultSle)
374 return tefBAD_LEDGER; // LCOV_EXCL_LINE
375 auto const vaultPseudo = vaultSle->at(sfAccount);
376 Asset const vaultAsset = vaultSle->at(sfAsset);
377
378 auto const counterparty = tx[~sfCounterparty].value_or(brokerOwner);
379 auto const borrower = counterparty == brokerOwner ? account_ : counterparty;
380 auto const borrowerSle = view.peek(keylet::account(borrower));
381 if (!borrowerSle)
382 {
383 return tefBAD_LEDGER; // LCOV_EXCL_LINE
384 }
385
386 auto const brokerPseudo = brokerSle->at(sfAccount);
387 auto const brokerPseudoSle = view.peek(keylet::account(brokerPseudo));
388 if (!brokerPseudoSle)
389 {
390 return tefBAD_LEDGER; // LCOV_EXCL_LINE
391 }
392 auto const principalRequested = tx[sfPrincipalRequested];
393
394 auto vaultAvailableProxy = vaultSle->at(sfAssetsAvailable);
395 auto vaultTotalProxy = vaultSle->at(sfAssetsTotal);
396 auto const vaultScale = getAssetsTotalScale(vaultSle);
397 if (vaultAvailableProxy < principalRequested)
398 {
399 JLOG(j_.warn())
400 << "Insufficient assets available in the Vault to fund the loan.";
401 return tecINSUFFICIENT_FUNDS;
402 }
403
404 TenthBips32 const interestRate{tx[~sfInterestRate].value_or(0)};
405
406 auto const paymentInterval =
407 tx[~sfPaymentInterval].value_or(defaultPaymentInterval);
408 auto const paymentTotal = tx[~sfPaymentTotal].value_or(defaultPaymentTotal);
409
410 auto const properties = computeLoanProperties(
411 vaultAsset,
412 principalRequested,
413 interestRate,
414 paymentInterval,
415 paymentTotal,
416 TenthBips16{brokerSle->at(sfManagementFeeRate)},
417 vaultScale);
418
419 LoanState const state = constructLoanState(
420 properties.loanState.valueOutstanding,
421 principalRequested,
422 properties.loanState.managementFeeDue);
423
424 auto const vaultMaximum = *vaultSle->at(sfAssetsMaximum);
425 XRPL_ASSERT_PARTS(
426 vaultMaximum == 0 || vaultMaximum > *vaultTotalProxy,
427 "xrpl::LoanSet::doApply",
428 "Vault is below maximum limit");
429 if (vaultMaximum != 0 && state.interestDue > vaultMaximum - vaultTotalProxy)
430 {
431 JLOG(j_.warn()) << "Loan would exceed the maximum assets of the vault";
432 return tecLIMIT_EXCEEDED;
433 }
434 // Check that relevant values won't lose precision. This is mostly only
435 // relevant for IOU assets.
436 for (auto const& field : getValueFields())
437 {
438 if (auto const value = tx[field];
439 value && !isRounded(vaultAsset, *value, properties.loanScale))
440 {
441 JLOG(j_.warn()) << field.f->getName() << " (" << *value
442 << ") has too much precision. Total loan value is "
443 << properties.loanState.valueOutstanding
444 << " with a scale of " << properties.loanScale;
445 return tecPRECISION_LOSS;
446 }
447 }
448
449 if (auto const ret = checkLoanGuards(
450 vaultAsset,
451 principalRequested,
452 interestRate != beast::zero,
453 paymentTotal,
454 properties,
455 j_))
456 return ret;
457
458 // Check that the other computed values are valid
459 if (properties.loanState.managementFeeDue < 0 ||
460 properties.loanState.valueOutstanding <= 0 ||
461 properties.periodicPayment <= 0)
462 {
463 // LCOV_EXCL_START
464 JLOG(j_.warn())
465 << "Computed loan properties are invalid. Does not compute."
466 << " Management fee: " << properties.loanState.managementFeeDue
467 << ". Total Value: " << properties.loanState.valueOutstanding
468 << ". PeriodicPayment: " << properties.periodicPayment;
469 return tecINTERNAL;
470 // LCOV_EXCL_STOP
471 }
472
473 auto const originationFee = tx[~sfLoanOriginationFee].value_or(Number{});
474
475 auto const loanAssetsToBorrower = principalRequested - originationFee;
476
477 auto const newDebtDelta = principalRequested + state.interestDue;
478 auto const newDebtTotal = brokerSle->at(sfDebtTotal) + newDebtDelta;
479 if (auto const debtMaximum = brokerSle->at(sfDebtMaximum);
480 debtMaximum != 0 && debtMaximum < newDebtTotal)
481 {
482 JLOG(j_.warn())
483 << "Loan would exceed the maximum debt limit of the LoanBroker.";
484 return tecLIMIT_EXCEEDED;
485 }
486 TenthBips32 const coverRateMinimum{brokerSle->at(sfCoverRateMinimum)};
487 {
488 // Round the minimum required cover up to be conservative. This ensures
489 // CoverAvailable never drops below the theoretical minimum, protecting
490 // the broker's solvency.
491 NumberRoundModeGuard mg(Number::upward);
492 if (brokerSle->at(sfCoverAvailable) <
493 tenthBipsOfValue(newDebtTotal, coverRateMinimum))
494 {
495 JLOG(j_.warn())
496 << "Insufficient first-loss capital to cover the loan.";
497 return tecINSUFFICIENT_FUNDS;
498 }
499 }
500
501 adjustOwnerCount(view, borrowerSle, 1, j_);
502 {
503 auto const ownerCount = borrowerSle->at(sfOwnerCount);
504 auto const balance = account_ == borrower
505 ? mPriorBalance
506 : borrowerSle->at(sfBalance).value().xrp();
507 if (balance < view.fees().accountReserve(ownerCount))
508 return tecINSUFFICIENT_RESERVE;
509 }
510
511 // Account for the origination fee using two payments
512 //
513 // 1. Transfer loanAssetsAvailable (principalRequested - originationFee)
514 // from vault pseudo-account to the borrower.
515 // Create a holding for the borrower if one does not already exist.
516
517 XRPL_ASSERT_PARTS(
518 borrower == account_ || borrower == counterparty,
519 "xrpl::LoanSet::doApply",
520 "borrower signed transaction");
521 if (auto const ter = addEmptyHolding(
522 view,
523 borrower,
524 borrowerSle->at(sfBalance).value().xrp(),
525 vaultAsset,
526 j_);
527 ter && ter != tecDUPLICATE)
528 // ignore tecDUPLICATE. That means the holding already exists, and
529 // is fine here
530 return ter;
531
532 if (auto const ter =
533 requireAuth(view, vaultAsset, borrower, AuthType::StrongAuth))
534 return ter;
535
536 // 2. Transfer originationFee, if any, from vault pseudo-account to
537 // LoanBroker owner.
538 if (originationFee != beast::zero)
539 {
540 // Create the holding if it doesn't already exist (necessary for MPTs).
541 // The owner may have deleted their MPT / line at some point.
542 XRPL_ASSERT_PARTS(
543 brokerOwner == account_ || brokerOwner == counterparty,
544 "xrpl::LoanSet::doApply",
545 "broker owner signed transaction");
546
547 if (auto const ter = addEmptyHolding(
548 view,
549 brokerOwner,
550 brokerOwnerSle->at(sfBalance).value().xrp(),
551 vaultAsset,
552 j_);
553 ter && ter != tecDUPLICATE)
554 // ignore tecDUPLICATE. That means the holding already exists,
555 // and is fine here
556 return ter;
557 }
558
559 if (auto const ter =
560 requireAuth(view, vaultAsset, brokerOwner, AuthType::StrongAuth))
561 return ter;
562
563 if (auto const ter = accountSendMulti(
564 view,
565 vaultPseudo,
566 vaultAsset,
567 {{borrower, loanAssetsToBorrower}, {brokerOwner, originationFee}},
568 j_,
569 WaiveTransferFee::Yes))
570 return ter;
571
572 // Get shortcuts to the loan property values
573 auto const startDate = getStartDate(view);
574 auto loanSequenceProxy = brokerSle->at(sfLoanSequence);
575
576 // Create the loan
577 auto loan =
578 std::make_shared<SLE>(keylet::loan(brokerID, *loanSequenceProxy));
579
580 // Prevent copy/paste errors
581 auto setLoanField =
582 [&loan, &tx](auto const& field, std::uint32_t const defValue = 0) {
583 // at() is smart enough to unseat a default field set to the default
584 // value
585 loan->at(field) = tx[field].value_or(defValue);
586 };
587
588 // Set required and fixed tx fields
589 loan->at(sfLoanScale) = properties.loanScale;
590 loan->at(sfStartDate) = startDate;
591 loan->at(sfPaymentInterval) = paymentInterval;
592 loan->at(sfLoanSequence) = *loanSequenceProxy;
593 loan->at(sfLoanBrokerID) = brokerID;
594 loan->at(sfBorrower) = borrower;
595 // Set all other transaction fields directly from the transaction
596 if (tx.isFlag(tfLoanOverpayment))
597 loan->setFlag(lsfLoanOverpayment);
598 setLoanField(~sfLoanOriginationFee);
599 setLoanField(~sfLoanServiceFee);
600 setLoanField(~sfLatePaymentFee);
601 setLoanField(~sfClosePaymentFee);
602 setLoanField(~sfOverpaymentFee);
603 setLoanField(~sfInterestRate);
604 setLoanField(~sfLateInterestRate);
605 setLoanField(~sfCloseInterestRate);
606 setLoanField(~sfOverpaymentInterestRate);
607 setLoanField(~sfGracePeriod, defaultGracePeriod);
608 // Set dynamic / computed fields to their initial values
609 loan->at(sfPrincipalOutstanding) = principalRequested;
610 loan->at(sfPeriodicPayment) = properties.periodicPayment;
611 loan->at(sfTotalValueOutstanding) = properties.loanState.valueOutstanding;
612 loan->at(sfManagementFeeOutstanding) =
613 properties.loanState.managementFeeDue;
614 loan->at(sfPreviousPaymentDueDate) = 0;
615 loan->at(sfNextPaymentDueDate) = startDate + paymentInterval;
616 loan->at(sfPaymentRemaining) = paymentTotal;
617 view.insert(loan);
618
619 // Update the balances in the vault
620 vaultAvailableProxy -= principalRequested;
621 vaultTotalProxy += state.interestDue;
622 XRPL_ASSERT_PARTS(
623 *vaultAvailableProxy <= *vaultTotalProxy,
624 "xrpl::LoanSet::doApply",
625 "assets available must not be greater than assets outstanding");
626 view.update(vaultSle);
627
628 // Update the balances in the loan broker
629 adjustImpreciseNumber(
630 brokerSle->at(sfDebtTotal), newDebtDelta, vaultAsset, vaultScale);
631 // The broker's owner count is solely for the number of outstanding loans,
632 // and is distinct from the broker's pseudo-account's owner count
633 adjustOwnerCount(view, brokerSle, 1, j_);
634 loanSequenceProxy += 1;
635 // The sequence should be extremely unlikely to roll over, but fail if it
636 // does
637 if (loanSequenceProxy == 0)
638 return tecMAX_SEQUENCE_REACHED;
639 view.update(brokerSle);
640
641 // Put the loan into the pseudo-account's directory
642 if (auto const ter = dirLink(view, brokerPseudo, loan, sfLoanBrokerNode))
643 return ter;
644 // Borrower is the owner of the loan
645 if (auto const ter = dirLink(view, borrower, loan, sfOwnerNode))
646 return ter;
647
648 associateAsset(*vaultSle, vaultAsset);
649 associateAsset(*brokerSle, vaultAsset);
650 associateAsset(*loan, vaultAsset);
651
652 return tesSUCCESS;
653}
