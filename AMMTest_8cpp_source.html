97void
98AMMTestBase::testAMM(
99 std::function<void(jtx::AMM&, jtx::Env&)>&& cb,
100 TestAMMArg const& arg)
101{
102 using namespace jtx;
103
104 std::string logs;
105
106 for (auto const& features : arg.features)
107 {
108 // Use small Number mantissas for the life of this test.
109 NumberMantissaScaleGuard const sg{xrpl::MantissaRange::small};
110
111 // For now, just disable SAV entirely, which locks in the small Number
112 // mantissas
113 Env env{
114 *this,
115 features - featureSingleAssetVault - featureLendingProtocol,
116 arg.noLog ? std::make_unique<CaptureLogs>(&logs) : nullptr};
117
118 auto const [asset1, asset2] =
119 arg.pool ? *arg.pool : std::make_pair(XRP(10000), USD(10000));
120 auto toFund = [&](STAmount const& a) -> STAmount {
121 if (a.native())
122 {
123 auto const defXRP = XRP(30000);
124 if (a <= defXRP)
125 return defXRP;
126 return a + XRP(1000);
127 }
128 auto const defIOU = STAmount{a.issue(), 30000};
129 if (a <= defIOU)
130 return defIOU;
131 return a + STAmount{a.issue(), 1000};
132 };
133 auto const toFund1 = toFund(asset1);
134 auto const toFund2 = toFund(asset2);
135 BEAST_EXPECT(asset1 <= toFund1 && asset2 <= toFund2);
136
137 if (!asset1.native() && !asset2.native())
138 fund(env, gw, {alice, carol}, {toFund1, toFund2}, Fund::All);
139 else if (asset1.native())
140 fund(env, gw, {alice, carol}, toFund1, {toFund2}, Fund::All);
141 else if (asset2.native())
142 fund(env, gw, {alice, carol}, toFund2, {toFund1}, Fund::All);
143
144 AMM ammAlice(
145 env,
146 alice,
147 asset1,
148 asset2,
149 CreateArg{.log = false, .tfee = arg.tfee, .err = arg.ter});
150 if (BEAST_EXPECT(
151 ammAlice.expectBalances(asset1, asset2, ammAlice.tokens())))
152 cb(ammAlice, env);
153 }
154}

182Json::Value
183AMMTest::find_paths_request(
184 jtx::Env& env,
185 jtx::Account const& src,
186 jtx::Account const& dst,
187 STAmount const& saDstAmount,
188 std::optional<STAmount> const& saSendMax,
189 std::optional<Currency> const& saSrcCurrency)
190{
191 using namespace jtx;
192
193 auto& app = env.app();
194 Resource::Charge loadType = Resource::feeReferenceRPC;
195 Resource::Consumer c;
196
197 RPC::JsonContext context{
198 {env.journal,
199 app,
200 loadType,
201 app.getOPs(),
202 app.getLedgerMaster(),
203 c,
204 Role::USER,
205 {},
206 {},
207 RPC::apiVersionIfUnspecified},
208 {},
209 {}};
210
211 Json::Value params = Json::objectValue;
212 params[jss::command] = "ripple_path_find";
213 params[jss::source_account] = toBase58(src);
214 params[jss::destination_account] = toBase58(dst);
215 params[jss::destination_amount] = saDstAmount.getJson(JsonOptions::none);
216 if (saSendMax)
217 params[jss::send_max] = saSendMax->getJson(JsonOptions::none);
218 if (saSrcCurrency)
219 {
220 auto& sc = params[jss::source_currencies] = Json::arrayValue;
221 Json::Value j = Json::objectValue;
222 j[jss::currency] = to_string(saSrcCurrency.value());
223 sc.append(j);
224 }
225
226 Json::Value result;
227 gate g;
228 app.getJobQueue().postCoro(jtCLIENT, "RPC-Client", [&](auto const& coro) {
229 context.params = std::move(params);
230 context.coro = coro;
231 RPC::doCommand(context, result);
232 g.signal();
233 });
234
235 using namespace std::chrono_literals;
236 BEAST_EXPECT(g.wait_for(5s));
237 BEAST_EXPECT(!result.isMember(jss::error));
238 return result;
239}

241std::tuple<STPathSet, STAmount, STAmount>
242AMMTest::find_paths(
243 jtx::Env& env,
244 jtx::Account const& src,
245 jtx::Account const& dst,
246 STAmount const& saDstAmount,
247 std::optional<STAmount> const& saSendMax,
248 std::optional<Currency> const& saSrcCurrency)
249{
250 Json::Value result = find_paths_request(
251 env, src, dst, saDstAmount, saSendMax, saSrcCurrency);
252 BEAST_EXPECT(!result.isMember(jss::error));
253
254 STAmount da;
255 if (result.isMember(jss::destination_amount))
256 da = amountFromJson(sfGeneric, result[jss::destination_amount]);
257
258 STAmount sa;
259 STPathSet paths;
260 if (result.isMember(jss::alternatives))
261 {
262 auto const& alts = result[jss::alternatives];
263 if (alts.size() > 0)
264 {
265 auto const& path = alts[0u];
266
267 if (path.isMember(jss::source_amount))
268 sa = amountFromJson(sfGeneric, path[jss::source_amount]);
269
270 if (path.isMember(jss::destination_amount))
271 da = amountFromJson(sfGeneric, path[jss::destination_amount]);
272
273 if (path.isMember(jss::paths_computed))
274 {
275 Json::Value p;
276 p["Paths"] = path[jss::paths_computed];
277 STParsedJSONObject po("generic", p);
278 paths = po.object->getFieldPathSet(sfPaths);
279 }
280 }
281 }
282
283 return std::make_tuple(std::move(paths), std::move(sa), std::move(da));
284}
