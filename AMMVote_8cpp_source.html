55static std::pair<TER, bool>
56applyVote(ApplyContext& ctx_, Sandbox& sb, AccountID const& account_, beast::Journal j_)
57{
58 auto const feeNew = ctx_.tx[sfTradingFee];
59 auto ammSle = sb.peek(keylet::amm(ctx_.tx[sfAsset], ctx_.tx[sfAsset2]));
60 if (!ammSle)
61 return {tecINTERNAL, false};
62 STAmount const lptAMMBalance = (*ammSle)[sfLPTokenBalance];
63 auto const lpTokensNew = ammLPHolds(sb, *ammSle, account_, ctx_.journal);
64 std::optional<STAmount> minTokens;
65 std::size_t minPos{0};
66 AccountID minAccount{0};
67 std::uint32_t minFee{0};
68 STArray updatedVoteSlots;
69 Number num{0};
70 Number den{0};
71 // Account already has vote entry
72 bool foundAccount = false;
73
74 // Iterate over the current vote entries and update each entry
75 // per current total tokens balance and each LP tokens balance.
76 // Find the entry with the least tokens and whether the account
77 // has the vote entry.
78 for (auto const& entry : ammSle->getFieldArray(sfVoteSlots))
79 {
80 auto const account = entry[sfAccount];
81 auto lpTokens = ammLPHolds(sb, *ammSle, account, ctx_.journal);
82 if (lpTokens == beast::zero)
83 {
84 JLOG(j_.debug()) << "AMMVote::applyVote, account " << account << " is not LP";
85 continue;
86 }
87 auto feeVal = entry[sfTradingFee];
88 STObject newEntry = STObject::makeInnerObject(sfVoteEntry);
89 // The account already has the vote entry.
90 if (account == account_)
91 {
92 lpTokens = lpTokensNew;
93 feeVal = feeNew;
94 foundAccount = true;
95 }
96 // Keep running numerator/denominator to calculate the updated fee.
97 num += feeVal * lpTokens;
98 den += lpTokens;
99 newEntry.setAccountID(sfAccount, account);
100 if (feeVal != 0)
101 newEntry.setFieldU16(sfTradingFee, feeVal);
102 newEntry.setFieldU32(
103 sfVoteWeight,
104 static_cast<std::int64_t>(Number(lpTokens) * VOTE_WEIGHT_SCALE_FACTOR / lptAMMBalance));
105
106 // Find an entry with the least tokens/fee. Make the order deterministic
107 // if the tokens/fees are equal.
108 if (!minTokens ||
109 (lpTokens < *minTokens ||
110 (lpTokens == *minTokens &&
111 (feeVal < minFee || (feeVal == minFee && account < minAccount)))))
112 {
113 minTokens = lpTokens;
114 minPos = updatedVoteSlots.size();
115 minAccount = account;
116 minFee = feeVal;
117 }
118 updatedVoteSlots.push_back(std::move(newEntry));
119 }
120
121 // The account doesn't have the vote entry.
122 if (!foundAccount)
123 {
124 auto update = [&](std::optional<std::uint8_t> const& minPos = std::nullopt) {
125 STObject newEntry = STObject::makeInnerObject(sfVoteEntry);
126 if (feeNew != 0)
127 newEntry.setFieldU16(sfTradingFee, feeNew);
128 newEntry.setFieldU32(
129 sfVoteWeight,
130 static_cast<std::int64_t>(
131 Number(lpTokensNew) * VOTE_WEIGHT_SCALE_FACTOR / lptAMMBalance));
132 newEntry.setAccountID(sfAccount, account_);
133 num += feeNew * lpTokensNew;
134 den += lpTokensNew;
135 if (minPos)
136 *(updatedVoteSlots.begin() + *minPos) = std::move(newEntry);
137 else
138 updatedVoteSlots.push_back(std::move(newEntry));
139 };
140 // Add new entry if the number of the vote entries
141 // is less than Max.
142 if (updatedVoteSlots.size() < VOTE_MAX_SLOTS)
143 update();
144 // Add the entry if the account has more tokens than
145 // the least token holder or same tokens and higher fee.
146 else if (lpTokensNew > *minTokens || (lpTokensNew == *minTokens && feeNew > minFee))
147 {
148 auto const entry = updatedVoteSlots.begin() + minPos;
149 // Remove the least token vote entry.
150 num -= Number((*entry)[~sfTradingFee].value_or(0)) * *minTokens;
151 den -= *minTokens;
152 update(minPos);
153 }
154 // All slots are full and the account does not hold more LPTokens.
155 // Update anyway to refresh the slots.
156 else
157 {
158 JLOG(j_.debug()) << "AMMVote::applyVote, insufficient tokens to "
159 "override other votes";
160 }
161 }
162
163 XRPL_ASSERT(
164 !ctx_.view().rules().enabled(fixInnerObjTemplate) || ammSle->isFieldPresent(sfAuctionSlot),
165 "xrpl::applyVote : has auction slot");
166
167 // Update the vote entries and the trading/discounted fee.
168 ammSle->setFieldArray(sfVoteSlots, updatedVoteSlots);
169 if (auto const fee = static_cast<std::int64_t>(num / den))
170 {
171 ammSle->setFieldU16(sfTradingFee, fee);
172 if (ammSle->isFieldPresent(sfAuctionSlot))
173 {
174 auto& auctionSlot = ammSle->peekFieldObject(sfAuctionSlot);
175 if (auto const discountedFee = fee / AUCTION_SLOT_DISCOUNTED_FEE_FRACTION)
176 auctionSlot.setFieldU16(sfDiscountedFee, discountedFee);
177 else if (auctionSlot.isFieldPresent(sfDiscountedFee))
178 auctionSlot.makeFieldAbsent(sfDiscountedFee);
179 }
180 }
181 else
182 {
183 if (ammSle->isFieldPresent(sfTradingFee))
184 ammSle->makeFieldAbsent(sfTradingFee);
185 if (ammSle->isFieldPresent(sfAuctionSlot))
186 {
187 auto& auctionSlot = ammSle->peekFieldObject(sfAuctionSlot);
188 if (auctionSlot.isFieldPresent(sfDiscountedFee))
189 auctionSlot.makeFieldAbsent(sfDiscountedFee);
190 }
191 }
192 sb.update(ammSle);
193
194 return {tesSUCCESS, true};
195}
