62std::optional<Blob>
63ConsensusTransSetSF::getNode(SHAMapHash const& nodeHash) const
64{
65 Blob nodeData;
66 if (m_nodeCache.retrieve(nodeHash, nodeData))
67 return nodeData;
68
69 auto txn =
70 app_.getMasterTransaction().fetch_from_cache(nodeHash.as_uint256());
71
72 if (txn)
73 {
74 // this is a transaction, and we have it
75 JLOG(j_.trace()) << "Node in our acquiring TX set is TXN we have";
76 Serializer s;
77 s.add32(HashPrefix::transactionID);
78 txn->getSTransaction()->add(s);
79 XRPL_ASSERT(
80 sha512Half(s.slice()) == nodeHash.as_uint256(),
81 "xrpl::ConsensusTransSetSF::getNode : transaction hash match");
82 nodeData = s.peekData();
83 return nodeData;
84 }
