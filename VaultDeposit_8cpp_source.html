32TER
33VaultDeposit::preclaim(PreclaimContext const& ctx)
34{
35 auto const vault = ctx.view.read(keylet::vault(ctx.tx[sfVaultID]));
36 if (!vault)
37 return tecNO_ENTRY;
38
39 auto const& account = ctx.tx[sfAccount];
40 auto const assets = ctx.tx[sfAmount];
41 auto const vaultAsset = vault->at(sfAsset);
42 if (assets.asset() != vaultAsset)
43 return tecWRONG_ASSET;
44
45 auto const& vaultAccount = vault->at(sfAccount);
46 if (auto ter = canTransfer(ctx.view, vaultAsset, account, vaultAccount); !isTesSuccess(ter))
47 {
48 JLOG(ctx.j.debug()) << "VaultDeposit: vault assets are non-transferable.";
49 return ter;
50 }
51
52 auto const mptIssuanceID = vault->at(sfShareMPTID);
53 auto const vaultShare = MPTIssue(mptIssuanceID);
54 if (vaultShare == assets.asset())
55 {
56 // LCOV_EXCL_START
57 JLOG(ctx.j.error()) << "VaultDeposit: vault shares and assets cannot be same.";
58 return tefINTERNAL;
59 // LCOV_EXCL_STOP
60 }
61
62 auto const sleIssuance = ctx.view.read(keylet::mptIssuance(mptIssuanceID));
63 if (!sleIssuance)
64 {
65 // LCOV_EXCL_START
66 JLOG(ctx.j.error()) << "VaultDeposit: missing issuance of vault shares.";
67 return tefINTERNAL;
68 // LCOV_EXCL_STOP
69 }
70
71 if (sleIssuance->isFlag(lsfMPTLocked))
72 {
73 // LCOV_EXCL_START
74 JLOG(ctx.j.error()) << "VaultDeposit: issuance of vault shares is locked.";
75 return tefINTERNAL;
76 // LCOV_EXCL_STOP
77 }
78
79 // Cannot deposit inside Vault an Asset frozen for the depositor
80 if (isFrozen(ctx.view, account, vaultAsset))
81 return vaultAsset.holds<Issue>() ? tecFROZEN : tecLOCKED;
82
83 // Cannot deposit if the shares of the vault are frozen
84 if (isFrozen(ctx.view, account, vaultShare))
85 return tecLOCKED;
86
87 if (vault->isFlag(lsfVaultPrivate) && account != vault->at(sfOwner))
88 {
89 auto const maybeDomainID = sleIssuance->at(~sfDomainID);
90 // Since this is a private vault and the account is not its owner, we
91 // perform authorization check based on DomainID read from sleIssuance.
92 // Had the vault shares been a regular MPToken, we would allow
93 // authorization granted by the Issuer explicitly, but Vault uses Issuer
94 // pseudo-account, which cannot grant an authorization.
95 if (maybeDomainID)
96 {
97 // As per validDomain documentation, we suppress tecEXPIRED error
98 // here, so we can delete any expired credentials inside doApply.
99 if (auto const err = credentials::validDomain(ctx.view, *maybeDomainID, account);
100 !isTesSuccess(err) && err != tecEXPIRED)
101 return err;
102 }
103 else
104 return tecNO_AUTH;
105 }
106
107 // Source MPToken must exist (if asset is an MPT)
108 if (auto const ter = requireAuth(ctx.view, vaultAsset, account); !isTesSuccess(ter))
109 return ter;
110
111 if (accountHolds(
112 ctx.view,
113 account,
114 vaultAsset,
115 FreezeHandling::fhZERO_IF_FROZEN,
116 AuthHandling::ahZERO_IF_UNAUTHORIZED,
117 ctx.j,
118 SpendableHandling::shFULL_BALANCE) < assets)
119 return tecINSUFFICIENT_FUNDS;
120
121 return tesSUCCESS;
122}

124TER
125VaultDeposit::doApply()
126{
127 auto const vault = view().peek(keylet::vault(ctx_.tx[sfVaultID]));
128 if (!vault)
129 return tefINTERNAL; // LCOV_EXCL_LINE
130 auto const vaultAsset = vault->at(sfAsset);
131
132 auto const amount = ctx_.tx[sfAmount];
133 // Make sure the depositor can hold shares.
134 auto const mptIssuanceID = (*vault)[sfShareMPTID];
135 auto const sleIssuance = view().read(keylet::mptIssuance(mptIssuanceID));
136 if (!sleIssuance)
137 {
138 // LCOV_EXCL_START
139 JLOG(j_.error()) << "VaultDeposit: missing issuance of vault shares.";
140 return tefINTERNAL;
141 // LCOV_EXCL_STOP
142 }
143
144 auto const& vaultAccount = vault->at(sfAccount);
145 // Note, vault owner is always authorized
146 if (vault->isFlag(lsfVaultPrivate) && account_ != vault->at(sfOwner))
147 {
148 if (auto const err = enforceMPTokenAuthorization(
149 ctx_.view(), mptIssuanceID, account_, mPriorBalance, j_);
150 !isTesSuccess(err))
151 return err;
152 }
153 else // !vault->isFlag(lsfVaultPrivate) || account_ == vault->at(sfOwner)
154 {
155 // No authorization needed, but must ensure there is MPToken
156 if (!view().exists(keylet::mptoken(mptIssuanceID, account_)))
157 {
158 if (auto const err = authorizeMPToken(
159 view(), mPriorBalance, mptIssuanceID->value(), account_, ctx_.journal);
160 !isTesSuccess(err))
161 return err;
162 }
163
164 // If the vault is private, set the authorized flag for the vault owner
165 if (vault->isFlag(lsfVaultPrivate))
166 {
167 // This follows from the reverse of the outer enclosing if condition
168 XRPL_ASSERT(
169 account_ == vault->at(sfOwner), "xrpl::VaultDeposit::doApply : account is owner");
170 if (auto const err = authorizeMPToken(
171 view(),
172 mPriorBalance, // priorBalance
173 mptIssuanceID->value(), // mptIssuanceID
174 sleIssuance->at(sfIssuer), // account
175 ctx_.journal,
176 {}, // flags
177 account_ // holderID
178 );
179 !isTesSuccess(err))
180 return err;
181 }
182 }
183
184 STAmount sharesCreated = {vault->at(sfShareMPTID)}, assetsDeposited;
185 try
186 {
187 // Compute exchange before transferring any amounts.
188 {
189 auto const maybeShares = assetsToSharesDeposit(vault, sleIssuance, amount);
190 if (!maybeShares)
191 return tecINTERNAL; // LCOV_EXCL_LINE
192 sharesCreated = *maybeShares;
193 }
194 if (sharesCreated == beast::zero)
195 return tecPRECISION_LOSS;
196
197 auto const maybeAssets = sharesToAssetsDeposit(vault, sleIssuance, sharesCreated);
198 if (!maybeAssets)
199 return tecINTERNAL; // LCOV_EXCL_LINE
200 else if (*maybeAssets > amount)
201 {
202 // LCOV_EXCL_START
203 JLOG(j_.error()) << "VaultDeposit: would take more than offered.";
204 return tecINTERNAL;
205 // LCOV_EXCL_STOP
206 }
207 assetsDeposited = *maybeAssets;
208 }
209 catch (std::overflow_error const&)
210 {
211 // It's easy to hit this exception from Number with large enough Scale
212 // so we avoid spamming the log and only use debug here.
213 JLOG(j_.debug()) //
214 << "VaultDeposit: overflow error with"
215 << " scale=" << (int)vault->at(sfScale).value() //
216 << ", assetsTotal=" << vault->at(sfAssetsTotal).value()
217 << ", sharesTotal=" << sleIssuance->at(sfOutstandingAmount) << ", amount=" << amount;
218 return tecPATH_DRY;
219 }
220
221 XRPL_ASSERT(
222 sharesCreated.asset() != assetsDeposited.asset(),
223 "xrpl::VaultDeposit::doApply : assets are not shares");
224
225 vault->at(sfAssetsTotal) += assetsDeposited;
226 vault->at(sfAssetsAvailable) += assetsDeposited;
227 view().update(vault);
228
229 // A deposit must not push the vault over its limit.
230 auto const maximum = *vault->at(sfAssetsMaximum);
231 if (maximum != 0 && *vault->at(sfAssetsTotal) > maximum)
232 return tecLIMIT_EXCEEDED;
233
234 // Transfer assets from depositor to vault.
235 if (auto const ter =
236 accountSend(view(), account_, vaultAccount, assetsDeposited, j_, WaiveTransferFee::Yes);
237 !isTesSuccess(ter))
238 return ter;
239
240 // Sanity check
241 if (accountHolds(
242 view(),
243 account_,
244 assetsDeposited.asset(),
245 FreezeHandling::fhIGNORE_FREEZE,
246 AuthHandling::ahIGNORE_AUTH,
247 j_) < beast::zero)
248 {
249 // LCOV_EXCL_START
250 JLOG(j_.error()) << "VaultDeposit: negative balance of account assets.";
251 return tefINTERNAL;
252 // LCOV_EXCL_STOP
253 }
254
255 // Transfer shares from vault to depositor.
256 if (auto const ter =
257 accountSend(view(), vaultAccount, account_, sharesCreated, j_, WaiveTransferFee::Yes);
258 !isTesSuccess(ter))
259 return ter;
260
261 associateAsset(*vault, vaultAsset);
262
263 return tesSUCCESS;
264}
