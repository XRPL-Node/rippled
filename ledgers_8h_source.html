309struct LedgerHistoryHelper
310{
311 LedgerOracle oracle;
312 Tx::ID nextTx{0};
313 std::unordered_map<std::string, Ledger> ledgers;
314 std::set<char> seen;
315
316 LedgerHistoryHelper()
317 {
318 ledgers[""] = Ledger{Ledger::MakeGenesis{}};
319 }
320
326 Ledger const&
327 operator[](std::string const& s)
328 {
329 auto it = ledgers.find(s);
330 if (it != ledgers.end())
331 return it->second;
332
333 // enforce that the new suffix has never been seen
334 assert(seen.emplace(s.back()).second);
335
336 Ledger const& parent = (*this)[s.substr(0, s.size() - 1)];
337 return ledgers.emplace(s, oracle.accept(parent, Tx{++nextTx}))
338 .first->second;
339 }
340};
