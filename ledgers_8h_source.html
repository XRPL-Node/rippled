297struct LedgerHistoryHelper
298{
299 LedgerOracle oracle;
300 Tx::ID nextTx{0};
301 std::unordered_map<std::string, Ledger> ledgers;
302 std::set<char> seen;
303
304 LedgerHistoryHelper()
305 {
306 ledgers[""] = Ledger{Ledger::MakeGenesis{}};
307 }
308
314 Ledger const&
315 operator[](std::string const& s)
316 {
317 auto it = ledgers.find(s);
318 if (it != ledgers.end())
319 return it->second;
320
321 // enforce that the new suffix has never been seen
322 assert(seen.emplace(s.back()).second);
323
324 Ledger const& parent = (*this)[s.substr(0, s.size() - 1)];
325 return ledgers.emplace(s, oracle.accept(parent, Tx{++nextTx})).first->second;
326 }
