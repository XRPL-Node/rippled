304struct LedgerHistoryHelper
305{
306 LedgerOracle oracle;
307 Tx::ID nextTx{0};
308 std::unordered_map<std::string, Ledger> ledgers;
309 std::set<char> seen;
310
311 LedgerHistoryHelper()
312 {
313 ledgers[""] = Ledger{Ledger::MakeGenesis{}};
314 }
315
321 Ledger const&
322 operator[](std::string const& s)
323 {
324 auto it = ledgers.find(s);
325 if (it != ledgers.end())
326 return it->second;
327
328 // enforce that the new suffix has never been seen
329 assert(seen.emplace(s.back()).second);
330
331 Ledger const& parent = (*this)[s.substr(0, s.size() - 1)];
332 return ledgers.emplace(s, oracle.accept(parent, Tx{++nextTx})).first->second;
333 }
