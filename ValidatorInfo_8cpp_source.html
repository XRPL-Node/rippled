13Json::Value
14doValidatorInfo(RPC::JsonContext& context)
15{
16 // return error if not configured as validator
17 auto const validationPK = context.app.getValidationPublicKey();
18 if (!validationPK)
19 return RPC::not_validator_error();
20
21 Json::Value ret;
22
23 // assume validationPK is ephemeral key, get master key
24 auto const mk = context.app.validatorManifests().getMasterKey(*validationPK);
25 ret[jss::master_key] = toBase58(TokenType::NodePublic, mk);
26
27 // validationPK is master key, this implies that there is no ephemeral
28 // key, no manifest, hence return
29 if (mk == validationPK)
30 return ret;
31
32 ret[jss::ephemeral_key] = toBase58(TokenType::NodePublic, *validationPK);
33
34 if (auto const manifest = context.app.validatorManifests().getManifest(mk))
35 ret[jss::manifest] = base64_encode(*manifest);
36
37 if (auto const seq = context.app.validatorManifests().getSequence(mk))
38 ret[jss::seq] = *seq;
39
40 if (auto const domain = context.app.validatorManifests().getDomain(mk))
41 ret[jss::domain] = *domain;
42
43 return ret;
44}
