363template <class T>
364NotTEC
365Transactor::invokePreflight(PreflightContext const& ctx)
366{
367 // Using this lookup does NOT require checking the fixDelegateV1_1. The data
368 // exists regardless of whether it is enabled.
369 auto const feature = Permission::getInstance().getTxFeature(ctx.tx.getTxnType());
370
371 if (feature && !ctx.rules.enabled(*feature))
372 return temDISABLED;
373
374 if (!T::checkExtraFeatures(ctx))
375 return temDISABLED;
376
377 if (auto const ret = preflight1(ctx, T::getFlagsMask(ctx)))
378 return ret;
379
380 if (auto const ret = T::preflight(ctx))
381 return ret;
382
383 if (auto const ret = preflight2(ctx))
384 return ret;
385
386 return T::preflightSigValidated(ctx);
387}
