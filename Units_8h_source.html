373template <class Source1, class Source2, unit::muldivable<Source1, Source2> Dest>
374std::optional<Dest>
375mulDivU(Source1 value, Dest mul, Source2 div)
376{
377 // values can never be negative in any context.
378 if (value.value() < 0 || mul.value() < 0 || div.value() < 0)
379 {
380 // split the asserts so if one hits, the user can tell which
381 // without a debugger.
382 XRPL_ASSERT(value.value() >= 0, "xrpl::unit::mulDivU : minimum value input");
383 XRPL_ASSERT(mul.value() >= 0, "xrpl::unit::mulDivU : minimum mul input");
384 XRPL_ASSERT(div.value() > 0, "xrpl::unit::mulDivU : minimum div input");
385 return std::nullopt;
386 }
387
388 using desttype = typename Dest::value_type;
389 constexpr auto max = std::numeric_limits<desttype>::max();
390
391 // Shortcuts, since these happen a lot in the real world
392 if (value == div)
393 return mul;
394 if (mul.value() == div.value())
395 {
396 if (value.value() > max)
397 return std::nullopt;
398 return Dest{static_cast<desttype>(value.value())};
399 }
400
401 using namespace boost::multiprecision;
402
403 uint128_t product;
404 product = multiply(product, static_cast<std::uint64_t>(value.value()), static_cast<std::uint64_t>(mul.value()));
405
406 auto quotient = product / div.value();
407
408 if (quotient > max)
409 return std::nullopt;
410
411 return Dest{static_cast<desttype>(quotient)};
412}
