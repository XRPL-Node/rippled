176// be deleted. Otherwise return the pointer to the function that can delete

197 case ltCREDENTIAL:
198 return removeCredentialFromLedger;
199 case ltDELEGATE:
200 return removeDelegateFromLedger;
201 default:
202 return nullptr;

208TER
209DeleteAccount::preclaim(PreclaimContext const& ctx)
210{
211 AccountID const account{ctx.tx[sfAccount]};
212 AccountID const dst{ctx.tx[sfDestination]};
213
214 auto sleDst = ctx.view.read(keylet::account(dst));
215
216 if (!sleDst)
217 return tecNO_DST;
218
219 if ((*sleDst)[sfFlags] & lsfRequireDestTag && !ctx.tx[~sfDestinationTag])
220 return tecDST_TAG_NEEDED;
221
222 // If credentials are provided - check them anyway
223 if (auto const err = credentials::valid(ctx.tx, ctx.view, account, ctx.j);
224 !isTesSuccess(err))
225 return err;
226
227 // if credentials then postpone auth check to doApply, to check for expired
228 // credentials
229 if (!ctx.tx.isFieldPresent(sfCredentialIDs))
230 {
231 // Check whether the destination account requires deposit authorization.
232 if (sleDst->getFlags() & lsfDepositAuth)
233 {
234 if (!ctx.view.exists(keylet::depositPreauth(dst, account)))
235 return tecNO_PERMISSION;
236 }
237 }
238
239 auto sleAccount = ctx.view.read(keylet::account(account));
240 XRPL_ASSERT(sleAccount, "xrpl::DeleteAccount::preclaim : non-null account");
241 if (!sleAccount)
242 return terNO_ACCOUNT;
243
244 // If an issuer has any issued NFTs resident in the ledger then it
245 // cannot be deleted.
246 if ((*sleAccount)[~sfMintedNFTokens] != (*sleAccount)[~sfBurnedNFTokens])
247 return tecHAS_OBLIGATIONS;
248
249 // If the account owns any NFTs it cannot be deleted.
250 Keylet const first = keylet::nftpage_min(account);
251 Keylet const last = keylet::nftpage_max(account);
252
253 auto const cp = ctx.view.read(Keylet(
254 ltNFTOKEN_PAGE,
255 ctx.view.succ(first.key, last.key.next()).value_or(last.key)));
256 if (cp)
257 return tecHAS_OBLIGATIONS;
258
259 // We don't allow an account to be deleted if its sequence number
260 // is within 256 of the current ledger. This prevents replay of old
261 // transactions if this account is resurrected after it is deleted.
262 //
263 // We look at the account's Sequence rather than the transaction's
264 // Sequence in preparation for Tickets.
265 constexpr std::uint32_t seqDelta{255};
266 if ((*sleAccount)[sfSequence] + seqDelta > ctx.view.seq())
267 return tecTOO_SOON;
268
269 // We don't allow an account to be deleted if
270 // <FirstNFTokenSequence + MintedNFTokens> is within 256 of the
271 // current ledger. This is to prevent having duplicate NFTokenIDs after
272 // account re-creation.
273 //
274 // Without this restriction, duplicate NFTokenIDs can be reproduced when
275 // authorized minting is involved. Because when the minter mints a NFToken,
276 // the issuer's sequence does not change. So when the issuer re-creates
277 // their account and mints a NFToken, it is possible that the
278 // NFTokenSequence of this NFToken is the same as the one that the
279 // authorized minter minted in a previous ledger.
280 if ((*sleAccount)[~sfFirstNFTokenSequence].value_or(0) +
281 (*sleAccount)[~sfMintedNFTokens].value_or(0) + seqDelta >
282 ctx.view.seq())
283 return tecTOO_SOON;
284
285 // Verify that the account does not own any objects that would prevent
286 // the account from being deleted.
287 Keylet const ownerDirKeylet{keylet::ownerDir(account)};
288 if (dirIsEmpty(ctx.view, ownerDirKeylet))
289 return tesSUCCESS;
290
291 std::shared_ptr<SLE const> sleDirNode{};
292 unsigned int uDirEntry{0};
293 uint256 dirEntry{beast::zero};
294
295 // Account has no directory at all. This _should_ have been caught
296 // by the dirIsEmpty() check earlier, but it's okay to catch it here.
297 if (!cdirFirst(
298 ctx.view, ownerDirKeylet.key, sleDirNode, uDirEntry, dirEntry))
299 return tesSUCCESS;
300
301 std::int32_t deletableDirEntryCount{0};
302 do
303 {
304 // Make sure any directory node types that we find are the kind
305 // we can delete.
306 auto sleItem = ctx.view.read(keylet::child(dirEntry));
307 if (!sleItem)
308 {
309 // Directory node has an invalid index. Bail out.
310 // LCOV_EXCL_START
311 JLOG(ctx.j.fatal())
312 << "DeleteAccount: directory node in ledger " << ctx.view.seq()
313 << " has index to object that is missing: "
314 << to_string(dirEntry);
315 return tefBAD_LEDGER;
316 // LCOV_EXCL_STOP
317 }
318
319 LedgerEntryType const nodeType{
320 safe_cast<LedgerEntryType>((*sleItem)[sfLedgerEntryType])};
321
322 if (!nonObligationDeleter(nodeType))
323 return tecHAS_OBLIGATIONS;
324
325 // We found a deletable directory entry. Count it. If we find too
326 // many deletable directory entries then bail out.
327 if (++deletableDirEntryCount > maxDeletableDirEntries)
328 return tefTOO_BIG;
329
330 } while (cdirNext(
331 ctx.view, ownerDirKeylet.key, sleDirNode, uDirEntry, dirEntry));
332
333 return tesSUCCESS;
334}

336TER
337DeleteAccount::doApply()
338{
339 auto src = view().peek(keylet::account(account_));
340 XRPL_ASSERT(src, "xrpl::DeleteAccount::doApply : non-null source account");
341
342 auto const dstID = ctx_.tx[sfDestination];
343 auto dst = view().peek(keylet::account(dstID));
344 XRPL_ASSERT(
345 dst, "xrpl::DeleteAccount::doApply : non-null destination account");
346
347 if (!src || !dst)
348 return tefBAD_LEDGER; // LCOV_EXCL_LINE
349
350 if (ctx_.tx.isFieldPresent(sfCredentialIDs))
351 {
352 if (auto err = verifyDepositPreauth(
353 ctx_.tx, ctx_.view(), account_, dstID, dst, ctx_.journal);
354 !isTesSuccess(err))
355 return err;
356 }
357
358 Keylet const ownerDirKeylet{keylet::ownerDir(account_)};
359 auto const ter = cleanupOnAccountDelete(
360 view(),
361 ownerDirKeylet,
362 [&](LedgerEntryType nodeType,
363 uint256 const& dirEntry,
364 std::shared_ptr<SLE>& sleItem) -> std::pair<TER, SkipEntry> {
365 if (auto deleter = nonObligationDeleter(nodeType))
366 {
367 TER const result{
368 deleter(ctx_.app, view(), account_, dirEntry, sleItem, j_)};
369
370 return {result, SkipEntry::No};
371 }
372
373 // LCOV_EXCL_START
374 UNREACHABLE(
375 "xrpl::DeleteAccount::doApply : undeletable item not found "
376 "in preclaim");
377 JLOG(j_.error()) << "DeleteAccount undeletable item not "
378 "found in preclaim.";
379 return {tecHAS_OBLIGATIONS, SkipEntry::No};
380 // LCOV_EXCL_STOP
381 },
382 ctx_.journal);
383 if (ter != tesSUCCESS)
384 return ter;
385
386 // Transfer any XRP remaining after the fee is paid to the destination:
387 (*dst)[sfBalance] = (*dst)[sfBalance] + mSourceBalance;
388 (*src)[sfBalance] = (*src)[sfBalance] - mSourceBalance;
389 ctx_.deliver(mSourceBalance);
390
391 XRPL_ASSERT(
392 (*src)[sfBalance] == XRPAmount(0),
393 "xrpl::DeleteAccount::doApply : source balance is zero");
394
395 // If there's still an owner directory associated with the source account
396 // delete it.
397 if (view().exists(ownerDirKeylet) && !view().emptyDirDelete(ownerDirKeylet))
398 {
399 JLOG(j_.error()) << "DeleteAccount cannot delete root dir node of "
400 << toBase58(account_);
401 return tecHAS_OBLIGATIONS;
402 }
403
404 // Re-arm the password change fee if we can and need to.
405 if (mSourceBalance > XRPAmount(0) && dst->isFlag(lsfPasswordSpent))
406 dst->clearFlag(lsfPasswordSpent);
407
408 view().update(dst);
409 view().erase(src);
410
411 return tesSUCCESS;
412}
