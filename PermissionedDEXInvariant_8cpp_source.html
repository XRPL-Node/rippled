40bool
41ValidPermissionedDEX::finalize(
42 STTx const& tx,
43 TER const result,
44 XRPAmount const,
45 ReadView const& view,
46 beast::Journal const& j)
47{
48 auto const txType = tx.getTxnType();
49 if ((txType != ttPAYMENT && txType != ttOFFER_CREATE) || result != tesSUCCESS)
50 return true;
51
52 // For each offercreate transaction, check if
53 // permissioned offers are valid
54 if (txType == ttOFFER_CREATE && badHybrids_)
55 {
56 JLOG(j.fatal()) << "Invariant failed: hybrid offer is malformed";
57 return false;
58 }
59
60 if (!tx.isFieldPresent(sfDomainID))
61 return true;
62
63 auto const domain = tx.getFieldH256(sfDomainID);
64
65 if (!view.exists(keylet::permissionedDomain(domain)))
66 {
67 JLOG(j.fatal()) << "Invariant failed: domain doesn't exist";
68 return false;
69 }
70
71 // for both payment and offercreate, there shouldn't be another domain
72 // that's different from the domain specified
73 for (auto const& d : domains_)
74 {
75 if (d != domain)
76 {
77 JLOG(j.fatal()) << "Invariant failed: transaction"
78 " consumed wrong domains";
79 return false;
80 }
81 }
82
83 if (regularOffers_)
84 {
85 JLOG(j.fatal()) << "Invariant failed: domain transaction"
86 " affected regular offers";
87 return false;
88 }
89
90 return true;
91}
