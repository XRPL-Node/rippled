24TER
25MPTokenAuthorize::preclaim(PreclaimContext const& ctx)
26{
27 auto const accountID = ctx.tx[sfAccount];
28 auto const holderID = ctx.tx[~sfHolder];
29
30 // if non-issuer account submits this tx, then they are trying either:
31 // 1. Unauthorize/delete MPToken
32 // 2. Use/create MPToken
33 //
34 // Note: `accountID` is holder's account
35 // `holderID` is NOT used
36 if (!holderID)
37 {
38 std::shared_ptr<SLE const> sleMpt =
39 ctx.view.read(keylet::mptoken(ctx.tx[sfMPTokenIssuanceID], accountID));
40
41 // There is an edge case where all holders have zero balance, issuance
42 // is legally destroyed, then outstanding MPT(s) are deleted afterwards.
43 // Thus, there is no need to check for the existence of the issuance if
44 // the MPT is being deleted with a zero balance. Check for unauthorize
45 // before fetching the MPTIssuance object.
46
47 // if holder wants to delete/unauthorize a mpt
48 if (ctx.tx.getFlags() & tfMPTUnauthorize)
49 {
50 if (!sleMpt)
51 return tecOBJECT_NOT_FOUND;
52
53 if ((*sleMpt)[sfMPTAmount] != 0)
54 {
55 auto const sleMptIssuance =
56 ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
57 if (!sleMptIssuance)
58 return tefINTERNAL; // LCOV_EXCL_LINE
59
60 return tecHAS_OBLIGATIONS;
61 }
62
63 if ((*sleMpt)[~sfLockedAmount].value_or(0) != 0)
64 {
65 auto const sleMptIssuance =
66 ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
67 if (!sleMptIssuance)
68 return tefINTERNAL; // LCOV_EXCL_LINE
69
70 return tecHAS_OBLIGATIONS;
71 }
72 if (ctx.view.rules().enabled(featureSingleAssetVault) && sleMpt->isFlag(lsfMPTLocked))
73 return tecNO_PERMISSION;
74
75 return tesSUCCESS;
76 }
77
78 // Now test when the holder wants to hold/create/authorize a new MPT
79 auto const sleMptIssuance = ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
80
81 if (!sleMptIssuance)
82 return tecOBJECT_NOT_FOUND;
83
84 if (accountID == (*sleMptIssuance)[sfIssuer])
85 return tecNO_PERMISSION;
86
87 // if holder wants to use and create a mpt
88 if (sleMpt)
89 return tecDUPLICATE;
90
91 return tesSUCCESS;
92 }
93
94 auto const sleHolder = ctx.view.read(keylet::account(*holderID));
95 if (!sleHolder)
96 return tecNO_DST;
97
98 auto const sleMptIssuance = ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
99 if (!sleMptIssuance)
100 return tecOBJECT_NOT_FOUND;
101
102 std::uint32_t const mptIssuanceFlags = sleMptIssuance->getFieldU32(sfFlags);
103
104 // If tx is submitted by issuer, they would either try to do the following
105 // for allowlisting:
106 // 1. authorize an account
107 // 2. unauthorize an account
108 //
109 // Note: `accountID` is issuer's account
110 // `holderID` is holder's account
111 if (accountID != (*sleMptIssuance)[sfIssuer])
112 return tecNO_PERMISSION;
113
114 // If tx is submitted by issuer, it only applies for MPT with
115 // lsfMPTRequireAuth set
116 if (!(mptIssuanceFlags & lsfMPTRequireAuth))
117 return tecNO_AUTH;
118
119 // The holder must create the MPT before the issuer can authorize it.
120 if (!ctx.view.exists(keylet::mptoken(ctx.tx[sfMPTokenIssuanceID], *holderID)))
121 return tecOBJECT_NOT_FOUND;
122
123 // Can't unauthorize the pseudo-accounts because they are implicitly
124 // always authorized. No need to amendment gate since Vault and LoanBroker
125 // can only be created if the Vault amendment is enabled.
126 if (isPseudoAccount(ctx.view, *holderID, {&sfVaultID, &sfLoanBrokerID}))
127 return tecNO_PERMISSION;
128
129 return tesSUCCESS;
130}

138{

143

148 (*mptoken)[sfAccount] = account;

151 (*mptoken)[sfOwnerNode] = *ownerNode;
