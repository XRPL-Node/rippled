44inline beast::xor_shift_engine&
45default_prng()
46{
47 // This is used to seed the thread-specific PRNGs on demand
48 static beast::xor_shift_engine seeder = [] {
49 std::random_device rng;
50 std::uniform_int_distribution<std::uint64_t> distribution{1};
51 return beast::xor_shift_engine(distribution(rng));
52 }();
53
54 // This protects the seeder
55 static std::mutex m;
56
57 // The thread-specific PRNGs:
58 thread_local beast::xor_shift_engine engine = [] {
59 std::uint64_t seed;
60 {
61 std::lock_guard lk(m);
62 std::uniform_int_distribution<std::uint64_t> distribution{1};
63 seed = distribution(seeder);
64 }
65 return beast::xor_shift_engine{seed};
66 }();
67
68 return engine;
69}
