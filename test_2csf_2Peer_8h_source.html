379 // Attempt to acquire the Ledger associated with the given ID
380 Ledger const*
381 acquireLedger(Ledger::ID const& ledgerID)
382 {
383 if (auto it = ledgers.find(ledgerID); it != ledgers.end())
384 {
385 return &(it->second);
386 }
387
388 // No peers
389 if (net.links(this).empty())
390 return nullptr;
391
392 // Don't retry if we already are acquiring it and haven't timed out
393 auto aIt = acquiringLedgers.find(ledgerID);
394 if (aIt != acquiringLedgers.end())
395 {
396 if (scheduler.now() < aIt->second)
397 return nullptr;
398 }
399
400 using namespace std::chrono_literals;
401 SimDuration minDuration{10s};
402 for (auto const link : net.links(this))
403 {
404 minDuration = std::min(minDuration, link.data.delay);
405
406 // Send a message to neighbors to find the ledger
407 net.send(this, link.target, [to = link.target, from = this, ledgerID]() {
408 if (auto it = to->ledgers.find(ledgerID); it != to->ledgers.end())
409 {
410 // if the ledger is found, send it back to the original
411 // requesting peer where it is added to the available
412 // ledgers
413 to->net.send(to, from, [from, ledger = it->second]() {
414 from->acquiringLedgers.erase(ledger.id());
415 from->ledgers.emplace(ledger.id(), ledger);
416 });
417 }
418 });
419 }
420 acquiringLedgers[ledgerID] = scheduler.now() + 2 * minDuration;
421 return nullptr;
422 }
423
424 // Attempt to acquire the TxSet associated with the given ID
425 TxSet const*
426 acquireTxSet(TxSet::ID const& setId)
427 {
428 if (auto it = txSets.find(setId); it != txSets.end())
429 {
430 return &(it->second);
431 }
432
433 // No peers
434 if (net.links(this).empty())
435 return nullptr;
436
437 // Don't retry if we already are acquiring it and haven't timed out
438 auto aIt = acquiringTxSets.find(setId);
439 if (aIt != acquiringTxSets.end())
440 {
441 if (scheduler.now() < aIt->second)
442 return nullptr;
443 }
444
445 using namespace std::chrono_literals;
446 SimDuration minDuration{10s};
447 for (auto const link : net.links(this))
448 {
449 minDuration = std::min(minDuration, link.data.delay);
450 // Send a message to neighbors to find the tx set
451 net.send(this, link.target, [to = link.target, from = this, setId]() {
452 if (auto it = to->txSets.find(setId); it != to->txSets.end())
453 {
454 // If the txSet is found, send it back to the original
455 // requesting peer, where it is handled like a TxSet
456 // that was broadcast over the network
457 to->net.send(to, from, [from, txSet = it->second]() {
458 from->acquiringTxSets.erase(txSet.id());
459 from->handle(txSet);
460 });
461 }
462 });
463 }
464 acquiringTxSets[setId] = scheduler.now() + 2 * minDuration;
465 return nullptr;
466 }

516 void
517 onAccept(
518 Result const& result,
519 Ledger const& prevLedger,
520 NetClock::duration const& closeResolution,
521 ConsensusCloseTimes const& rawCloseTimes,
522 ConsensusMode const& mode,
523 Json::Value&& consensusJson,
524 bool const validating)
525 {
526 schedule(delays.ledgerAccept, [=, this]() {
527 bool const proposing = mode == ConsensusMode::proposing;
528 bool const consensusFail = result.state == ConsensusState::MovedOn;
529
530 TxSet const acceptedTxs = injectTxs(prevLedger, result.txns);
531 Ledger const newLedger = oracle.accept(
532 prevLedger, acceptedTxs.txs(), closeResolution, result.position.closeTime());
533 ledgers[newLedger.id()] = newLedger;
534
535 issue(AcceptLedger{newLedger, lastClosedLedger});
536 prevProposers = result.proposers;
537 prevRoundTime = result.roundTime.read();
538 lastClosedLedger = newLedger;
539
540 auto const it = std::remove_if(openTxs.begin(), openTxs.end(), [&](Tx const& tx) {
541 return acceptedTxs.exists(tx.id());
542 });
543 openTxs.erase(it, openTxs.end());
544
545 // Only send validation if the new ledger is compatible with our
546 // fully validated ledger
547 bool const isCompatible = newLedger.isAncestor(fullyValidatedLedger);
548
549 // Can only send one validated ledger per seq
550 if (runAsValidator && isCompatible && !consensusFail &&
551 validations.canValidateSeq(newLedger.seq()))
552 {
553 bool isFull = proposing;
554
555 Validation v{newLedger.id(), newLedger.seq(), now(), now(), key, id, isFull};
556 // share the new validation; it is trusted by the receiver
557 share(v);
558 // we trust ourselves
559 addTrustedValidation(v);
560 }
561
562 checkFullyValidated(newLedger);
563
564 // kick off the next round...
565 // in the actual implementation, this passes back through
566 // network ops
567 ++completedLedgers;
568 // startRound sets the LCL state, so we need to call it once after
569 // the last requested round completes
570 if (completedLedgers <= targetLedgers)
571 {
572 startRound();
573 }
574 });
575 }

641 bool
642 addTrustedValidation(Validation v)
643 {
644 v.setTrusted();
645 v.setSeen(now());
646 ValStatus const res = validations.add(v.nodeID(), v);
647
648 if (res == ValStatus::stale)
649 return false;
650
651 // Acquire will try to get from network if not already local
652 if (Ledger const* lgr = acquireLedger(v.ledgerID()))
653 checkFullyValidated(*lgr);
654 return true;
655 }
656
658 void
659 checkFullyValidated(Ledger const& ledger)
660 {
661 // Only consider ledgers newer than our last fully validated ledger
662 if (ledger.seq() <= fullyValidatedLedger.seq())
663 return;
664
665 std::size_t const count = validations.numTrustedForLedger(ledger.id());
666 std::size_t const numTrustedPeers = trustGraph.graph().outDegree(this);
667 quorum = static_cast<std::size_t>(std::ceil(numTrustedPeers * 0.8));
668 if (count >= quorum && ledger.isAncestor(fullyValidatedLedger))
669 {
670 issue(FullyValidateLedger{ledger, fullyValidatedLedger});
671 fullyValidatedLedger = ledger;
672 }
673 }

678 // Basic Sequence number router
679 // A message that will be flooded across the network is tagged with a
680 // sequence number by the origin node in a BroadcastMesg. Receivers will
681 // ignore a message as stale if they've already processed a newer sequence
682 // number, or will process and potentially relay the message along.
683 //
684 // The various bool handle(MessageType) members do the actual processing
685 // and should return true if the message should continue to be sent to
686 // peers.
687 //
688 // WARN: This assumes messages are received and processed in the order they
689 // are sent, so that a peer receives a message with seq 1 from node 0
690 // before seq 2 from node 0, etc.
691 // TODO: Break this out into a class and identify type interface to allow
692 // alternate routing strategies
693 template <class M>
694 struct BroadcastMesg
695 {
696 M msg;
697 std::size_t seq;
698 PeerID origin;
699 };
700
701 struct Router
702 {
703 std::size_t nextSeq = 1;
704 bc::flat_map<PeerID, std::size_t> lastObservedSeq;
705 };
706
707 Router router;
708
709 // Send a broadcast message to all peers
710 template <class M>
711 void
712 send(BroadcastMesg<M> const& bm, PeerID from)
713 {
714 for (auto const link : net.links(this))
715 {
716 if (link.target->id != from && link.target->id != bm.origin)
717 {
718 // cheat and don't bother sending if we know it has already been
719 // used on the other end
720 if (link.target->router.lastObservedSeq[bm.origin] < bm.seq)
721 {
722 issue(Relay<M>{link.target->id, bm.msg});
723 net.send(this, link.target, [to = link.target, bm, id = this->id] {
724 to->receive(bm, id);
725 });
726 }
727 }
728 }
729 }
730
731 // Receive a shared message, process it and consider continuing to relay it
732 template <class M>
733 void
734 receive(BroadcastMesg<M> const& bm, PeerID from)
735 {
736 issue(Receive<M>{from, bm.msg});
737 if (router.lastObservedSeq[bm.origin] < bm.seq)
738 {
739 router.lastObservedSeq[bm.origin] = bm.seq;
740 schedule(delays.onReceive(bm.msg), [this, bm, from] {
741 if (handle(bm.msg))
742 send(bm, from);
743 });
744 }
745 }

747 // Type specific receive handlers, return true if the message should
748 // continue to be broadcast to peers
749 bool
750 handle(Proposal const& p)
751 {
752 // Only relay untrusted proposals on the same ledger
753 if (!trusts(p.nodeID()))
754 return p.prevLedger() == lastClosedLedger.id();
755
756 // TODO: This always suppresses relay of peer positions already seen
757 // Should it allow forwarding if for a recent ledger ?
758 auto& dest = peerPositions[p.prevLedger()];
759 if (std::find(dest.begin(), dest.end(), p) != dest.end())
760 return false;
761
762 dest.push_back(p);
763
764 // Rely on consensus to decide whether to relay
765 return consensus.peerProposal(now(), Position{p});
766 }

778 bool
779 handle(Tx const& tx)
780 {
781 // Ignore and suppress relay of transactions already in last ledger
782 TxSetType const& lastClosedTxs = lastClosedLedger.txs();

784 return false;

898 NetClock::time_point
899 now() const
900 {
901 // We don't care about the actual epochs, but do want the
902 // generated NetClock time to be well past its epoch to ensure
903 // any subtractions of two NetClock::time_point in the consensus
904 // code are positive. (e.g. proposeFRESHNESS)
905 using namespace std::chrono;
906 using namespace std::chrono_literals;
907 return NetClock::time_point(
908 duration_cast<NetClock::duration>(
909 scheduler.now().time_since_epoch() + 86400s + clockSkew));
910 }
