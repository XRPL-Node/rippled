76inline FeatureBitset
77testable_amendments()
78{
79 static FeatureBitset const ids = [] {
80 auto const& sa = allAmendments();
81 std::vector<uint256> feats;
82 feats.reserve(sa.size());
83 for (auto const& [s, vote] : sa)
84 {
85 (void)vote;
86 if (auto const f = getRegisteredFeature(s))
87 feats.push_back(*f);
88 else
89 Throw<std::runtime_error>("Unknown feature: " + s + " in allAmendments.");
90 }
91 return FeatureBitset(feats);
92 }();
93 return ids;
94}
