53inline FeatureBitset
54testable_amendments()
55{
56 static FeatureBitset const ids = [] {
57 auto const& sa = allAmendments();
58 std::vector<uint256> feats;
59 feats.reserve(sa.size());
60 for (auto const& [s, vote] : sa)
61 {
62 (void)vote;
63 if (auto const f = getRegisteredFeature(s))
64 feats.push_back(*f);
65 else
66 Throw<std::runtime_error>("Unknown feature: " + s + " in allAmendments.");
67 }
68 return FeatureBitset(feats);
69 }();
70 return ids;
71}
