27template <typename Type>
28class SlabAllocator
29{
30 static_assert(
31 sizeof(Type) >= sizeof(std::uint8_t*),
32 "SlabAllocator: the requested object must be larger than a pointer.");
33
34 static_assert(alignof(Type) == 8 || alignof(Type) == 4);
35
37 struct SlabBlock
38 {
39 // A mutex to protect the freelist for this block:
40 std::mutex m_;
41
42 // A linked list of appropriately sized free buffers:
43 std::uint8_t* l_ = nullptr;
44
45 // The next memory block
46 SlabBlock* next_;
47
48 // The underlying memory block:
49 std::uint8_t const* const p_ = nullptr;
50
51 // The extent of the underlying memory block:
52 std::size_t const size_;
53
54 SlabBlock(SlabBlock* next, std::uint8_t* data, std::size_t size, std::size_t item)
55 : next_(next), p_(data), size_(size)
56 {
57 // We don't need to grab the mutex here, since we're the only
58 // ones with access at this moment.
59
60 while (data + item <= p_ + size_)
61 {
62 // Use memcpy to avoid unaligned UB
63 // (will optimize to equivalent code)
64 std::memcpy(data, &l_, sizeof(std::uint8_t*));
65 l_ = data;
66 data += item;
67 }
68 }
69
70 ~SlabBlock()
71 {
72 // Calling this destructor will release the allocated memory but
73 // will not properly destroy any objects that are constructed in
74 // the block itself.
75 }
76
77 SlabBlock(SlabBlock const& other) = delete;
78 SlabBlock&
79 operator=(SlabBlock const& other) = delete;
80
81 SlabBlock(SlabBlock&& other) = delete;
82 SlabBlock&
83 operator=(SlabBlock&& other) = delete;
84
86 bool
87 own(std::uint8_t const* p) const noexcept
88 {
89 return (p >= p_) && (p < p_ + size_);
90 }
91
92 std::uint8_t*
93 allocate() noexcept
94 {
95 std::uint8_t* ret;
96
97 {
98 std::lock_guard l(m_);
99
100 ret = l_;
101
102 if (ret)
103 {
104 // Use memcpy to avoid unaligned UB
105 // (will optimize to equivalent code)
106 std::memcpy(&l_, ret, sizeof(std::uint8_t*));
107 }
108 }
109
110 return ret;
111 }
112
122 void
123 deallocate(std::uint8_t* ptr) noexcept
124 {
125 XRPL_ASSERT(own(ptr), "xrpl::SlabAllocator::SlabBlock::deallocate : own input");
126
127 std::lock_guard l(m_);
128
129 // Use memcpy to avoid unaligned UB
130 // (will optimize to equivalent code)
131 std::memcpy(ptr, &l_, sizeof(std::uint8_t*));
132 l_ = ptr;
133 }
134 };
135
136private:
137 // A linked list of slabs
138 std::atomic<SlabBlock*> slabs_ = nullptr;
139
140 // The alignment requirements of the item we're allocating:
141 std::size_t const itemAlignment_;
142
143 // The size of an item, including the extra bytes requested and
144 // any padding needed for alignment purposes:
145 std::size_t const itemSize_;
146
147 // The size of each individual slab:
148 std::size_t const slabSize_;
149
150public:
159 constexpr explicit SlabAllocator(std::size_t extra, std::size_t alloc = 0, std::size_t align = 0)
160 : itemAlignment_(align ? align : alignof(Type))
161 , itemSize_(boost::alignment::align_up(sizeof(Type) + extra, itemAlignment_))
162 , slabSize_(alloc)
163 {
164 XRPL_ASSERT(
165 (itemAlignment_ & (itemAlignment_ - 1)) == 0, "xrpl::SlabAllocator::SlabAllocator : valid alignment");
166 }
167
168 SlabAllocator(SlabAllocator const& other) = delete;
169 SlabAllocator&
170 operator=(SlabAllocator const& other) = delete;
171
172 SlabAllocator(SlabAllocator&& other) = delete;
173 SlabAllocator&
174 operator=(SlabAllocator&& other) = delete;
175
176 ~SlabAllocator()
177 {
178 // FIXME: We can't destroy the memory blocks we've allocated, because
179 // we can't be sure that they are not being used. Cleaning the
180 // shutdown process up could make this possible.
181 }
182
184 constexpr std::size_t
185 size() const noexcept
186 {
187 return itemSize_;
188 }
189
195 std::uint8_t*
196 allocate() noexcept
197 {
198 auto slab = slabs_.load();
199
200 while (slab != nullptr)
201 {
202 if (auto ret = slab->allocate())
203 return ret;
204
205 slab = slab->next_;
206 }
207
208 // No slab can satisfy our request, so we attempt to allocate a new
209 // one here:
210 std::size_t size = slabSize_;
211
212 // We want to allocate the memory at a 2 MiB boundary, to make it
213 // possible to use hugepage mappings on Linux:
214 auto buf = boost::alignment::aligned_alloc(megabytes(std::size_t(2)), size);
215
216 // clang-format off
217 if (!buf) [[unlikely]]
218 return nullptr;
219 // clang-format on
220
221#if BOOST_OS_LINUX
222 // When allocating large blocks, attempt to leverage Linux's
223 // transparent hugepage support. It is unclear and difficult
224 // to accurately determine if doing this impacts performance
225 // enough to justify using platform-specific tricks.
226 if (size >= megabytes(std::size_t(4)))
227 madvise(buf, size, MADV_HUGEPAGE);
228#endif
229
230 // We need to carve out a bit of memory for the slab header
231 // and then align the rest appropriately:
232 auto slabData = reinterpret_cast<void*>(reinterpret_cast<std::uint8_t*>(buf) + sizeof(SlabBlock));
233 auto slabSize = size - sizeof(SlabBlock);
234
235 // This operation is essentially guaranteed not to fail but
236 // let's be careful anyways.
237 if (!boost::alignment::align(itemAlignment_, itemSize_, slabData, slabSize))
238 {
239 boost::alignment::aligned_free(buf);
240 return nullptr;
241 }
242
243 slab = new (buf) SlabBlock(slabs_.load(), reinterpret_cast<std::uint8_t*>(slabData), slabSize, itemSize_);
244
245 // Link the new slab
246 while (!slabs_.compare_exchange_weak(slab->next_, slab, std::memory_order_release, std::memory_order_relaxed))
247 {
248 ; // Nothing to do
249 }
250
251 return slab->allocate();
252 }
253
261 bool
262 deallocate(std::uint8_t* ptr) noexcept
263 {
264 XRPL_ASSERT(
265 ptr,
266 "xrpl::SlabAllocator::SlabAllocator::deallocate : non-null "
267 "input");
268
269 for (auto slab = slabs_.load(); slab != nullptr; slab = slab->next_)
270 {
271 if (slab->own(ptr))
272 {
273 slab->deallocate(ptr);
274 return true;
275 }
276 }
277
278 return false;
279 }
280};

283template <typename Type>
284class SlabAllocatorSet
285{
286private:
287 // The list of allocators that belong to this set
288 boost::container::static_vector<SlabAllocator<Type>, 64> allocators_;
289
290 std::size_t maxSize_ = 0;
291
292public:
293 class SlabConfig
294 {
295 friend class SlabAllocatorSet;
296
297 private:
298 std::size_t extra;
299 std::size_t alloc;
300 std::size_t align;
301
302 public:
303 constexpr SlabConfig(std::size_t extra_, std::size_t alloc_ = 0, std::size_t align_ = alignof(Type))
304 : extra(extra_), alloc(alloc_), align(align_)
305 {
306 }
307 };
308
309 constexpr SlabAllocatorSet(std::vector<SlabConfig> cfg)
310 {
311 // Ensure that the specified allocators are sorted from smallest to
312 // largest by size:
313 std::sort(
314 std::begin(cfg), std::end(cfg), [](SlabConfig const& a, SlabConfig const& b) { return a.extra < b.extra; });
315
316 // We should never have two slabs of the same size
317 if (std::adjacent_find(std::begin(cfg), std::end(cfg), [](SlabConfig const& a, SlabConfig const& b) {
318 return a.extra == b.extra;
319 }) != cfg.end())
320 {
321 throw std::runtime_error("SlabAllocatorSet<" + beast::type_name<Type>() + ">: duplicate slab size");
322 }
323
324 for (auto const& c : cfg)
325 {
326 auto& a = allocators_.emplace_back(c.extra, c.alloc, c.align);
327
328 if (a.size() > maxSize_)
329 maxSize_ = a.size();
330 }
331 }
332
333 SlabAllocatorSet(SlabAllocatorSet const& other) = delete;
334 SlabAllocatorSet&
335 operator=(SlabAllocatorSet const& other) = delete;
336
337 SlabAllocatorSet(SlabAllocatorSet&& other) = delete;
338 SlabAllocatorSet&
339 operator=(SlabAllocatorSet&& other) = delete;
340
341 ~SlabAllocatorSet()
342 {
343 }
344
353 std::uint8_t*
354 allocate(std::size_t extra) noexcept
355 {
356 if (auto const size = sizeof(Type) + extra; size <= maxSize_)
357 {
358 for (auto& a : allocators_)
359 {
360 if (a.size() >= size)
361 return a.allocate();
362 }
363 }
364
365 return nullptr;
366 }
367
375 bool
376 deallocate(std::uint8_t* ptr) noexcept
377 {
378 for (auto& a : allocators_)
379 {
380 if (a.deallocate(ptr))
381 return true;
382 }
383
384 return false;
385 }
386};
