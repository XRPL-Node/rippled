41bool
42ValidMPTIssuance::finalize(
43 STTx const& tx,
44 TER const result,
45 XRPAmount const _fee,
46 ReadView const& view,
47 beast::Journal const& j)
48{
49 if (result == tesSUCCESS)
50 {
51 auto const& rules = view.rules();
52 [[maybe_unused]]
53 bool enforceCreatedByIssuer =
54 rules.enabled(featureSingleAssetVault) || rules.enabled(featureLendingProtocol);
55 if (mptCreatedByIssuer_)
56 {
57 JLOG(j.fatal()) << "Invariant failed: MPToken created for the MPT issuer";
58 // The comment above starting with "assert(enforce)" explains this
59 // assert.
60 XRPL_ASSERT_PARTS(
61 enforceCreatedByIssuer, "xrpl::ValidMPTIssuance::finalize", "no issuer MPToken");
62 if (enforceCreatedByIssuer)
63 return false;
64 }
65
66 auto const txnType = tx.getTxnType();
67 if (hasPrivilege(tx, createMPTIssuance))
68 {
69 if (mptIssuancesCreated_ == 0)
70 {
71 JLOG(j.fatal()) << "Invariant failed: transaction "
72 "succeeded without creating a MPT issuance";
73 }
74 else if (mptIssuancesDeleted_ != 0)
75 {
76 JLOG(j.fatal()) << "Invariant failed: transaction "
77 "succeeded while removing MPT issuances";
78 }
79 else if (mptIssuancesCreated_ > 1)
80 {
81 JLOG(j.fatal()) << "Invariant failed: transaction "
82 "succeeded but created multiple issuances";
83 }
84
85 return mptIssuancesCreated_ == 1 && mptIssuancesDeleted_ == 0;
86 }
87
88 if (hasPrivilege(tx, destroyMPTIssuance))
89 {
90 if (mptIssuancesDeleted_ == 0)
91 {
92 JLOG(j.fatal()) << "Invariant failed: MPT issuance deletion "
93 "succeeded without removing a MPT issuance";
94 }
95 else if (mptIssuancesCreated_ > 0)
96 {
97 JLOG(j.fatal()) << "Invariant failed: MPT issuance deletion "
98 "succeeded while creating MPT issuances";
99 }
100 else if (mptIssuancesDeleted_ > 1)
101 {
102 JLOG(j.fatal()) << "Invariant failed: MPT issuance deletion "
103 "succeeded but deleted multiple issuances";
104 }
105
106 return mptIssuancesCreated_ == 0 && mptIssuancesDeleted_ == 1;
107 }
108
109 bool const lendingProtocolEnabled = view.rules().enabled(featureLendingProtocol);
110 // ttESCROW_FINISH may authorize an MPT, but it can't have the
111 // mayAuthorizeMPT privilege, because that may cause
112 // non-amendment-gated side effects.
113 bool const enforceEscrowFinish = (txnType == ttESCROW_FINISH) &&
114 (view.rules().enabled(featureSingleAssetVault) || lendingProtocolEnabled);
115 if (hasPrivilege(tx, mustAuthorizeMPT | mayAuthorizeMPT) || enforceEscrowFinish)
116 {
117 bool const submittedByIssuer = tx.isFieldPresent(sfHolder);
118
119 if (mptIssuancesCreated_ > 0)
120 {
121 JLOG(j.fatal()) << "Invariant failed: MPT authorize "
122 "succeeded but created MPT issuances";
123 return false;
124 }
125 else if (mptIssuancesDeleted_ > 0)
126 {
127 JLOG(j.fatal()) << "Invariant failed: MPT authorize "
128 "succeeded but deleted issuances";
129 return false;
130 }
131 else if (lendingProtocolEnabled && mptokensCreated_ + mptokensDeleted_ > 1)
132 {
133 JLOG(j.fatal()) << "Invariant failed: MPT authorize succeeded "
134 "but created/deleted bad number mptokens";
135 return false;
136 }
137 else if (submittedByIssuer && (mptokensCreated_ > 0 || mptokensDeleted_ > 0))
138 {
139 JLOG(j.fatal()) << "Invariant failed: MPT authorize submitted by issuer "
140 "succeeded but created/deleted mptokens";
141 return false;
142 }
143 else if (
144 !submittedByIssuer && hasPrivilege(tx, mustAuthorizeMPT) &&
145 (mptokensCreated_ + mptokensDeleted_ != 1))
146 {
147 // if the holder submitted this tx, then a mptoken must be
148 // either created or deleted.
149 JLOG(j.fatal()) << "Invariant failed: MPT authorize submitted by holder "
150 "succeeded but created/deleted bad number of mptokens";
151 return false;
152 }
153
154 return true;
155 }
156 if (txnType == ttESCROW_FINISH)
157 {
158 // ttESCROW_FINISH may authorize an MPT, but it can't have the
159 // mayAuthorizeMPT privilege, because that may cause
160 // non-amendment-gated side effects.
161 XRPL_ASSERT_PARTS(
162 !enforceEscrowFinish, "xrpl::ValidMPTIssuance::finalize", "not escrow finish tx");
163 return true;
164 }
165
166 if (hasPrivilege(tx, mayDeleteMPT) && mptokensDeleted_ == 1 && mptokensCreated_ == 0 &&
167 mptIssuancesCreated_ == 0 && mptIssuancesDeleted_ == 0)
168 return true;
169 }
170
171 if (mptIssuancesCreated_ != 0)
172 {
173 JLOG(j.fatal()) << "Invariant failed: a MPT issuance was created";
174 }
175 else if (mptIssuancesDeleted_ != 0)
176 {
177 JLOG(j.fatal()) << "Invariant failed: a MPT issuance was deleted";
178 }
179 else if (mptokensCreated_ != 0)
180 {
181 JLOG(j.fatal()) << "Invariant failed: a MPToken was created";
182 }
183 else if (mptokensDeleted_ != 0)
184 {
185 JLOG(j.fatal()) << "Invariant failed: a MPToken was deleted";
186 }
187
188 return mptIssuancesCreated_ == 0 && mptIssuancesDeleted_ == 0 && mptokensCreated_ == 0 &&
189 mptokensDeleted_ == 0;
190}
