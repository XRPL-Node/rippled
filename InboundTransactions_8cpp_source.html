116 void
117 gotData(
118 LedgerHash const& hash,
119 std::shared_ptr<Peer> peer,
120 std::shared_ptr<protocol::TMLedgerData> packet_ptr) override
121 {
122 protocol::TMLedgerData& packet = *packet_ptr;
123
124 JLOG(j_.trace()) << "Got data (" << packet.nodes().size()
125 << ") for acquiring ledger: " << hash;
126
127 TransactionAcquire::pointer ta = getAcquire(hash);
128
129 if (ta == nullptr)
130 {
131 peer->charge(Resource::feeUselessData, "ledger_data");
132 return;
133 }
134
135 std::vector<std::pair<SHAMapNodeID, Slice>> data;
136 data.reserve(packet.nodes().size());
137
138 for (auto const& node : packet.nodes())
139 {
140 if (!node.has_nodeid() || !node.has_nodedata())
141 {
142 peer->charge(Resource::feeMalformedRequest, "ledger_data");
143 return;
144 }
145
146 auto const id = deserializeSHAMapNodeID(node.nodeid());
147
148 if (!id)
149 {
150 peer->charge(Resource::feeInvalidData, "ledger_data");
151 return;
152 }
153
154 data.emplace_back(std::make_pair(*id, makeSlice(node.nodedata())));
155 }
156
157 if (!ta->takeNodes(data, peer).isUseful())
158 peer->charge(Resource::feeUselessData, "ledger_data not useful");
159 }

163 {

179 inboundSet.mAcquire.reset();
180 }

186 void
187 newRound(std::uint32_t seq) override
188 {
189 std::lock_guard lock(mLock);
190
191 // Protect zero set from expiration
192 m_zeroSet.mSeq = seq;
193
194 if (m_seq != seq)
195 {
196 m_seq = seq;
197
198 auto it = m_map.begin();
199
200 std::uint32_t const minSeq = (seq < setKeepRounds) ? 0 : (seq - setKeepRounds);
201 std::uint32_t maxSeq = seq + setKeepRounds;
202
203 while (it != m_map.end())
204 {
205 if (it->second.mSeq < minSeq || it->second.mSeq > maxSeq)
206 it = m_map.erase(it);
207 else
208 ++it;
209 }
210 }
211 }
