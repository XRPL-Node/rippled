48void
49LedgerDeltaAcquire::trigger(std::size_t limit, ScopedLockType& sl)
50{
51 fullLedger_ = app_.getLedgerMaster().getLedgerByHash(hash_);
52 if (fullLedger_)
53 {
54 complete_ = true;
55 JLOG(journal_.trace()) << "existing ledger " << hash_;
56 notify(sl);
57 return;
58 }
59
60 if (!fallBack_)
61 {
62 peerSet_->addPeers(
63 limit,
64 [this](auto peer) {
65 return peer->supportsFeature(ProtocolFeature::LedgerReplay) &&
66 peer->hasLedger(hash_, ledgerSeq_);
67 },
68 [this](auto peer) {
69 if (peer->supportsFeature(ProtocolFeature::LedgerReplay))
70 {
71 JLOG(journal_.trace()) << "Add a peer " << peer->id() << " for " << hash_;
72 protocol::TMReplayDeltaRequest request;
73 request.set_ledgerhash(hash_.data(), hash_.size());
74 peerSet_->sendRequest(request, peer);
75 }
76 else
77 {
78 if (++noFeaturePeerCount >= LedgerReplayParameters::MAX_NO_FEATURE_PEER_COUNT)
79 {
80 JLOG(journal_.debug()) << "Fall back for " << hash_;
81 timerInterval_ = LedgerReplayParameters::SUB_TASK_FALLBACK_TIMEOUT;
82 fallBack_ = true;
83 }
84 }
85 });
86 }
87
88 if (fallBack_)
89 inboundLedgers_.acquire(hash_, ledgerSeq_, InboundLedger::Reason::GENERIC);
90}

117 std::map<std::uint32_t, std::shared_ptr<STTx const>>&& orderedTxns)
118{
119 ScopedLockType sl(mtx_);
120 JLOG(journal_.trace()) << "got data for " << hash_;
121 if (isDone())
122 return;
123
124 if (info.seq == ledgerSeq_)
125 {
126 // create a temporary ledger for building a LedgerReplay object later
127 replayTemp_ = std::make_shared<Ledger>(info, app_.config(), app_.getNodeFamily());
128 if (replayTemp_)
129 {
130 complete_ = true;
131 orderedTxns_ = std::move(orderedTxns);
132 JLOG(journal_.debug()) << "ready to replay " << hash_;
133 notify(sl);
134 return;
135 }
136 }
137
138 failed_ = true;
139 JLOG(journal_.error()) << "failed to create a (info only) ledger from verified data " << hash_;
140 notify(sl);

162std::shared_ptr<Ledger const>
163LedgerDeltaAcquire::tryBuild(std::shared_ptr<Ledger const> const& parent)
164{
165 ScopedLockType sl(mtx_);
166
167 if (fullLedger_)
168 return fullLedger_;
169
170 if (failed_ || !complete_ || !replayTemp_)
171 return {};
172
173 XRPL_ASSERT(
174 parent->seq() + 1 == replayTemp_->seq(),
175 "xrpl::LedgerDeltaAcquire::tryBuild : parent sequence match");
176 XRPL_ASSERT(
177 parent->header().hash == replayTemp_->header().parentHash,
178 "xrpl::LedgerDeltaAcquire::tryBuild : parent hash match");
179 // build ledger
180 LedgerReplay replayData(parent, replayTemp_, std::move(orderedTxns_));
181 fullLedger_ = buildLedger(replayData, tapNONE, app_, journal_);
182 if (fullLedger_ && fullLedger_->header().hash == hash_)
183 {
184 JLOG(journal_.info()) << "Built " << hash_;
185 onLedgerBuilt(sl);
186 return fullLedger_;
187 }
188 else
189 {
190 failed_ = true;
191 complete_ = false;
192 JLOG(journal_.error()) << "tryBuild failed " << hash_ << " with parent "
193 << parent->header().hash;
194 Throw<std::runtime_error>("Cannot replay ledger");
195 }
196}

198void
199LedgerDeltaAcquire::onLedgerBuilt(ScopedLockType& sl, std::optional<InboundLedger::Reason> reason)
200{
201 JLOG(journal_.debug()) << "onLedgerBuilt " << hash_ << (reason ? " for a new reason" : "");
202
203 std::vector<InboundLedger::Reason> reasons(reasons_.begin(), reasons_.end());
204 bool firstTime = true;
205 if (reason) // small chance
206 {
207 reasons.clear();
208 reasons.push_back(*reason);
209 firstTime = false;
210 }
211 app_.getJobQueue().addJob(
212 jtREPLAY_TASK, "OnLedBuilt", [=, ledger = this->fullLedger_, &app = this->app_]() {
213 for (auto reason : reasons)
214 {
215 switch (reason)
216 {
217 case InboundLedger::Reason::GENERIC:
218 app.getLedgerMaster().storeLedger(ledger);
219 break;
220 default:
221 // TODO for other use cases
222 break;
223 }
224 }
225
226 if (firstTime)
227 app.getLedgerMaster().tryAdvance();
228 });
229}

232LedgerDeltaAcquire::notify(ScopedLockType& sl)
233{
234 XRPL_ASSERT(isDone(), "xrpl::LedgerDeltaAcquire::notify : is done");
235 std::vector<OnDeltaDataCB> toCall;
236 std::swap(toCall, dataReadyCallbacks_);
237 auto const good = !failed_;
238 sl.unlock();
239
240 for (auto& cb : toCall)
241 {
242 cb(good, hash_);
243 }
244
245 sl.lock();
246}
