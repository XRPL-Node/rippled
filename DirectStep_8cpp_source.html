303std::uint32_t
304DirectIPaymentStep::quality(ReadView const& sb, QualityDirection qDir) const
305{
306 if (src_ == dst_)
307 return QUALITY_ONE;
308
309 auto const sle = sb.read(keylet::line(dst_, src_, currency_));
310
311 if (!sle)
312 return QUALITY_ONE;
313
314 auto const& field = [&, this]() -> SF_UINT32 const& {
315 if (qDir == QualityDirection::in)
316 {
317 // compute dst quality in
318 if (this->dst_ < this->src_)
319 return sfLowQualityIn;
320 else
321 return sfHighQualityIn;
322 }
323 else
324 {
325 // compute src quality out
326 if (this->src_ < this->dst_)
327 return sfLowQualityOut;
328 else
329 return sfHighQualityOut;
330 }
331 }();
332
333 if (!sle->isFieldPresent(field))
334 return QUALITY_ONE;
335
336 auto const q = (*sle)[field];
337 if (!q)
338 return QUALITY_ONE;
339 return q;
340}

356std::pair<IOUAmount, DebtDirection>
357DirectIOfferCrossingStep::maxFlow(ReadView const& sb, IOUAmount const& desired) const
358{
359 // When isLast and offer crossing then ignore trust line limits. Offer
360 // crossing has the ability to exceed the limit set by a trust line.
361 // We presume that if someone is creating an offer then they intend to
362 // fill as much of that offer as possible, even if the offer exceeds
363 // the limit that a trust line sets.
364 //
365 // A note on using "out" as the desired parameter for maxFlow. In some
366 // circumstances during payments we end up needing a value larger than
367 // "out" for "maxSrcToDst". But as of now (June 2016) that never happens
368 // during offer crossing. That's because, due to a couple of factors,
369 // "dstQIn" is always QUALITY_ONE for offer crossing.
370
371 if (isLast_)
372 return {desired, DebtDirection::issues};
373
374 return maxPaymentFlow(sb);
375}

377TER
378DirectIPaymentStep::check(StrandContext const& ctx, std::shared_ptr<const SLE> const& sleSrc) const
379{
380 // Since this is a payment a trust line must be present. Perform all
381 // trust line related checks.
382 {
383 auto const sleLine = ctx.view.read(keylet::line(src_, dst_, currency_));
384 if (!sleLine)
385 {
386 JLOG(j_.trace()) << "DirectStepI: No credit line. " << *this;
387 return terNO_LINE;
388 }
389
390 auto const authField = (src_ > dst_) ? lsfHighAuth : lsfLowAuth;
391
392 if (((*sleSrc)[sfFlags] & lsfRequireAuth) && !((*sleLine)[sfFlags] & authField) &&
393 (*sleLine)[sfBalance] == beast::zero)
394 {
395 JLOG(j_.debug()) << "DirectStepI: can't receive IOUs from issuer without auth."
396 << " src: " << src_;
397 return terNO_AUTH;
398 }
399
400 if (ctx.prevStep)
401 {
402 if (ctx.prevStep->bookStepBook())
403 {
404 auto const noRippleSrcToDst =
405 ((*sleLine)[sfFlags] & ((src_ > dst_) ? lsfHighNoRipple : lsfLowNoRipple));
406 if (noRippleSrcToDst)
407 return terNO_RIPPLE;
408 }
409 }
410 }
411
412 {
413 auto const owed = creditBalance(ctx.view, dst_, src_, currency_);
414 if (owed <= beast::zero)
415 {
416 auto const limit = creditLimit(ctx.view, dst_, src_, currency_);
417 if (-owed >= limit)
418 {
419 JLOG(j_.debug()) << "DirectStepI: dry: owed: " << owed << " limit: " << limit;
420 return tecPATH_DRY;
421 }
422 }
423 }
424 return tesSUCCESS;
425}

438template <class TDerived>
439std::pair<IOUAmount, DebtDirection>
440DirectStepI<TDerived>::maxPaymentFlow(ReadView const& sb) const
441{
442 auto const srcOwed =
443 toAmount<IOUAmount>(accountHolds(sb, src_, currency_, dst_, fhIGNORE_FREEZE, j_));
444
445 if (srcOwed.signum() > 0)
446 return {srcOwed, DebtDirection::redeems};
447
448 // srcOwed is negative or zero
449 return {creditLimit2(sb, dst_, src_, currency_) + srcOwed, DebtDirection::issues};
450}

463template <class TDerived>
464std::pair<IOUAmount, IOUAmount>
465DirectStepI<TDerived>::revImp(
466 PaymentSandbox& sb,
467 ApplyView& /*afView*/,
468 boost::container::flat_set<uint256>& /*ofrsToRm*/,
469 IOUAmount const& out)
470{
471 cache_.reset();
472
473 auto const [maxSrcToDst, srcDebtDir] = static_cast<TDerived const*>(this)->maxFlow(sb, out);
474
475 auto const [srcQOut, dstQIn] = qualities(sb, srcDebtDir, StrandDirection::reverse);
476 XRPL_ASSERT(
477 static_cast<TDerived const*>(this)->verifyDstQualityIn(dstQIn),
478 "xrpl::DirectStepI : valid destination quality");
479
480 Issue const srcToDstIss(currency_, redeems(srcDebtDir) ? dst_ : src_);
481
482 JLOG(j_.trace()) << "DirectStepI::rev"
483 << " srcRedeems: " << redeems(srcDebtDir) << " outReq: " << to_string(out)
484 << " maxSrcToDst: " << to_string(maxSrcToDst) << " srcQOut: " << srcQOut
485 << " dstQIn: " << dstQIn;
486
487 if (maxSrcToDst.signum() <= 0)
488 {
489 JLOG(j_.trace()) << "DirectStepI::rev: dry";
490 cache_.emplace(
491 IOUAmount(beast::zero), IOUAmount(beast::zero), IOUAmount(beast::zero), srcDebtDir);
492 return {beast::zero, beast::zero};
493 }
494
495 IOUAmount const srcToDst = mulRatio(out, QUALITY_ONE, dstQIn, /*roundUp*/ true);
496
497 if (srcToDst <= maxSrcToDst)
498 {
499 IOUAmount const in = mulRatio(srcToDst, srcQOut, QUALITY_ONE, /*roundUp*/ true);
500 cache_.emplace(in, srcToDst, out, srcDebtDir);
501 rippleCredit(
502 sb,
503 src_,
504 dst_,
505 toSTAmount(srcToDst, srcToDstIss),
506 /*checkIssuer*/ true,
507 j_);
508 JLOG(j_.trace()) << "DirectStepI::rev: Non-limiting"
509 << " srcRedeems: " << redeems(srcDebtDir) << " in: " << to_string(in)
510 << " srcToDst: " << to_string(srcToDst) << " out: " << to_string(out);
511 return {in, out};
512 }
513
514 // limiting node
515 IOUAmount const in = mulRatio(maxSrcToDst, srcQOut, QUALITY_ONE, /*roundUp*/ true);
516 IOUAmount const actualOut = mulRatio(maxSrcToDst, dstQIn, QUALITY_ONE, /*roundUp*/ false);
517 cache_.emplace(in, maxSrcToDst, actualOut, srcDebtDir);
518 rippleCredit(
519 sb,
520 src_,
521 dst_,
522 toSTAmount(maxSrcToDst, srcToDstIss),
523 /*checkIssuer*/ true,
524 j_);
525 JLOG(j_.trace()) << "DirectStepI::rev: Limiting"
526 << " srcRedeems: " << redeems(srcDebtDir) << " in: " << to_string(in)
527 << " srcToDst: " << to_string(maxSrcToDst) << " out: " << to_string(out);
528 return {in, actualOut};
529}

531// The forward pass should never have more liquidity than the reverse
532// pass. But sometimes rounding differences cause the forward pass to
533// deliver more liquidity. Use the cached values from the reverse pass
534// to prevent this.
535template <class TDerived>
536void
537DirectStepI<TDerived>::setCacheLimiting(
538 IOUAmount const& fwdIn,
539 IOUAmount const& fwdSrcToDst,
540 IOUAmount const& fwdOut,
541 DebtDirection srcDebtDir)
542{
543 if (cache_->in < fwdIn)
544 {
545 IOUAmount const smallDiff(1, -9);
546 auto const diff = fwdIn - cache_->in;
547 if (diff > smallDiff)
548 {
549 if (fwdIn.exponent() != cache_->in.exponent() || !cache_->in.mantissa() ||
550 (double(fwdIn.mantissa()) / double(cache_->in.mantissa())) > 1.01)
551 {
552 // Detect large diffs on forward pass so they may be
553 // investigated
554 JLOG(j_.warn()) << "DirectStepI::fwd: setCacheLimiting"
555 << " fwdIn: " << to_string(fwdIn)
556 << " cacheIn: " << to_string(cache_->in)
557 << " fwdSrcToDst: " << to_string(fwdSrcToDst)
558 << " cacheSrcToDst: " << to_string(cache_->srcToDst)
559 << " fwdOut: " << to_string(fwdOut)
560 << " cacheOut: " << to_string(cache_->out);
561 cache_.emplace(fwdIn, fwdSrcToDst, fwdOut, srcDebtDir);
562 return;
563 }
564 }
565 }
566 cache_->in = fwdIn;
567 if (fwdSrcToDst < cache_->srcToDst)
568 cache_->srcToDst = fwdSrcToDst;
569 if (fwdOut < cache_->out)
570 cache_->out = fwdOut;
571 cache_->srcDebtDir = srcDebtDir;
572};

574template <class TDerived>
575std::pair<IOUAmount, IOUAmount>
576DirectStepI<TDerived>::fwdImp(
577 PaymentSandbox& sb,
578 ApplyView& /*afView*/,
579 boost::container::flat_set<uint256>& /*ofrsToRm*/,
580 IOUAmount const& in)
581{
582 XRPL_ASSERT(cache_, "xrpl::DirectStepI::fwdImp : cache is set");
583
584 auto const [maxSrcToDst, srcDebtDir] =
585 static_cast<TDerived const*>(this)->maxFlow(sb, cache_->srcToDst);
586
587 auto const [srcQOut, dstQIn] = qualities(sb, srcDebtDir, StrandDirection::forward);
588
589 Issue const srcToDstIss(currency_, redeems(srcDebtDir) ? dst_ : src_);
590
591 JLOG(j_.trace()) << "DirectStepI::fwd"
592 << " srcRedeems: " << redeems(srcDebtDir) << " inReq: " << to_string(in)
593 << " maxSrcToDst: " << to_string(maxSrcToDst) << " srcQOut: " << srcQOut
594 << " dstQIn: " << dstQIn;
595
596 if (maxSrcToDst.signum() <= 0)
597 {
598 JLOG(j_.trace()) << "DirectStepI::fwd: dry";
599 cache_.emplace(
600 IOUAmount(beast::zero), IOUAmount(beast::zero), IOUAmount(beast::zero), srcDebtDir);
601 return {beast::zero, beast::zero};
602 }
603
604 IOUAmount const srcToDst = mulRatio(in, QUALITY_ONE, srcQOut, /*roundUp*/ false);
605
606 if (srcToDst <= maxSrcToDst)
607 {
608 IOUAmount const out = mulRatio(srcToDst, dstQIn, QUALITY_ONE, /*roundUp*/ false);
609 setCacheLimiting(in, srcToDst, out, srcDebtDir);
610 rippleCredit(
611 sb,
612 src_,
613 dst_,
614 toSTAmount(cache_->srcToDst, srcToDstIss),
615 /*checkIssuer*/ true,
616 j_);
617 JLOG(j_.trace()) << "DirectStepI::fwd: Non-limiting"
618 << " srcRedeems: " << redeems(srcDebtDir) << " in: " << to_string(in)
619 << " srcToDst: " << to_string(srcToDst) << " out: " << to_string(out);
620 }
621 else
622 {
623 // limiting node
624 IOUAmount const actualIn = mulRatio(maxSrcToDst, srcQOut, QUALITY_ONE, /*roundUp*/ true);
625 IOUAmount const out = mulRatio(maxSrcToDst, dstQIn, QUALITY_ONE, /*roundUp*/ false);
626 setCacheLimiting(actualIn, maxSrcToDst, out, srcDebtDir);
627 rippleCredit(
628 sb,
629 src_,
630 dst_,
631 toSTAmount(cache_->srcToDst, srcToDstIss),
632 /*checkIssuer*/ true,
633 j_);
634 JLOG(j_.trace()) << "DirectStepI::rev: Limiting"
635 << " srcRedeems: " << redeems(srcDebtDir) << " in: " << to_string(actualIn)
636 << " srcToDst: " << to_string(srcToDst) << " out: " << to_string(out);
637 }
638 return {cache_->in, cache_->out};
639}

641template <class TDerived>
642std::pair<bool, EitherAmount>
643DirectStepI<TDerived>::validFwd(PaymentSandbox& sb, ApplyView& afView, EitherAmount const& in)
644{
645 if (!cache_)
646 {
647 JLOG(j_.trace()) << "Expected valid cache in validFwd";
648 return {false, EitherAmount(IOUAmount(beast::zero))};
649 }
650
651 auto const savCache = *cache_;
652
653 XRPL_ASSERT(!in.native, "xrpl::DirectStepI::validFwd : input is not XRP");
654
655 auto const [maxSrcToDst, srcDebtDir] =
656 static_cast<TDerived const*>(this)->maxFlow(sb, cache_->srcToDst);
657 (void)srcDebtDir;
658
659 try
660 {
661 boost::container::flat_set<uint256> dummy;
662 fwdImp(sb, afView, dummy, in.iou); // changes cache
663 }
664 catch (FlowException const&)
665 {
666 return {false, EitherAmount(IOUAmount(beast::zero))};
667 }
668
669 if (maxSrcToDst < cache_->srcToDst)
670 {
671 JLOG(j_.warn()) << "DirectStepI: Strand re-execute check failed."
672 << " Exceeded max src->dst limit"
673 << " max src->dst: " << to_string(maxSrcToDst)
674 << " actual src->dst: " << to_string(cache_->srcToDst);
675 return {false, EitherAmount(cache_->out)};
676 }
677
678 if (!(checkNear(savCache.in, cache_->in) && checkNear(savCache.out, cache_->out)))
679 {
680 JLOG(j_.warn()) << "DirectStepI: Strand re-execute check failed."
681 << " ExpectedIn: " << to_string(savCache.in)
682 << " CachedIn: " << to_string(cache_->in)
683 << " ExpectedOut: " << to_string(savCache.out)
684 << " CachedOut: " << to_string(cache_->out);
685 return {false, EitherAmount(cache_->out)};
686 }
687 return {true, EitherAmount(cache_->out)};
688}

690// Returns srcQOut, dstQIn
691template <class TDerived>
692std::pair<std::uint32_t, std::uint32_t>
693DirectStepI<TDerived>::qualitiesSrcRedeems(ReadView const& sb) const
694{
695 if (!prevStep_)
696 return {QUALITY_ONE, QUALITY_ONE};
697
698 auto const prevStepQIn = prevStep_->lineQualityIn(sb);
699 auto srcQOut = static_cast<TDerived const*>(this)->quality(sb, QualityDirection::out);
700
701 if (prevStepQIn > srcQOut)
702 srcQOut = prevStepQIn;
703 return {srcQOut, QUALITY_ONE};
704}

707template <class TDerived>
708std::pair<std::uint32_t, std::uint32_t>
709DirectStepI<TDerived>::qualitiesSrcIssues(ReadView const& sb, DebtDirection prevStepDebtDirection)
710 const
711{
712 // Charge a transfer rate when issuing and previous step redeems
713
714 XRPL_ASSERT(
715 static_cast<TDerived const*>(this)->verifyPrevStepDebtDirection(prevStepDebtDirection),
716 "xrpl::DirectStepI::qualitiesSrcIssues : will prevStepDebtDirection "
717 "issue");
718
719 std::uint32_t const srcQOut =
720 redeems(prevStepDebtDirection) ? transferRate(sb, src_).value : QUALITY_ONE;
721 auto dstQIn = static_cast<TDerived const*>(this)->quality(sb, QualityDirection::in);
722
723 if (isLast_ && dstQIn > QUALITY_ONE)
724 dstQIn = QUALITY_ONE;
725 return {srcQOut, dstQIn};
726}

728// Returns srcQOut, dstQIn
729template <class TDerived>
730std::pair<std::uint32_t, std::uint32_t>
731DirectStepI<TDerived>::qualities(
732 ReadView const& sb,
733 DebtDirection srcDebtDir,
734 StrandDirection strandDir) const
735{
736 if (redeems(srcDebtDir))
737 {
738 return qualitiesSrcRedeems(sb);
739 }
740 else
741 {
742 auto const prevStepDebtDirection = [&] {
743 if (prevStep_)
744 return prevStep_->debtDirection(sb, strandDir);
745 return DebtDirection::issues;
746 }();
747 return qualitiesSrcIssues(sb, prevStepDebtDirection);
748 }
749}

759template <class TDerived>
760std::pair<std::optional<Quality>, DebtDirection>
761DirectStepI<TDerived>::qualityUpperBound(ReadView const& v, DebtDirection prevStepDir) const
762{
763 auto const dir = this->debtDirection(v, StrandDirection::forward);
764
765 auto const [srcQOut, dstQIn] =
766 redeems(dir) ? qualitiesSrcRedeems(v) : qualitiesSrcIssues(v, prevStepDir);
767
768 Issue const iss{currency_, src_};
769 // Be careful not to switch the parameters to `getRate`. The
770 // `getRate(offerOut, offerIn)` function is usually used for offers. It
771 // returns offerIn/offerOut. For a direct step, the rate is srcQOut/dstQIn
772 // (Input*dstQIn/srcQOut = Output; So rate = srcQOut/dstQIn). Although the
773 // first parameter is called `offerOut`, it should take the `dstQIn`
774 // variable.
775 return {Quality(getRate(STAmount(iss, dstQIn), STAmount(iss, srcQOut))), dir};
776}

778template <class TDerived>
779TER
780DirectStepI<TDerived>::check(StrandContext const& ctx) const
781{
782 // The following checks apply for both payments and offer crossing.
783 if (!src_ || !dst_)
784 {
785 JLOG(j_.debug()) << "DirectStepI: specified bad account.";
786 return temBAD_PATH;
787 }
788
789 if (src_ == dst_)
790 {
791 JLOG(j_.debug()) << "DirectStepI: same src and dst.";
792 return temBAD_PATH;
793 }
794
795 auto const sleSrc = ctx.view.read(keylet::account(src_));
796 if (!sleSrc)
797 {
798 JLOG(j_.warn()) << "DirectStepI: can't receive IOUs from non-existent issuer: " << src_;
799 return terNO_ACCOUNT;
800 }
801
802 // pure issue/redeem can't be frozen
803 if (!(ctx.isLast && ctx.isFirst))
804 {
805 auto const ter = checkFreeze(ctx.view, src_, dst_, currency_);
806 if (ter != tesSUCCESS)
807 return ter;
808 }
809
810 // If previous step was a direct step then we need to check
811 // no ripple flags.
812 if (ctx.prevStep)
813 {
814 if (auto prevSrc = ctx.prevStep->directStepSrcAcct())
815 {
816 auto const ter = checkNoRipple(ctx.view, *prevSrc, src_, dst_, currency_, j_);
817 if (ter != tesSUCCESS)
818 return ter;
819 }
820 }
821 {
822 Issue const srcIssue{currency_, src_};
823 Issue const dstIssue{currency_, dst_};
824
825 if (ctx.seenBookOuts.count(srcIssue))
826 {
827 if (!ctx.prevStep)
828 {
829 // LCOV_EXCL_START
830 UNREACHABLE(
831 "xrpl::DirectStepI::check : prev seen book without a "
832 "prev step");
833 return temBAD_PATH_LOOP;
834 // LCOV_EXCL_STOP
835 }
836
837 // This is OK if the previous step is a book step that outputs this
838 // issue
839 if (auto book = ctx.prevStep->bookStepBook())
840 {
841 if (book->out != srcIssue)
842 return temBAD_PATH_LOOP;
843 }
844 }
845
846 if (!ctx.seenDirectIssues[0].insert(srcIssue).second ||
847 !ctx.seenDirectIssues[1].insert(dstIssue).second)
848 {
849 JLOG(j_.debug()) << "DirectStepI: loop detected: Index: " << ctx.strandSize << ' '
850 << *this;
851 return temBAD_PATH_LOOP;
852 }
853 }
854
855 return static_cast<TDerived const*>(this)->check(ctx, sleSrc);
856}

880make_DirectStepI(
881 StrandContext const& ctx,
882 AccountID const& src,
883 AccountID const& dst,
884 Currency const& c)
885{
886 TER ter = tefINTERNAL;
887 std::unique_ptr<Step> r;
888 if (ctx.offerCrossing)
889 {
890 auto offerCrossingStep = std::make_unique<DirectIOfferCrossingStep>(ctx, src, dst, c);
891 ter = offerCrossingStep->check(ctx);
892 r = std::move(offerCrossingStep);
893 }
894 else // payment
895 {
896 auto paymentStep = std::make_unique<DirectIPaymentStep>(ctx, src, dst, c);
897 ter = paymentStep->check(ctx);
898 r = std::move(paymentStep);
899 }
900 if (ter != tesSUCCESS)
901 return {ter, nullptr};
902
903 return {tesSUCCESS, std::move(r)};
