16protocol::TMProofPathResponse
17LedgerReplayMsgHandler::processProofPathRequest(
18 std::shared_ptr<protocol::TMProofPathRequest> const& msg)
19{
20 protocol::TMProofPathRequest& packet = *msg;
21 protocol::TMProofPathResponse reply;
22
23 if (!packet.has_key() || !packet.has_ledgerhash() || !packet.has_type() ||
24 packet.ledgerhash().size() != uint256::size() || packet.key().size() != uint256::size() ||
25 !protocol::TMLedgerMapType_IsValid(packet.type()))
26 {
27 JLOG(journal_.debug()) << "getProofPath: Invalid request";
28 reply.set_error(protocol::TMReplyError::reBAD_REQUEST);
29 return reply;
30 }
31 reply.set_key(packet.key());
32 reply.set_ledgerhash(packet.ledgerhash());
33 reply.set_type(packet.type());
34
35 uint256 const key(packet.key());
36 uint256 const ledgerHash(packet.ledgerhash());
37 auto ledger = app_.getLedgerMaster().getLedgerByHash(ledgerHash);
38 if (!ledger)
39 {
40 JLOG(journal_.debug()) << "getProofPath: Don't have ledger " << ledgerHash;
41 reply.set_error(protocol::TMReplyError::reNO_LEDGER);
42 return reply;
43 }
44
45 auto const path = [&]() -> std::optional<std::vector<Blob>> {
46 switch (packet.type())
47 {
48 case protocol::lmACCOUNT_STATE:
49 return ledger->stateMap().getProofPath(key);
50 case protocol::lmTRANSACTION:
51 return ledger->txMap().getProofPath(key);
52 default:
53 // should not be here
54 // because already tested with TMLedgerMapType_IsValid()
55 return {};
56 }
57 }();
58
59 if (!path)
60 {
61 JLOG(journal_.debug()) << "getProofPath: Don't have the node " << key << " of ledger "
62 << ledgerHash;
63 reply.set_error(protocol::TMReplyError::reNO_NODE);
64 return reply;
65 }
66
67 // pack header
68 Serializer nData(128);
69 addRaw(ledger->header(), nData);
70 reply.set_ledgerheader(nData.getDataPtr(), nData.getLength());
71 // pack path
72 for (auto const& b : *path)
73 reply.add_path(b.data(), b.size());
74
75 JLOG(journal_.debug()) << "getProofPath for the node " << key << " of ledger " << ledgerHash
76 << " path length " << path->size();
77 return reply;
78}

80bool
81LedgerReplayMsgHandler::processProofPathResponse(
82 std::shared_ptr<protocol::TMProofPathResponse> const& msg)
83{
84 protocol::TMProofPathResponse& reply = *msg;
85 if (reply.has_error() || !reply.has_key() || !reply.has_ledgerhash() || !reply.has_type() ||
86 !reply.has_ledgerheader() || reply.path_size() == 0)
87 {
88 JLOG(journal_.debug()) << "Bad message: Error reply";
89 return false;
90 }
91
92 if (reply.type() != protocol::lmACCOUNT_STATE)
93 {
94 JLOG(journal_.debug()) << "Bad message: we only support the state ShaMap for now";
95 return false;
96 }
97
98 // deserialize the header
99 auto info = deserializeHeader({reply.ledgerheader().data(), reply.ledgerheader().size()});
100 uint256 replyHash(reply.ledgerhash());
101 if (calculateLedgerHash(info) != replyHash)
102 {
103 JLOG(journal_.debug()) << "Bad message: Hash mismatch";
104 return false;
105 }
106 info.hash = replyHash;
107
108 uint256 key(reply.key());
109 if (key != keylet::skip().key)
110 {
111 JLOG(journal_.debug()) << "Bad message: we only support the short skip list for now. "
112 "Key in reply "
113 << key;
114 return false;
115 }
116
117 // verify the skip list
118 std::vector<Blob> path;
119 path.reserve(reply.path_size());
120 for (int i = 0; i < reply.path_size(); ++i)
121 {
122 path.emplace_back(reply.path(i).begin(), reply.path(i).end());
123 }
124
125 if (!SHAMap::verifyProofPath(info.accountHash, key, path))
126 {
127 JLOG(journal_.debug()) << "Bad message: Proof path verify failed";
128 return false;
129 }
130
131 // deserialize the SHAMapItem
132 auto node = SHAMapTreeNode::makeFromWire(makeSlice(path.front()));
133 if (!node || !node->isLeaf())
134 {
135 JLOG(journal_.debug()) << "Bad message: Cannot deserialize";
136 return false;
137 }
138
139 if (auto item = static_cast<SHAMapLeafNode*>(node.get())->peekItem())
140 {
141 replayer_.gotSkipList(info, item);
142 return true;
143 }
144
145 JLOG(journal_.debug()) << "Bad message: Cannot get ShaMapItem";
146 return false;
147}

149protocol::TMReplayDeltaResponse
150LedgerReplayMsgHandler::processReplayDeltaRequest(
151 std::shared_ptr<protocol::TMReplayDeltaRequest> const& msg)
152{
153 protocol::TMReplayDeltaRequest& packet = *msg;
154 protocol::TMReplayDeltaResponse reply;
155
156 if (!packet.has_ledgerhash() || packet.ledgerhash().size() != uint256::size())
157 {
158 JLOG(journal_.debug()) << "getReplayDelta: Invalid request";
159 reply.set_error(protocol::TMReplyError::reBAD_REQUEST);
160 return reply;
161 }
162 reply.set_ledgerhash(packet.ledgerhash());
163
164 uint256 const ledgerHash{packet.ledgerhash()};
165 auto ledger = app_.getLedgerMaster().getLedgerByHash(ledgerHash);
166 if (!ledger || !ledger->isImmutable())
167 {
168 JLOG(journal_.debug()) << "getReplayDelta: Don't have ledger " << ledgerHash;
169 reply.set_error(protocol::TMReplyError::reNO_LEDGER);
170 return reply;
171 }
172
173 // pack header
174 Serializer nData(128);
175 addRaw(ledger->header(), nData);
176 reply.set_ledgerheader(nData.getDataPtr(), nData.getLength());
177 // pack transactions
178 auto const& txMap = ledger->txMap();
179 txMap.visitLeaves([&](boost::intrusive_ptr<SHAMapItem const> const& txNode) {
180 reply.add_transaction(txNode->data(), txNode->size());
181 });
182
183 JLOG(journal_.debug()) << "getReplayDelta for ledger " << ledgerHash << " txMap hash "
184 << txMap.getHash().as_uint256();
185 return reply;
186}

188bool
189LedgerReplayMsgHandler::processReplayDeltaResponse(
190 std::shared_ptr<protocol::TMReplayDeltaResponse> const& msg)
191{
192 protocol::TMReplayDeltaResponse& reply = *msg;
193 if (reply.has_error() || !reply.has_ledgerheader())
194 {
195 JLOG(journal_.debug()) << "Bad message: Error reply";
196 return false;
197 }
198
199 auto info = deserializeHeader({reply.ledgerheader().data(), reply.ledgerheader().size()});
200 uint256 replyHash(reply.ledgerhash());
201 if (calculateLedgerHash(info) != replyHash)
202 {
203 JLOG(journal_.debug()) << "Bad message: Hash mismatch";
204 return false;
205 }
206 info.hash = replyHash;
207
208 auto numTxns = reply.transaction_size();
209 std::map<std::uint32_t, std::shared_ptr<STTx const>> orderedTxns;
210 SHAMap txMap(SHAMapType::TRANSACTION, app_.getNodeFamily());
211 try
212 {
213 for (int i = 0; i < numTxns; ++i)
214 {
215 // deserialize:
216 // -- TxShaMapItem for building a ShaMap for verification
217 // -- Tx
218 // -- TxMetaData for Tx ordering
219 Serializer shaMapItemData(reply.transaction(i).data(), reply.transaction(i).size());
220
221 SerialIter txMetaSit(makeSlice(reply.transaction(i)));
222 SerialIter txSit(txMetaSit.getSlice(txMetaSit.getVLDataLength()));
223 SerialIter metaSit(txMetaSit.getSlice(txMetaSit.getVLDataLength()));
224
225 auto tx = std::make_shared<STTx const>(txSit);
226 if (!tx)
227 {
228 JLOG(journal_.debug()) << "Bad message: Cannot deserialize";
229 return false;
230 }
231 auto tid = tx->getTransactionID();
232 STObject meta(metaSit, sfMetadata);
233 orderedTxns.emplace(meta[sfTransactionIndex], std::move(tx));
234
235 if (!txMap.addGiveItem(
236 SHAMapNodeType::tnTRANSACTION_MD, make_shamapitem(tid, shaMapItemData.slice())))
237 {
238 JLOG(journal_.debug()) << "Bad message: Cannot deserialize";
239 return false;
240 }
241 }
242 }
243 catch (std::exception const&)
244 {
245 JLOG(journal_.debug()) << "Bad message: Cannot deserialize";
246 return false;
247 }
248
249 if (txMap.getHash().as_uint256() != info.txHash)
250 {
251 JLOG(journal_.debug()) << "Bad message: Transactions verify failed";
252 return false;
253 }
254
255 replayer_.gotReplayDelta(info, std::move(orderedTxns));
256 return true;
257}
