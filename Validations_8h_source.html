129inline bool
130isCurrent(
131 ValidationParms const& p,
132 NetClock::time_point now,
133 NetClock::time_point signTime,
134 NetClock::time_point seenTime)
135{
136 // Because this can be called on untrusted, possibly
137 // malicious validations, we do our math in a way
138 // that avoids any chance of overflowing or underflowing
139 // the signing time. All of the expressions below are
140 // promoted from unsigned 32 bit to signed 64 bit prior
141 // to computation.
142
143 return (signTime > (now - p.validationCURRENT_EARLY)) && (signTime < (now + p.validationCURRENT_WALL)) &&
144 ((seenTime == NetClock::time_point{}) || (seenTime < (now + p.validationCURRENT_LOCAL)));
145}

391 void
392 updateTrie(
393 std::lock_guard<Mutex> const& lock,
394 NodeID const& nodeID,
395 Validation const& val,
396 std::optional<std::pair<Seq, ID>> prior)
397 {
398 XRPL_ASSERT(val.trusted(), "xrpl::Validations::updateTrie : trusted input validation");
399
400 // Clear any prior acquiring ledger for this node
401 if (prior)
402 {
403 auto it = acquiring_.find(*prior);
404 if (it != acquiring_.end())
405 {
406 it->second.erase(nodeID);
407 if (it->second.empty())
408 acquiring_.erase(it);
409 }
410 }
411
412 checkAcquired(lock);
413
414 std::pair<Seq, ID> valPair{val.seq(), val.ledgerID()};
415 auto it = acquiring_.find(valPair);
416 if (it != acquiring_.end())
417 {
418 it->second.insert(nodeID);
419 }
420 else
421 {
422 if (std::optional<Ledger> ledger = adaptor_.acquire(val.ledgerID()))
423 updateTrie(lock, nodeID, *ledger);
424 else
425 acquiring_[valPair].insert(nodeID);
426 }

467 template <class Pre, class F>
468 void
469 current(std::lock_guard<Mutex> const& lock, Pre&& pre, F&& f)
470 {
471 NetClock::time_point t = adaptor_.now();
472 pre(current_.size());
473 auto it = current_.begin();
474 while (it != current_.end())
475 {
476 // Check for staleness
477 if (!isCurrent(parms_, t, it->second.signTime(), it->second.seenTime()))
478 {
479 removeTrie(lock, it->first, it->second);
480 it = current_.erase(it);
481 }
482 else
483 {
484 auto cit = typename decltype(current_)::const_iterator{it};
485 // contains a live record
486 f(cit->first, cit->second);
487 ++it;
488 }
489 }

570 ValStatus
571 add(NodeID const& nodeID, Validation const& val)
572 {
573 if (!isCurrent(parms_, adaptor_.now(), val.signTime(), val.seenTime()))
574 return ValStatus::stale;
575
576 {
577 std::lock_guard lock{mutex_};
578
579 // Check that validation sequence is greater than any non-expired
580 // validations sequence from that validator; if it's not, perform
581 // additional work to detect Byzantine validations
582 auto const now = byLedger_.clock().now();
583
584 auto const [seqit, seqinserted] = bySequence_[val.seq()].emplace(nodeID, val);
585
586 if (!seqinserted)
587 {
588 // Check if the entry we're already tracking was signed
589 // long enough ago that we can disregard it.
590 auto const diff = std::max(seqit->second.signTime(), val.signTime()) -
591 std::min(seqit->second.signTime(), val.signTime());
592
593 if (diff > parms_.validationCURRENT_WALL && val.signTime() > seqit->second.signTime())
594 seqit->second = val;
595 }
596
597 // Enforce monotonically increasing sequences for validations
598 // by a given node, and run the active Byzantine detector:
599 if (auto& enf = seqEnforcers_[nodeID]; !enf(now, val.seq(), parms_))
600 {
601 // If the validation is for the same sequence as one we are
602 // tracking, check it closely:
603 if (seqit->second.seq() == val.seq())
604 {
605 // Two validations for the same sequence but for different
606 // ledgers. This could be the result of misconfiguration
607 // but it can also mean a Byzantine validator.
608 if (seqit->second.ledgerID() != val.ledgerID())
609 return ValStatus::conflicting;
610
611 // Two validations for the same sequence and for the same
612 // ledger with different sign times. This could be the
613 // result of a misconfiguration but it can also mean a
614 // Byzantine validator.
615 if (seqit->second.signTime() != val.signTime())
616 return ValStatus::conflicting;
617
618 // Two validations for the same sequence but with different
619 // cookies. This is probably accidental misconfiguration.
620 if (seqit->second.cookie() != val.cookie())
621 return ValStatus::multiple;
622 }
623
624 return ValStatus::badSeq;
625 }
626
627 byLedger_[val.ledgerID()].insert_or_assign(nodeID, val);
628
629 auto const [it, inserted] = current_.emplace(nodeID, val);
630 if (!inserted)
631 {
632 // Replace existing only if this one is newer
633 Validation& oldVal = it->second;
634 if (val.signTime() > oldVal.signTime())
635 {
636 std::pair<Seq, ID> old(oldVal.seq(), oldVal.ledgerID());
637 it->second = val;
638 if (val.trusted())
639 updateTrie(lock, nodeID, val, old);
640 }
641 else
642 return ValStatus::stale;
643 }
644 else if (val.trusted())
645 {
646 updateTrie(lock, nodeID, val, std::nullopt);
647 }
648 }
649
650 return ValStatus::current;
651 }

672 void
673 expire(beast::Journal& j)
674 {
675 auto const start = std::chrono::steady_clock::now();
676 {
677 std::lock_guard lock{mutex_};
678 if (toKeep_)
679 {
680 // We only need to refresh the keep range when it's just about
681 // to expire. Track the next time we need to refresh.
682 static std::chrono::steady_clock::time_point refreshTime;
683 if (auto const now = byLedger_.clock().now(); refreshTime <= now)
684 {
685 // The next refresh time is shortly before the expiration
686 // time from now.
687 refreshTime = now + parms_.validationSET_EXPIRES - parms_.validationFRESHNESS;
688
689 for (auto i = byLedger_.begin(); i != byLedger_.end(); ++i)
690 {
691 auto const& validationMap = i->second;
692 if (!validationMap.empty())
693 {
694 auto const seq = validationMap.begin()->second.seq();
695 if (toKeep_->low_ <= seq && seq < toKeep_->high_)
696 {
697 byLedger_.touch(i);
698 }
699 }
700 }
701
702 for (auto i = bySequence_.begin(); i != bySequence_.end(); ++i)
703 {
704 if (toKeep_->low_ <= i->first && i->first < toKeep_->high_)
705 {
706 bySequence_.touch(i);
707 }
708 }
709 }
710 }
711
712 beast::expire(byLedger_, parms_.validationSET_EXPIRES);
713 beast::expire(bySequence_, parms_.validationSET_EXPIRES);
714 }
715 JLOG(j.debug())
716 << "Validations sets sweep lock duration "
717 << std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - start).count()
718 << "ms";
719 }

730 void
731 trustChanged(hash_set<NodeID> const& added, hash_set<NodeID> const& removed)
732 {
733 std::lock_guard lock{mutex_};
734
735 for (auto& [nodeId, validation] : current_)
736 {
737 if (added.find(nodeId) != added.end())
738 {
739 validation.setTrusted();
740 updateTrie(lock, nodeId, validation, std::nullopt);
741 }
742 else if (removed.find(nodeId) != removed.end())
743 {
744 validation.setUntrusted();
745 removeTrie(lock, nodeId, validation);
746 }
747 }
748
749 for (auto& [_, validationMap] : byLedger_)
750 {
751 (void)_;
752 for (auto& [nodeId, validation] : validationMap)
753 {
754 if (added.find(nodeId) != added.end())
755 {
756 validation.setTrusted();
757 }
758 else if (removed.find(nodeId) != removed.end())
759 {
760 validation.setUntrusted();
761 }
762 }
763 }

785 std::optional<std::pair<Seq, ID>>
786 getPreferred(Ledger const& curr)
787 {
788 std::lock_guard lock{mutex_};
789 std::optional<SpanTip<Ledger>> preferred =
790 withTrie(lock, [this](LedgerTrie<Ledger>& trie) { return trie.getPreferred(localSeqEnforcer_.largest()); });
791 // No trusted validations to determine branch
792 if (!preferred)
793 {
794 // fall back to majority over acquiring ledgers
795 auto it = std::max_element(acquiring_.begin(), acquiring_.end(), [](auto const& a, auto const& b) {
796 std::pair<Seq, ID> const& aKey = a.first;
797 typename hash_set<NodeID>::size_type const& aSize = a.second.size();
798 std::pair<Seq, ID> const& bKey = b.first;
799 typename hash_set<NodeID>::size_type const& bSize = b.second.size();
800 // order by number of trusted peers validating that ledger
801 // break ties with ledger ID
802 return std::tie(aSize, aKey.second) < std::tie(bSize, bKey.second);
803 });
804 if (it != acquiring_.end())
805 return it->first;
806 return std::nullopt;
807 }
808
809 // If we are the parent of the preferred ledger, stick with our
810 // current ledger since we might be about to generate it
811 if (preferred->seq == curr.seq() + Seq{1} && preferred->ancestor(curr.seq()) == curr.id())
812 return std::make_pair(curr.seq(), curr.id());
813
814 // A ledger ahead of us is preferred regardless of whether it is
815 // a descendant of our working ledger or it is on a different chain
816 if (preferred->seq > curr.seq())
817 return std::make_pair(preferred->seq, preferred->id);
818
819 // Only switch to earlier or same sequence number
820 // if it is a different chain.
821 if (curr[preferred->seq] != preferred->id)
822 return std::make_pair(preferred->seq, preferred->id);
823
824 // Stick with current ledger
825 return std::make_pair(curr.seq(), curr.id());
826 }

862 ID
