91 std::mutex mutex_;
92 std::condition_variable cond_;
93 std::map<Child*, std::weak_ptr<Child>> list_;
94 bool closed_ = false;
95
96 public:
97 ~Base()
98 {
99 // Derived class must call wait() in the destructor
100 assert(list_.empty());
101 }
102
103 void
104 add(std::shared_ptr<Child> const& child)
105 {
106 std::lock_guard lock(mutex_);
107 list_.emplace(child.get(), child);
108 }
109
110 void
111 remove(Child* child)
112 {
113 std::lock_guard lock(mutex_);
114 list_.erase(child);
115 if (list_.empty())
116 cond_.notify_one();
117 }
118
119 void
120 close()
121 {
122 std::vector<std::shared_ptr<Child>> v;
123 {
124 std::lock_guard lock(mutex_);
125 v.reserve(list_.size());
126 if (closed_)
127 return;
128 closed_ = true;
129 for (auto const& c : list_)
130 {
131 if (auto p = c.second.lock())
132 {
133 p->close();
134 // Must destroy shared_ptr outside the
135 // lock otherwise deadlock from the
136 // managed object's destructor.
137 v.emplace_back(std::move(p));
138 }
139 }
140 }
141 }
142
143 void
144 wait()
145 {
146 std::unique_lock<std::mutex> lock(mutex_);
147 while (!list_.empty())
148 cond_.wait(lock);
149 }
